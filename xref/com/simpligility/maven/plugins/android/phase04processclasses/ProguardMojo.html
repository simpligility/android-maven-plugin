58       * ProGuard configuration. ProGuard is disabled by default. Set the skip parameter to false to activate proguard.
59       * A complete configuartion can include any of the following:
60       * </p>
61       * <p/>
62       * <pre>
63       * &lt;proguard&gt;
64       *    &lt;skip&gt;true|false&lt;/skip&gt;
65       *    &lt;config&gt;proguard.cfg&lt;/config&gt;
66       *    &lt;configs&gt;
67       *      &lt;config&gt;${env.ANDROID_HOME}/tools/proguard/proguard-android.txt&lt;/config&gt;
68       *    &lt;/configs&gt;
69       *    &lt;proguardJarPath&gt;someAbsolutePathToProguardJar&lt;/proguardJarPath&gt;
70       *    &lt;filterMavenDescriptor&gt;true|false&lt;/filterMavenDescriptor&gt;
71       *    &lt;filterManifest&gt;true|false&lt;/filterManifest&gt;
72       *    &lt;customFilter&gt;filter1,filter2&lt;/customFilter&gt;
73       *    &lt;jvmArguments&gt;
74       *     &lt;jvmArgument&gt;-Xms256m&lt;/jvmArgument&gt;

322     private static class ProGuardInput
323     {
324 
325         private String path;
326 
327         private Collection< String > excludedFilter;
328 
329         ProGuardInput( String path, Collection< String > excludedFilter )
330         {
331             this.path = path;
332             this.excludedFilter = excludedFilter;
333         }
334 
335         public String toPath()
336         {
337             if ( excludedFilter != null && !excludedFilter.isEmpty() )
338             {
339                 String middleQuote, endQuote;
340 
341                 if ( !Os.isFamily( Os.FAMILY_WINDOWS ) )
342                 {
343                     middleQuote = "(";
344                     endQuote = ")";
345                 }
346                 else
347                 {
348                     middleQuote = "(";
349                     endQuote = ")";
350                 }
351 
352                 StringBuilder sb = new StringBuilder();
353                 sb.append( path );
354                 sb.append( middleQuote );
355                 for ( Iterator< String > it = excludedFilter.iterator(); it.hasNext(); )
356                 {
357                     sb.append( '!' ).append( it.next() );
358                     if ( it.hasNext() )
359                     {
360                         sb.append( ',' );
361                     }
362                 }
363                 sb.append( endQuote );
364                 return sb.toString();
365             }
366             else
367             {
368                 return path;
369             }
370         }
371 
372         @Override
373         public String toString()
374         {
375             return "ProGuardInput{"
376                     + "path='" + path + '\''
377                     + ", excludedFilter=" + excludedFilter
378                     + '}';
379         }
380     }

382     @Override
383     public void execute() throws MojoExecutionException, MojoFailureException
384     {
385         ConfigHandler configHandler = new ConfigHandler( this, this.session, this.execution );
386         configHandler.parseConfiguration();
387 
388         if ( !parsedSkip )
389         {
390             if ( parsedConfig.exists() )
391             {
392                 // TODO: make the property name a constant sometime after switching to @Mojo
393                 project.getProperties().setProperty( "android.proguard.obfuscatedJar", obfuscatedJar );
394 
395                 executeProguard();
396             }
397             else
398             {
399                 getLog().info( String
400                     .format( "Proguard skipped because the configuration file doesn't exist: %s", parsedConfig ) );
401             }
402         }
403     }

405     private void executeProguard() throws MojoExecutionException
406     {
407         final File proguardDir = this.parsedOutputDirectory;
408 
409         if ( !proguardDir.exists() && !proguardDir.mkdir() )
410         {
411             throw new MojoExecutionException( "Cannot create proguard output directory" );
412         }
413         else
414         {
415             if ( proguardDir.exists() && !proguardDir.isDirectory() )
416             {
417                 throw new MojoExecutionException( "Non-directory exists at " + proguardDir.getAbsolutePath() );
418             }
419         }
420 
421         getLog().info( "Proguarding output" );
422         CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
423         executor.setLogger( this.getLog() );
424         List< String > commands = new ArrayList< String >();
425 
426         collectJvmArguments( commands );
427 
428         commands.add( "-jar" );
429         commands.add( parsedProguardJarPath );
430 
431         List<String> proguardCommands = new ArrayList<String>();
432 
433         proguardCommands.add( "@" + parsedConfig + "" );
434 
435         for ( String config : parsedConfigs )
436         {
437             proguardCommands.add( "@" + config );
438         }
439 
440         if ( proguardFile != null )
441         {
442             proguardCommands.add( "@" + proguardFile.getAbsolutePath() );
443         }
444 
445         for ( Artifact artifact : getTransitiveDependencyArtifacts( AAR ) )
446         {
447             File unpackedLibFolder = getUnpackedLibFolder( artifact );
448             File proguardFile = new File( unpackedLibFolder, "proguard.txt" );
449             if ( proguardFile.exists() )
450             {
451                 proguardCommands.add( "@" + proguardFile.getAbsolutePath() );
452             }
453         }
454 
455         collectInputFiles( proguardCommands );
456 
457         proguardCommands.add( "-outjars" );
458         proguardCommands.add( obfuscatedJar );
459 
460         proguardCommands.add( "-dump" );
461         proguardCommands.add( proguardDir + File.separator + "dump.txt" );
462         proguardCommands.add( "-printseeds" );
463         proguardCommands.add( proguardDir + File.separator + "seeds.txt" );
464         proguardCommands.add( "-printusage" );
465         proguardCommands.add( proguardDir + File.separator + "usage.txt" );
466 
467         File mapFile = new File( proguardDir, "mapping.txt" );
468 
469         proguardCommands.add( "-printmapping" );
470         proguardCommands.add( mapFile.toString() );
471 
472         proguardCommands.addAll( Arrays.asList( parsedOptions ) );
473 
474         final String javaExecutable = getJavaExecutable().getAbsolutePath();
475 
476         getLog().debug( javaExecutable + " " + commands.toString() + proguardCommands.toString() );
477 
478         FileOutputStream tempConfigFileOutputStream = null;
479         try
480         {
481             File tempConfigFile = new File ( proguardDir , "temp_config.cfg" );
482 
483             StringBuilder commandStringBuilder = new StringBuilder();
484             for ( String command : proguardCommands )
485             {
486                 commandStringBuilder.append( command );
487                 commandStringBuilder.append( SystemUtils.LINE_SEPARATOR );
488             }
489             tempConfigFileOutputStream = new FileOutputStream( tempConfigFile );
490             IOUtils.write( commandStringBuilder, tempConfigFileOutputStream );
491 
492             executor.setCaptureStdOut( true );
493             commands.add( "@" + tempConfigFile.getAbsolutePath() + "" );
494             executor.executeCommand( javaExecutable, commands, project.getBasedir(), false );
495         }
496         catch ( ExecutionException e )
497         {
498             throw new MojoExecutionException( "", e );
499         }
500         catch ( IOException e )
501         {
502             throw new MojoExecutionException( "Error writing proguard commands to temporary file", e );
503         }
504         finally
505         {
506             IOUtils.closeQuietly( tempConfigFileOutputStream );
507         }
508 
509         if ( parsedAttachMap )
510         {
511             projectHelper.attachArtifact( project, "map", mapFile );
512         }
513     }

515     /**
516      * Convert the jvm arguments in parsedJvmArguments as populated by the config in format as needed by the java
517      * command. Also preserve backwards compatibility in terms of dashes required or not..
518      */
519     private void collectJvmArguments( List< String > commands )
520     {
521         if ( parsedJvmArguments != null )
522         {
523             for ( String jvmArgument : parsedJvmArguments )
524             {
525                 // preserve backward compatibility allowing argument with or without dash (e.g.
526                 // Xmx512m as well as -Xmx512m should work) (see
527                 // http://code.google.com/p/maven-android-plugin/issues/detail?id=153)
528                 if ( !jvmArgument.startsWith( "-" ) )
529                 {
530                     jvmArgument = "-" + jvmArgument;
531                 }
532                 commands.add( jvmArgument );
533             }
534         }
535     }

537     private void collectInputFiles( List< String > commands ) throws MojoExecutionException
538     {
539         // commons-logging breaks everything horribly, so we skip it from the program
540         // dependencies and declare it to be a library dependency instead
541         skipArtifact( "commons-logging", "commons-logging", true );
542 
543         final List< ProGuardInput > inJars = getProgramInputFiles();
544         if ( isAPKBuild() )
545         {
546             inJars.addAll( getProjectDependencyFiles() );
547         }
548 
549         for ( final ProGuardInput injar : inJars )
550         {
551             getLog().debug( "Added injar : " + injar );
552             commands.add( "-injars" );
553             commands.add( injar.toPath() );
554         }
555 
556         final List< ProGuardInput > libraryJars = getLibraryInputFiles();
557         if ( !isAPKBuild() )
558         {
559             getLog().info( "Library project - not adding project dependencies to the obfuscated JAR" );
560             libraryJars.addAll( getProjectDependencyFiles() );
561         }
562 
563         for ( final ProGuardInput libraryjar : libraryJars )
564         {
565             getLog().debug( "Added libraryJar : " + libraryjar );
566             commands.add( "-libraryjars" );
567             commands.add( libraryjar.toPath() );
568         }
569     }

583     private void skipArtifact( String groupId, String artifactId, boolean shiftToLibraries )
584             throws MojoExecutionException
585     {
586         final ArtifactPrototype artifact = new ArtifactPrototype( groupId, artifactId );
587         artifactBlacklist.add( artifact );
588         if ( shiftToLibraries )
589         {
590             artifactsToShift.add( artifact );
591         }

594     private boolean isBlacklistedArtifact( Artifact artifact )
595     {
596         for ( ArtifactPrototype artifactToSkip : artifactBlacklist )
597         {
598             if ( artifactToSkip.groupId.equals( artifact.getGroupId() )
599                     && artifactToSkip.artifactId.equals( artifact.getArtifactId() ) )
600             {
601                 return true;
602             }
603         }
604         return false;
605     }

607     private boolean isShiftedArtifact( Artifact artifact )
608     {
609         for ( ArtifactPrototype artifactToShift : artifactsToShift )
610         {
611             if ( artifactToShift.groupId.equals( artifact.getGroupId() )
612                     && artifactToShift.artifactId.equals( artifact.getArtifactId() ) )
613             {
614                 return true;
615             }
616         }
617         return false;
618     }

627     private List< ProGuardInput > getProjectDependencyFiles()
628     {
629         final Collection< String > globalInJarExcludes = new HashSet< String >();
630         final List< ProGuardInput > inJars = new LinkedList< ProguardMojo.ProGuardInput >();
631 
632         if ( parsedFilterManifest )
633         {
634             globalInJarExcludes.addAll( META_INF_MANIFEST );
635         }
636         if ( parsedFilterMavenDescriptor )
637         {
638             globalInJarExcludes.addAll( MAVEN_DESCRIPTOR );
639         }
640         if ( parsedCustomFilter != null )
641         {
642             globalInJarExcludes.addAll( Arrays.asList( parsedCustomFilter.split( "," ) ) );
643         }
644 
645         // we then add all its dependencies (incl. transitive ones), unless they're blacklisted
646         for ( Artifact artifact : filterArtifacts( getTransitiveDependencyArtifacts(), skipDependencies,
647                     artifactTypeSet.getIncludes(), artifactTypeSet.getExcludes(), artifactSet.getIncludes(),
648                     artifactSet.getExcludes() ) )
649         {
650             if ( isBlacklistedArtifact( artifact ) )
651             {
652                 getLog().debug( "Excluding (blacklisted) dependency as input jar : " + artifact );
653                 continue;
654             }
655 
656             if ( JAR_DEPENDENCY_TYPE.equals( artifact.getType() ) )
657             {
658                 getLog().debug( "Including dependency as input jar : " + artifact );
659                 inJars.add( createProguardInput( artifact.getFile().getAbsolutePath(), globalInJarExcludes ) );
660             }
661             else if ( AAR.equals( artifact.getType() ) )
662             {
663                 // The Aar classes.jar should now be automatically included
664                 // because it will be a transitive dependency. As should any jars in the libs folder.
665             }
666             else
667             {
668                 getLog().debug( "Excluding dependency as input jar : " + artifact );
669             }
670         }
671 
672         return inJars;
673     }

685     private List< ProGuardInput > getLibraryInputFiles()
686     {
687         final List< ProGuardInput > libraryJars = new LinkedList< ProguardMojo.ProGuardInput >();
688 
689         if ( parsedIncludeJdkLibs )
690         {
691             // we have to add the Java framework classes to the library JARs, since they are not
692             // distributed with the JAR on Central, and since we'll strip them out of the android.jar
693             // that is shipped with the SDK (since that is not a complete Java distribution)
694             File rtJar = getJVMLibrary( "rt.jar" );
695             if ( rtJar == null )
696             {
697                 rtJar = getJVMLibrary( "classes.jar" );
698             }
699             if ( rtJar != null )
700             {
701                 libraryJars.add( createProguardInput( rtJar.getPath() ) );
702             }
703 
704             // we also need to add the JAR containing e.g. javax.servlet
705             File jsseJar = getJVMLibrary( "jsse.jar" );
706             if ( jsseJar != null )
707             {
708                 libraryJars.add( createProguardInput( jsseJar.getPath() ) );
709             }
710 
711             // and the javax.crypto stuff
712             File jceJar = getJVMLibrary( "jce.jar" );
713             if ( jceJar != null )
714             {
715                 libraryJars.add( createProguardInput( jceJar.getPath() ) );
716             }
717         }
718 
719         // we treat any dependencies with provided scope as library JARs
720         for ( Artifact artifact : project.getArtifacts() )
721         {
722             if ( artifact.getScope().equals( Artifact.SCOPE_PROVIDED ) )
723             {
724                 if ( artifact.getArtifactId().equals( "android" ) && parsedIncludeJdkLibs )
725                 {
726                     getLog().debug( "Including dependency as (android) library jar : " + artifact );
727                     libraryJars.add(
728                             createProguardInput( artifact.getFile().getAbsolutePath(), ANDROID_LIBRARY_EXCLUDED_FILTER )
729                     );
730                 }
731                 else
732                 {
733                     getLog().debug( "Including dependency as (provided) library jar : " + artifact );
734                     libraryJars.add( createProguardInput( artifact.getFile().getAbsolutePath() ) );
735                 }
736             }
737             else
738             {
739                 if ( isShiftedArtifact( artifact ) )
740                 {
741                     // this is a blacklisted artifact that should be processed as a library instead
742                     getLog().debug( "Including dependency as (shifted) library jar : " + artifact );
743                     libraryJars.add( createProguardInput( artifact.getFile().getAbsolutePath() ) );
744                 }
745                 else
746                 {
747                     getLog().debug( "Excluding dependency as library jar : " + artifact );
748                 }
749             }
750         }
751 
752         return libraryJars;
753     }

755     /**
756      * Get the path to the proguard jar.
757      */
758     @SuppressWarnings( "unused" ) // NB Used to populate the parsedProguardJarPath attribute via reflection.
759     private String getProguardJarPath() throws MojoExecutionException
760     {
761         String proguardJarPath = getProguardJarPathFromDependencies();
762         if ( StringUtils.isEmpty( proguardJarPath ) )
763         {
764             File proguardJarPathFile = new File( getAndroidSdk().getToolsPath(), "proguard/lib/proguard.jar" );
765             return proguardJarPathFile.getAbsolutePath();
766         }
767         return proguardJarPath;
768     }

770     private String getProguardJarPathFromDependencies() throws MojoExecutionException
771     {
772         Artifact proguardArtifact = null;
773         int proguardArtifactDistance = -1;
774         for ( Artifact artifact : pluginDependencies )
775         {
776             getLog().debug( "pluginArtifact: " + artifact.getFile() );
777             if ( ( "proguard".equals( artifact.getArtifactId() ) ) || ( "proguard-base"
778                 .equals( artifact.getArtifactId() ) ) )
779             {
780                 int distance = artifact.getDependencyTrail().size();
781                 getLog().debug( "proguard DependencyTrail: " + distance );
782                 if ( proguardArtifactDistance == -1 )
783                 {
784                     proguardArtifact = artifact;
785                     proguardArtifactDistance = distance;
786                 }
787                 else
788                 {
789                     if ( distance < proguardArtifactDistance )
790                     {
791                         proguardArtifact = artifact;
792                         proguardArtifactDistance = distance;
793                     }
794                 }
795             }
796         }
797         if ( proguardArtifact != null )
798         {
799             getLog().debug( "proguardArtifact: " + proguardArtifact.getFile() );
800             return proguardArtifact.getFile().getAbsoluteFile().toString();
801         }
802         else
803         {
804             return null;
805         }

842     /**
843      * Finds a library file in either the primary or alternate lib directory.
844      * @param fileName The base name of the file.
845      * @return Either a canonical filename, or {@code null} if not found.
846      */
847     private File getJVMLibrary( String fileName )
848     {
849         File libFile = new File( getJavaLibDir(), fileName );
850         if ( !libFile.exists() )
851         {
852             libFile = new File( getAltJavaLibDir(), fileName );
853             if ( !libFile.exists() )
854             {
855                 libFile = null;
856             }
857         }
858         return libFile;
859     }
