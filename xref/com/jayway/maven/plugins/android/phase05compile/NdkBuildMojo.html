296     /**
297      * @throws MojoExecutionException
298      * @throws MojoFailureException
299      */
300     public void execute() throws MojoExecutionException, MojoFailureException
301     {
302         // Validate the NDK
303         final File ndkBuildFile = new File( getAndroidNdk().getNdkBuildPath() );
304         NativeHelper.validateNDKVersion( ndkBuildFile.getParentFile() );
305 
306         validateMakefile( project, makefile );
307 
308         final String[] resolvedNDKArchitectures = NativeHelper.getNdkArchitectures(
309             ndkArchitecture != null ? ndkArchitecture : ndkArchitectures, applicationMakefile,
310             project.getBasedir() );
311 
312         for ( String architecture : resolvedNDKArchitectures )
313         {
314             try
315             {
316                 compileForArchitecture( architecture );
317             }
318             catch ( IOException e )
319             {
320                 getLog().error( "Error while executing: " + e.getMessage() );
321                 throw new MojoExecutionException( e.getMessage(), e );
322             }
323             catch ( ExecutionException e )
324             {
325                 getLog().error( "Error while executing: " + e.getMessage() );
326                 throw new MojoExecutionException( e.getMessage(), e );
327             }
328         }
329     }

331     private void compileForArchitecture( String architecture )
332         throws MojoExecutionException, IOException, ExecutionException
333     {
334         getLog().debug( "Resolving for NDK architecture : " + architecture );
335 
336         // Start setting up the command line to be executed
337         final CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
338         // Add an error listener to the build - this allows the build to conditionally fail
339         // depending on a) the output of the build b) whether or not build errors (output on stderr) should be
340         // ignored and c) whether the pattern matches or not
341         executor.setErrorListener( getNdkErrorListener() );
342 
343         final Set<Artifact> nativeLibraryArtifacts = findNativeLibraryDependencies();
344 
345         // If there are any static libraries the code needs to link to, include those in the make file
346         final Set<Artifact> resolveNativeLibraryArtifacts =
347             getArtifactResolverHelper().resolveArtifacts( nativeLibraryArtifacts );
348 
349         getLog().debug( "resolveArtifacts found " + resolveNativeLibraryArtifacts.size()
350             + ": " + resolveNativeLibraryArtifacts.toString() );
351 
352         final File buildFolder = new File( ndkBuildDirectory, architecture );
353         buildFolder.mkdirs();
354 
355         final File androidMavenMakefile = new File( buildFolder, "android_maven_plugin_makefile.mk" );
356         final MakefileHelper makefileHelper = new MakefileHelper( getLog(),
357             getArtifactResolverHelper(),
358             harArtifactHandler, getUnpackedLibsDirectory()
359         );
360 
361         final MakefileHelper.MakefileHolder makefileHolder = makefileHelper
362             .createMakefileFromArtifacts( resolveNativeLibraryArtifacts, architecture, "armeabi", useHeaderArchives );
363 
364         final FileOutputStream output = new FileOutputStream( androidMavenMakefile );
365         try
366         {
367             IOUtil.copy( makefileHolder.getMakeFile(), output );
368         }
369         finally
370         {
371             output.close();
372         }
373 
374         // Add the path to the generated makefile - this is picked up by the build (by an include from the user)
375         executor.addEnvironment( "ANDROID_MAVEN_PLUGIN_MAKEFILE", androidMavenMakefile.getAbsolutePath() );
376 
377         setupNativeLibraryEnvironment( makefileHelper, executor, resolveNativeLibraryArtifacts,
378             architecture );
379 
380         // Adds the location of the Makefile capturer file - this file will after the build include
381         // things like header files, flags etc.  It is processed after the build to retrieve the headers
382         // and also capture flags etc ...
383         final File makefileCaptureFile = File.createTempFile( "android_maven_plugin_makefile_captures", ".tmp" );
384         makefileCaptureFile.deleteOnExit();
385         executor.addEnvironment( MakefileHelper.MAKEFILE_CAPTURE_FILE, makefileCaptureFile.getAbsolutePath() );
386 
387         // Add any defined system properties
388         if ( systemProperties != null && ! systemProperties.isEmpty() )
389         {
390             for ( Map.Entry<String, String> entry : systemProperties.entrySet() )
391             {
392                 executor.addEnvironment( entry.getKey(), entry.getValue() );
393             }
394         }
395         executor.setLogger( this.getLog() );
396         // Setup the command line for the make
397         final List<String> commands = new ArrayList<String>();
398         // Setup the build directory (defaults to the current directory) but may be different depending
399         // on user configuration
400         commands.add( "-C" );
401         commands.add( project.getBasedir().getAbsolutePath() );
402 
403         // If the build should use a custom makefile or not - some validation is done to ensure
404         // this exists and all
405         if ( makefile != null )
406         {
407             File makeFile = new File( project.getBasedir(), makefile );
408             if ( ! makeFile.exists() )
409             {
410                 getLog().error( "Specified makefile " + makeFile + " does not exist" );
411                 throw new MojoExecutionException( "Specified makefile " + makeFile + " does not exist" );
412             }
413             commands.add( "-f" );
414             commands.add( makefile );
415         }
416 
417         configureApplicationMakefile( commands );
418         configureMaxJobs( commands );
419         configureNdkToolchain( architecture, commands );
420         configureAdditionalCommands( commands );
421 
422         // If a build target is specified, tag that onto the command line as the very last of the parameters
423         if ( target != null )
424         {
425             commands.add( target );
426         }
427         else /*if ( Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )*/
428         {
429             commands.add( project.getArtifactId() );
430         }
431 
432         final String ndkBuildPath = resolveNdkBuildExecutable();
433         getLog().debug( ndkBuildPath + " " + commands.toString() );
434         getLog().info( "Executing NDK " + architecture + " make at : " + ndkBuildDirectory );
435 
436         executor.setCaptureStdOut( true );
437         executor.executeCommand( ndkBuildPath, commands, ndkBuildDirectory, true );
438         getLog().debug( "Executed NDK " + architecture + " make at : " + ndkBuildDirectory );
439 
440         // Where the NDK build creates the libs.
441         final File nativeLibOutputDirectory = new File( nativeLibrariesOutputDirectory, architecture );
442         nativeLibOutputDirectory.mkdirs();
443 
444         // Move the built native libs into the packaging folder.
445         // We don't create them there to start with because the NDK build seems determined to create them in the root.
446         final File destinationDirectory = new File( ndkOutputDirectory, architecture );
447         FileUtils.moveDirectory( nativeLibOutputDirectory, destinationDirectory );
448 
449         // Attempt to attach the native library if the project is defined as a "pure" native Android library
450         // (packaging is 'so' or 'a') or if the plugin has been configured to attach the native library to the build
451         if ( Const.ArtifactType.NATIVE_SYMBOL_OBJECT.equals( project.getPackaging() )
452             || Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() )
453             || attachNativeArtifacts )
454         {
455             attachNativeLib( destinationDirectory, architecture );
456         }
457 
458         // Process conditionally any of the headers to include into the header archive file
459         if ( attachHeaderFiles )
460         {
461             attachHeaderFiles( makefileCaptureFile, architecture );
462         }
463 
464         // If we created a makefile for the build we should be polite and remove any extracted include
465         // directories after we're done
466         getLog().info( "Cleaning up extracted include directories used for build" );
467         MakefileHelper.cleanupAfterBuild( makefileHolder );
468     }

507     private void configureNdkToolchain( String architecture, List<String> commands )
508             throws MojoExecutionException
509     {
510         if ( ndkToolchain != null )
511         {
512             // Setup the correct toolchain to use
513             // FIXME: perform a validation that this toolchain exists in the NDK
514             commands.add( "NDK_TOOLCHAIN=" + ndkToolchain );
515         }
516         else
517         {
518             // Resolve the toolchain from the architecture
519             // <ndkArchitectures>
520             //   <x86>x86-4.6</x86>
521             //   <armeabi>x86-4.6</armeabi>
522             // </ndkArchitectures>
523             final String toolchainFromArchitecture = getAndroidNdk().getToolchainFromArchitecture(
524                     architecture, ndkArchitectureToolchainMappings );
525             getLog().debug( "Resolved toolchain for " + architecture + " to " + toolchainFromArchitecture );
526             commands.add( "NDK_TOOLCHAIN=" + toolchainFromArchitecture );
527             commands.add( "APP_ABI=" + architecture );
528 
529         }

532     /**
533      * Attaches native libs to project.
534      */
535     private void attachNativeLib( File destinationDirectory, String architecture )
536             throws IOException, MojoExecutionException
537     {
538         final File nativeArtifactFile;
539         if ( ndkFinalLibraryName == null )
540         {
541             nativeArtifactFile = findNativeLibrary( destinationDirectory );
542         }
543         else
544         {
545             nativeArtifactFile = nativeLibraryFromName( destinationDirectory );
546         }
547 
548         final String artifactType = resolveArtifactType( nativeArtifactFile );
549         if ( nativeArtifactFile.getName().endsWith( ".so" ) && ! skipStripping )
550         {
551             getLog().debug( "Post processing (stripping) native compiled artifact: " + nativeArtifactFile );
552             invokeNDKStripper( nativeArtifactFile );
553         }
554 
555         getLog().debug( "Adding native compiled artifact: " + nativeArtifactFile );
556 
557         final String classifier = ( ndkClassifier == null )
558                 ? architecture
559                 : architecture + "-" + ndkClassifier;
560 
561         projectHelper.attachArtifact( this.project, artifactType, classifier, nativeArtifactFile );
562     }

564     /**
565      * Search the specified directory for native artifacts that match the artifact Id
566      */
567     private File findNativeLibrary( File nativeLibDirectory ) throws MojoExecutionException
568     {
569         getLog().info( "Searching " + nativeLibDirectory + " for built library" );
570         File[] files = nativeLibDirectory.listFiles( new FilenameFilter()
571         {
572             public boolean accept( final File dir, final String name )
573             {
574                 String libraryName = ndkFinalLibraryName;
575                 
576                 if ( libraryName == null || libraryName.isEmpty() )
577                 {
578                     libraryName = project.getArtifactId();
579                 }
580 
581                 // FIXME: The following logic won't work for an APKLIB building a static library
582                 if ( Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )
583                 {
584                     return name.startsWith(
585                             "lib" + libraryName ) && name.endsWith( ".a" );
586                 }
587                 else
588                 {
589                     return name.startsWith(
590                             "lib" + libraryName ) && name.endsWith( ".so" );
591                 }
592             }
593         } );
594         // slight limitation at this stage - we only handle a single .so artifact
595         if ( files == null || files.length != 1 )
596         {
597             getLog().warn( "Error while detecting native compile artifacts: "
598                     + ( files == null || files.length == 0
599                             ? "None found"
600                             : "Found more than 1 artifact" ) );
601             if ( target != null )
602             {
603                 getLog().warn( "Using the 'target' configuration option to specify the output file name "
604                         + "is no longer supported, use 'ndkFinalLibraryName' instead." );
605             }
606 
607             if ( files != null && files.length > 1 )
608             {
609                 getLog().debug( "List of files found: " + Arrays.asList( files ) );
610                 getLog().error(
611                         "Currently, only a single, final native library is supported by the build" );
612                 throw new MojoExecutionException(
613                         "Currently, only a single, final native library is supported by the build" );
614             }
615             else
616             {
617                 getLog().error( "No native compiled library found, did the native compile complete "
618                         + "successfully?" );
619                 throw new MojoExecutionException( "No native compiled library found, did the native "
620                         + "compile complete successfully?" );
621             }
622         }
623         return files[ 0 ];
624     }

626     private File nativeLibraryFromName( File nativeLibDirectory ) throws MojoExecutionException
627     {
628         final File libraryFile;
629         // Find the nativeArtifactFile in the nativeLibDirectory/ndkFinalLibraryName
630         if ( Const.ArtifactType.NATIVE_SYMBOL_OBJECT.equals( project.getPackaging() )
631                 || Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )
632         {
633             libraryFile = new File( nativeLibDirectory,
634                     "lib" + ndkFinalLibraryName + "." + project.getPackaging() );
635         }
636         else
637         {
638             final File staticLib = new File( nativeLibDirectory,
639                     "lib" + ndkFinalLibraryName + ".a" );
640             if ( staticLib.exists() )
641             {
642                 libraryFile = staticLib;
643             }
644             else
645             {
646                 libraryFile = new File( nativeLibDirectory,
647                         "lib" + ndkFinalLibraryName + ".so" );
648             }
649         }
650         if ( ! libraryFile.exists() )
651         {
652             getLog().error(
653                     "Could not locate final native library using the provided ndkFinalLibraryName "
654                             + ndkFinalLibraryName + " (tried " + libraryFile.getAbsolutePath()
655                             + ")" );
656             throw new MojoExecutionException(
657                     "Could not locate final native library using the provided ndkFinalLibraryName "
658                     + ndkFinalLibraryName + " (tried " + libraryFile.getAbsolutePath()
659                     + ")" );
660         }
661 
662         return libraryFile;
663     }

665     private CommandExecutor.ErrorListener getNdkErrorListener()
666     {
667         return new CommandExecutor.ErrorListener()
668         {
669             @Override
670             public boolean isError( String error )
671             {
672 
673                 // Unconditionally ignore *All* build warning if configured to
674                 if ( ignoreBuildWarnings )
675                 {
676                     return false;
677                 }
678 
679                 final Pattern pattern = Pattern.compile( buildWarningsRegularExpression );
680                 final Matcher matcher = pattern.matcher( error );
681 
682                 // If the the reg.exp actually matches, we can safely say this is not an error
683                 // since in theory the user told us so
684                 if ( matcher.matches() )
685                 {
686                     return false;
687                 }
688 
689                 // Otherwise, it is just another error
690                 return true;
691             }
692         };
693     }

705     private void invokeNDKStripper( File file ) throws MojoExecutionException
706     {
707         try
708         {
709             getLog().debug( "Detected shared library artifact, will now strip it" );
710             // Execute the strip command
711             final CommandExecutor stripCommandExecutor = CommandExecutor.Factory.createDefaultCommmandExecutor();
712             stripCommandExecutor.setErrorListener( new CommandExecutor.ErrorListener()
713             {
714                 public boolean isError( String error )
715                 {
716                     getLog().error( "Error while stripping binary: " + error );
717                     return true;
718                 }
719             } );
720             stripCommandExecutor.setLogger( getLog() );
721             stripCommandExecutor.setCaptureStdOut( true );
722             stripCommandExecutor.executeCommand( resolveNdkStripper( file ).getAbsolutePath(),
723                                                  Arrays.asList( file.getAbsolutePath() ) );
724         }
725         catch ( ExecutionException e )
726         {
727             getLog().error( "Error while attempting to strip shared library", e );
728             throw new MojoExecutionException( "Error while attempting to strip shared library" );
729         }
730     }

754     private void attachHeaderFiles( File localCIncludesFile, String architecture )
755         throws MojoExecutionException, IOException
756     {
757 
758         final List<HeaderFilesDirective> finalHeaderFilesDirectives = new ArrayList<HeaderFilesDirective>();
759 
760         if ( useLocalSrcIncludePaths )
761         {
762             Properties props = new Properties();
763             props.load( new FileInputStream( localCIncludesFile ) );
764             String localCIncludes = props.getProperty( "LOCAL_C_INCLUDES" );
765             if ( localCIncludes != null && ! localCIncludes.trim().isEmpty() )
766             {
767                 String[] includes = localCIncludes.split( " " );
768                 for ( String include : includes )
769                 {
770                     final HeaderFilesDirective headerFilesDirective = new HeaderFilesDirective();
771                     File includeDir = new File( project.getBasedir(), include );
772                     headerFilesDirective.setDirectory( includeDir.getAbsolutePath() );
773                     headerFilesDirective.setIncludes( new String[]{ "**/*.h" } );
774                     finalHeaderFilesDirectives.add( headerFilesDirective );
775                 }
776             }
777         }
778         else
779         {
780             if ( headerFilesDirectives != null )
781             {
782                 finalHeaderFilesDirectives.addAll( headerFilesDirectives );
783             }
784         }
785         if ( finalHeaderFilesDirectives.isEmpty() )
786         {
787             getLog().debug( "No header files included, will add default set" );
788             final HeaderFilesDirective e = new HeaderFilesDirective();
789             e.setDirectory( new File( project.getBasedir() + "/jni" ).getAbsolutePath() );
790             e.setIncludes( new String[]{ "**/*.h" } );
791             finalHeaderFilesDirectives.add( e );
792         }
793         createHeaderArchive( finalHeaderFilesDirectives, architecture );
794     }

796     private void createHeaderArchive( List<HeaderFilesDirective> finalHeaderFilesDirectives, String architecture )
797             throws MojoExecutionException
798     {
799         try
800         {
801             MavenArchiver mavenArchiver = new MavenArchiver();
802             mavenArchiver.setArchiver( jarArchiver );
803 
804             final File jarFile = new File( targetDirectory, finalName + ".har" );
805 
806             mavenArchiver.setOutputFile( jarFile );
807 
808             for ( HeaderFilesDirective headerFilesDirective : finalHeaderFilesDirectives )
809             {
810                 mavenArchiver.getArchiver().addDirectory( new File( headerFilesDirective.getDirectory() ),
811                         headerFilesDirective.getIncludes(), headerFilesDirective.getExcludes() );
812             }
813 
814             final MavenArchiveConfiguration mavenArchiveConfiguration = new MavenArchiveConfiguration();
815             mavenArchiveConfiguration.setAddMavenDescriptor( false );
816 
817             mavenArchiver.createArchive( project, mavenArchiveConfiguration );
818 
819             String classifier = architecture;
820             if ( ndkClassifier != null )
821             {
822                 classifier += "-" + ndkClassifier;
823             }
824 
825             getLog().debug( "Attaching 'har' classifier=" + classifier + " file=" + jarFile );
826             projectHelper.attachArtifact( project, Const.ArtifactType.NATIVE_HEADER_ARCHIVE, classifier, jarFile );
827 
828         }
829         catch ( Exception e )
830         {
831             throw new MojoExecutionException( e.getMessage() );
832         }
833     }

861     private Set<Artifact> findNativeLibraryDependencies() throws MojoExecutionException
862     {
863         final NativeHelper nativeHelper = getNativeHelper();
864         final Set<Artifact> staticLibraryArtifacts = nativeHelper
865                 .getNativeDependenciesArtifacts( this, getUnpackedLibsDirectory(), false );
866         final Set<Artifact> sharedLibraryArtifacts = nativeHelper
867                 .getNativeDependenciesArtifacts( this, getUnpackedLibsDirectory(), true );
868         
869         final Set<Artifact> mergedArtifacts = new LinkedHashSet<Artifact>();
870         filterNativeDependencies( mergedArtifacts, staticLibraryArtifacts );
871         filterNativeDependencies( mergedArtifacts, sharedLibraryArtifacts );
872 
873         getLog().debug( "findNativeLibraryDependencies found " + mergedArtifacts.size()
874                 + ": " + mergedArtifacts.toString() );
875 
876         return mergedArtifacts;
877     }

879     /**
880      * Selectively add artifacts from source to target excluding any whose groupId and artifactId match
881      * the current build.
882      *
883      * Introduced to work around an issue when the ndk-build is executed twice by maven for example when
884      * invoking maven 'install site'. In this case the artifacts attached by the first invocation are
885      * found but are not valid dependencies and must be excluded.
886      *
887      * @param targetSet artifact Set to copy in to
888      * @param source artifact Set to filter
889      */
890     private void filterNativeDependencies( Set<Artifact> targetSet, Set<Artifact> source )
891     {
892         for ( Artifact a : source )
893         {
894             if ( project.getGroupId().equals( a.getGroupId() ) 
895                     && project.getArtifactId().equals( a.getArtifactId() ) )
896             {
897                 getLog().warn( "Excluding native dependency attached by this build" );
898             }
899             else
900             {
901                 targetSet.add( a );
902             }
903         }
904     }

906     /**
907      * Resolve the artifact type from the current project and the specified file.  If the project packaging is
908      * either 'a' or 'so' it will use the packaging, otherwise it checks the file for the extension
909      *
910      * @param file The file being added as an artifact
911      * @return The artifact type (so or a)
912      */
913     private String resolveArtifactType( File file )
914     {
915         if ( Const.ArtifactType.NATIVE_SYMBOL_OBJECT.equals( project.getPackaging() )
916                 || Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )
917         {
918             return project.getPackaging();
919         }
920         else
921         {
922             // At this point, the file (as found by our filtering previously will end with either 'so' or 'a'
923             return file.getName().endsWith( Const.ArtifactType.NATIVE_SYMBOL_OBJECT )
924                     ? Const.ArtifactType.NATIVE_SYMBOL_OBJECT
925                     : Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE;
926         }
