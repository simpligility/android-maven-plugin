330     /**
331      *
332      * @throws MojoExecutionException
333      * @throws MojoFailureException
334      */
335     public void execute() throws MojoExecutionException, MojoFailureException
336     {
337         try
338         {
339             // Validate the NDK
340             final File ndkBuildFile = new File( getAndroidNdk().getNdkBuildPath() );
341             NativeHelper.validateNDKVersion( ndkBuildFile.getParentFile() );
342 
343             // Validate the makefile - if our packaging type is so (for example) and there are
344             // dependencies on .a files (or shared files for that matter) the makefile should include
345             // the include of our Android Maven plugin generated makefile.
346             validateMakefile( project, makefile );
347 
348             String[] resolvedNDKArchitectures = NativeHelper.getNdkArchitectures(
349                     ndkArchitecture != null ? ndkArchitecture : ndkArchitectures, applicationMakefile,
350                     project.getBasedir() );
351 
352             for ( String ndkArchitecture : resolvedNDKArchitectures )
353             {
354                 getLog().debug( "Resolving for NDK architecture : " + ndkArchitecture );
355                 final Preparation preparation = new Preparation().invoke( ndkArchitecture );
356                 boolean libsDirectoryExists = preparation.isLibsDirectoryExists();
357                 final File directoryToRemove = preparation.getDirectoryToRemove();
358 
359                 // Start setting up the command line to be executed
360                 final CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
361                 // Add an error listener to the build - this allows the build to conditionally fail
362                 // depending on a) the output of the build b) whether or not build errors (output on stderr) should be
363                 // ignored and c) whether the pattern matches or not
364                 executor.setErrorListener( getNdkErrorListener() );
365 
366                 final Set<Artifact> nativeLibraryArtifacts = findNativeLibraryDependencies();
367 
368                 // If there are any static libraries the code needs to link to, include those in the make file
369                 final Set<Artifact> resolveNativeLibraryArtifacts =
370                         getArtifactResolverHelper().resolveArtifacts( nativeLibraryArtifacts );
371 
372                 getLog().debug( "resolveArtifacts found " + resolveNativeLibraryArtifacts.size()
373                         + ": " + resolveNativeLibraryArtifacts.toString() );
374 
375                 final File makefileDir = new File( project.getBuild().getDirectory(), NDK_MAKFILE_DIRECTORY );
376                 makefileDir.mkdirs();
377                 final File androidMavenMakefile = new File( makefileDir, "android_maven_plugin_makefile.mk" );
378 
379                 // set the ndk build directory
380                 if ( ndkBuildDirectory == null )
381                 {
382                     ndkBuildDirectory = project.getBasedir().getAbsolutePath();
383                 }
384 
385                 final MakefileHelper makefileHelper = new MakefileHelper( getLog(),
386                         getUnpackedLibHelper(), getArtifactResolverHelper(),
387                         harArtifactHandler, getUnpackedLibsDirectory()
388                 );
389                 final MakefileHelper.MakefileHolder makefileHolder = makefileHelper
390                         .createMakefileFromArtifacts( new File( ndkBuildDirectory ),
391                                                       resolveNativeLibraryArtifacts, ndkArchitecture, "armeabi",
392                                                       useHeaderArchives );
393 
394                 final FileOutputStream output = new FileOutputStream( androidMavenMakefile );
395                 try
396                 {
397                     IOUtil.copy( makefileHolder.getMakeFile(), output );
398                 }
399                 finally
400                 {
401                     output.close();
402                 }
403 
404                 // Add the path to the generated makefile - this is picked up by the build (by an include from the user)
405                 executor.addEnvironment( "ANDROID_MAVEN_PLUGIN_MAKEFILE", androidMavenMakefile.getAbsolutePath() );
406 
407                 setupNativeLibraryEnvironment( makefileHelper, executor, resolveNativeLibraryArtifacts,
408                                                ndkArchitecture );
409 
410                 // Adds the location of the Makefile capturer file - this file will after the build include
411                 // things like header files, flags etc.  It is processed after the build to retrieve the headers
412                 // and also capture flags etc ...
413                 final File makefileCaptureFile = File.createTempFile( "android_maven_plugin_makefile_captures",
414                                                                       ".tmp" );
415                 makefileCaptureFile.deleteOnExit();
416                 executor.addEnvironment( MakefileHelper.MAKEFILE_CAPTURE_FILE, makefileCaptureFile.getAbsolutePath() );
417 
418                 // Add any defined system properties
419                 if ( systemProperties != null && ! systemProperties.isEmpty() )
420                 {
421                     for ( Map.Entry<String, String> entry : systemProperties.entrySet() )
422                     {
423                         executor.addEnvironment( entry.getKey(), entry.getValue() );
424                     }
425                 }
426                 executor.setLogger( this.getLog() );
427                 // Setup the command line for the make
428                 final List<String> commands = new ArrayList<String>();
429                 // Setup the build directory (defaults to the current directory) but may be different depending
430                 // on user configuration
431                 commands.add( "-C" );
432                 commands.add( ndkBuildDirectory );
433 
434                 // If the build should use a custom makefile or not - some validation is done to ensure
435                 // this exists and all
436                 if ( makefile != null )
437                 {
438                     File makeFile = new File( project.getBasedir(), makefile );
439                     if ( ! makeFile.exists() )
440                     {
441                         getLog().error( "Specified makefile " + makeFile + " does not exist" );
442                         throw new MojoExecutionException( "Specified makefile " + makeFile + " does not exist" );
443                     }
444                     commands.add( "-f" );
445                     commands.add( makefile );
446                 }
447 
448                 configureApplicationMakefile( commands );
449                 configureMaxJobs( commands );
450                 configureNdkToolchain( ndkArchitecture, commands );
451                 configureAdditionalCommands( commands );
452 
453                 // If a build target is specified, tag that onto the command line as the very last of the parameters
454                 if ( target != null )
455                 {
456                     commands.add( target );
457                 }
458                 else /*if ( Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )*/
459                 {
460                     commands.add( project.getArtifactId() );
461                 }
462 
463                 final String ndkBuildPath = resolveNdkBuildExecutable();
464                 getLog().debug( ndkBuildPath + " " + commands.toString() );
465                 getLog().info( "Executing NDK " + ndkArchitecture + " make at : " + ndkBuildDirectory );
466 
467                 executor.setCaptureStdOut( true );
468                 executor.executeCommand( ndkBuildPath, commands, project.getBasedir(), true );
469 
470                 cleanUp( preparation.getNativeLibDirectory(), ndkArchitecture, libsDirectoryExists, directoryToRemove,
471                          makefileHolder, makefileCaptureFile );
472             }
473         }
474         catch ( MojoExecutionException e )
475         {
476             getLog().error( "Error during build: " + e.getMessage(), e );
477             throw e;
478         }
479         catch ( Exception e )
480         {
481             getLog().error( "Error while executing: " + e.getMessage() );
482             throw new MojoExecutionException( e.getMessage(), e );
483         }
484 
485     }

487     private void configureAdditionalCommands( final List<String> commands )
488     {
489         // Anything else on the command line the user wants to add - simply splice it up and
490         // add it one by one to the command line
491         if ( ndkBuildAdditionalCommandline != null )
492         {
493             String[] additionalCommands = ndkBuildAdditionalCommandline.split( " " );
494             for ( final String command : additionalCommands )
495             {
496                 commands.add( command );
497             }
498         }
499     }

501     private void configureApplicationMakefile( List<String> commands )
502         throws MojoExecutionException
503     {
504         if ( applicationMakefile != null )
505         {
506             File appMK = new File( project.getBasedir(), applicationMakefile );
507             if ( ! appMK.exists() )
508             {
509                 getLog().error( "Specified application makefile " + appMK + " does not exist" );
510                 throw new MojoExecutionException( "Specified application makefile " + appMK + " does not exist" );
511             }
512             commands.add( "NDK_APPLICATION_MK=" + applicationMakefile );
513         }
514     }

516     private void configureMaxJobs( List<String> commands )
517     {
518         if ( maxJobs )
519         {
520             String jobs = String.valueOf( Runtime.getRuntime().availableProcessors() );
521             getLog().info( "executing " + jobs + " parallel jobs" );
522             commands.add( "-j" );
523             commands.add( jobs );
524         }

527     private void configureNdkToolchain( String ndkArchitecture, List<String> commands )
528             throws MojoExecutionException
529     {
530         if ( ndkToolchain != null )
531         {
532             // Setup the correct toolchain to use
533             // FIXME: perform a validation that this toolchain exists in the NDK
534             commands.add( "NDK_TOOLCHAIN=" + ndkToolchain );
535         }
536         else
537         {
538             // Resolve the toolchain from the architecture
539             // <ndkArchitectures>
540             //   <x86>x86-4.6</x86>
541             //   <armeabi>x86-4.6</armeabi>
542             // </ndkArchitectures>
543             final String toolchainFromArchitecture = getAndroidNdk().getToolchainFromArchitecture(
544                     ndkArchitecture, ndkArchitectureToolchainMappings );
545             getLog().debug( "Resolved toolchain for " + ndkArchitecture + " to " + toolchainFromArchitecture );
546             commands.add( "NDK_TOOLCHAIN=" + toolchainFromArchitecture );
547             commands.add( "APP_ABI=" + ndkArchitecture );
548 
549         }
550     }

552     private void cleanUp( File nativeLibDirectory, String ndkArchitecture, boolean libsDirectoryExists,
553                           File directoryToRemove, MakefileHelper.MakefileHolder makefileHolder,
554                           File makefileCaptureFile )
555             throws IOException, MojoExecutionException
556     {
557         try
558         {
559             // Cleanup libs/armeabi directory if needed - this implies moving any native artifacts into target/libs
560             if ( clearNativeArtifacts )
561             {
562                 nativeLibDirectory = cleanUpNativeArtifacts( nativeLibDirectory, ndkArchitecture, libsDirectoryExists );
563             }
564 
565             // Attempt to attach the native library if the project is defined as a "pure" native Android library
566             // (packaging is 'so' or 'a') or if the plugin has been configured to attach the native library to the
567             // build
568             if ( Const.ArtifactType.NATIVE_SYMBOL_OBJECT.equals( project.getPackaging() )
569                     || Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() )
570                     || attachNativeArtifacts )
571             {
572 
573                 final File nativeArtifactFile;
574                 if ( ndkFinalLibraryName == null )
575                 {
576                     nativeArtifactFile = findNativeLibrary( nativeLibDirectory );
577                 }
578                 else
579                 {
580                     nativeArtifactFile = nativeLibraryFromName( nativeLibDirectory );
581                 }
582 
583                 final String artifactType = resolveArtifactType( nativeArtifactFile );
584                 if ( nativeArtifactFile.getName().endsWith( ".so" ) && ! skipStripping )
585                 {
586                     getLog().debug( "Post processing (stripping) native compiled artifact: " + nativeArtifactFile );
587                     invokeNDKStripper( nativeArtifactFile );
588                 }
589 
590                 getLog().debug( "Adding native compiled artifact: " + nativeArtifactFile );
591 
592                 File fileToAttach = nativeArtifactFile;
593                 if ( ! libsDirectoryExists && !clearNativeArtifacts )
594                 {
595                     final String destFileName = ndkArchitecture + File.separator + nativeArtifactFile.getName();
596                     final File destFile = new File( ndkOutputDirectory, destFileName );
597                     if ( !destFile.equals( nativeArtifactFile ) )
598                     {
599                         getLog().debug( "Moving native compiled artifact to target directory for preservation" );
600                         // This indicates the output directory was created by the build (us) and that we should really
601                         // move it to the target (needed to preserve the attached artifact once install is invoked)
602                         if ( destFile.exists() )
603                         {
604                             destFile.delete();
605                         }
606                         getLog().debug( nativeArtifactFile + " -> " + destFile );
607                         FileUtils.moveFile( nativeArtifactFile, destFile );
608                         fileToAttach = destFile;
609                     }
610                     else
611                     {
612                         getLog().debug( "Not moving native compiled artifact "
613                             + nativeArtifactFile + " to target as they point to the same file" );
614                         fileToAttach = nativeArtifactFile;
615                     }
616                 }
617 
618                 String classifier = ndkArchitecture;
619                 if ( ndkClassifier != null )
620                 {
621                     classifier += "-" + ndkClassifier;
622                 }
623 
624                 projectHelper.attachArtifact( this.project, artifactType, classifier, fileToAttach );
625             }
626 
627             // Process conditionally any of the headers to include into the header archive file
628             processMakefileCapture( makefileCaptureFile, ndkArchitecture );
629 
630         }
631         finally
632         {
633             // If we created any directories as part of the build, blow those away after we're done
634             if ( ! libsDirectoryExists )
635             {
636                 getLog().info( "Cleaning up native library output directory after build" );
637                 getLog().debug( "Removing directory: " + directoryToRemove ); // AJE - removes 'obj' directory
638                 FileUtils.deleteDirectory( directoryToRemove );
639             }
640 
641             // If we created a makefile for the build we should be polite and remove any extracted include
642             // directories after we're done
643             if ( makefileHolder != null )
644             {
645                 getLog().info( "Cleaning up extracted include directories used for build" );
646                 MakefileHelper.cleanupAfterBuild( makefileHolder );
647             }
648 
649         }
650     }

652     /**
653      * Search the specified directory for native artifacts that match the artifact Id
654      */
655     private File findNativeLibrary( File nativeLibDirectory ) throws MojoExecutionException
656     {
657         getLog().info( "Searching " + nativeLibDirectory + " for built library" );
658         File[] files = nativeLibDirectory.listFiles( new FilenameFilter()
659         {
660             public boolean accept( final File dir, final String name )
661             {
662                 String libraryName = ndkFinalLibraryName;
663                 
664                 if ( libraryName == null || libraryName.isEmpty() )
665                 {
666                     libraryName = project.getArtifactId();
667                 }
668 
669                 // FIXME: The following logic won't work for an APKLIB building a static library
670                 if ( Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )
671                 {
672                     return name.startsWith(
673                             "lib" + libraryName ) && name.endsWith( ".a" );
674                 }
675                 else
676                 {
677                     return name.startsWith(
678                             "lib" + libraryName ) && name.endsWith( ".so" );
679                 }
680             }
681         } );
682         // slight limitation at this stage - we only handle a single .so artifact
683         if ( files == null || files.length != 1 )
684         {
685             getLog().warn( "Error while detecting native compile artifacts: "
686                     + ( files == null || files.length == 0
687                             ? "None found"
688                             : "Found more than 1 artifact" ) );
689             if ( target != null )
690             {
691                 getLog().warn( "Using the 'target' configuration option to specify the output file name "
692                         + "is no longer supported, use 'ndkFinalLibraryName' instead." );
693             }
694 
695             if ( files != null && files.length > 1 )
696             {
697                 getLog().debug( "List of files found: " + Arrays.asList( files ) );
698                 getLog().error(
699                         "Currently, only a single, final native library is supported by the build" );
700                 throw new MojoExecutionException(
701                         "Currently, only a single, final native library is supported by the build" );
702             }
703             else
704             {
705                 getLog().error( "No native compiled library found, did the native compile complete "
706                         + "successfully?" );
707                 throw new MojoExecutionException( "No native compiled library found, did the native "
708                         + "compile complete successfully?" );
709             }
710         }
711         return files[ 0 ];
712     }

714     private File nativeLibraryFromName( File nativeLibDirectory ) throws MojoExecutionException
715     {
716         final File libraryFile;
717         // Find the nativeArtifactFile in the nativeLibDirectory/ndkFinalLibraryName
718         if ( Const.ArtifactType.NATIVE_SYMBOL_OBJECT.equals( project.getPackaging() )
719                 || Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )
720         {
721             libraryFile = new File( nativeLibDirectory,
722                     "lib" + ndkFinalLibraryName + "." + project.getPackaging() );
723         }
724         else
725         {
726             final File staticLib = new File( nativeLibDirectory,
727                     "lib" + ndkFinalLibraryName + ".a" );
728             if ( staticLib.exists() )
729             {
730                 libraryFile = staticLib;
731             }
732             else
733             {
734                 libraryFile = new File( nativeLibDirectory,
735                         "lib" + ndkFinalLibraryName + ".so" );
736             }
737         }
738         if ( ! libraryFile.exists() )
739         {
740             getLog().error(
741                     "Could not locate final native library using the provided ndkFinalLibraryName "
742                             + ndkFinalLibraryName + " (tried " + libraryFile.getAbsolutePath()
743                             + ")" );
744             throw new MojoExecutionException(
745                     "Could not locate final native library using the provided ndkFinalLibraryName "
746                     + ndkFinalLibraryName + " (tried " + libraryFile.getAbsolutePath()
747                     + ")" );
748         }
749 
750         return libraryFile;
751     }

753     private CommandExecutor.ErrorListener getNdkErrorListener()
754     {
755         return new CommandExecutor.ErrorListener()
756         {
757             @Override
758             public boolean isError( String error )
759             {
760 
761                 // Unconditionally ignore *All* build warning if configured to
762                 if ( ignoreBuildWarnings )
763                 {
764                     return false;
765                 }
766 
767                 final Pattern pattern = Pattern.compile( buildWarningsRegularExpression );
768                 final Matcher matcher = pattern.matcher( error );
769 
770                 // If the the reg.exp actually matches, we can safely say this is not an error
771                 // since in theory the user told us so
772                 if ( matcher.matches() )
773                 {
774                     return false;
775                 }
776 
777                 // Otherwise, it is just another error
778                 return true;
779             }
780         };
781     }

805     private void invokeNDKStripper( File file ) throws MojoExecutionException
806     {
807         try
808         {
809             getLog().debug( "Detected shared library artifact, will now strip it" );
810             // Execute the strip command
811             final CommandExecutor stripCommandExecutor = CommandExecutor.Factory.createDefaultCommmandExecutor();
812             stripCommandExecutor.setErrorListener( new CommandExecutor.ErrorListener()
813             {
814                 public boolean isError( String error )
815                 {

818                 }
819             } );
820             stripCommandExecutor.setLogger( getLog() );
821             stripCommandExecutor.setCaptureStdOut( true );
822             stripCommandExecutor.executeCommand( resolveNdkStripper( file ).getAbsolutePath(),
823                                                  Arrays.asList( file.getAbsolutePath() ) );
824         }
825         catch ( ExecutionException e )
826         {
827             getLog().error( "Error while attempting to strip shared library", e );
828             throw new MojoExecutionException( "Error while attempting to strip shared library" );
829         }

854     private void processMakefileCapture( File localCIncludesFile, String ndkArchitecture ) throws MojoExecutionException
855     {
856 
857         try
858         {
859             if ( attachHeaderFiles )
860             {
861 
862                 final List<HeaderFilesDirective> finalHeaderFilesDirectives = new ArrayList<HeaderFilesDirective>();
863 
864                 if ( useLocalSrcIncludePaths )
865                 {
866                     Properties props = new Properties();
867                     props.load( new FileInputStream( localCIncludesFile ) );
868                     String localCIncludes = props.getProperty( "LOCAL_C_INCLUDES" );
869                     if ( localCIncludes != null && ! localCIncludes.trim().isEmpty() )
870                     {
871                         String[] includes = localCIncludes.split( " " );
872                         for ( String include : includes )
873                         {
874                             final HeaderFilesDirective headerFilesDirective = new HeaderFilesDirective();
875                             File includeDir = new File( project.getBasedir(), include );
876                             headerFilesDirective.setDirectory( includeDir.getAbsolutePath() );
877                             headerFilesDirective.setIncludes( new String[]{ "**/*.h" } );
878                             finalHeaderFilesDirectives.add( headerFilesDirective );
879                         }
880                     }
881                 }
882                 else
883                 {
884                     if ( headerFilesDirectives != null )
885                     {
886                         finalHeaderFilesDirectives.addAll( headerFilesDirectives );
887                     }
888                 }
889                 if ( finalHeaderFilesDirectives.isEmpty() )
890                 {
891                     getLog().debug( "No header files included, will add default set" );
892                     final HeaderFilesDirective e = new HeaderFilesDirective();
893                     e.setDirectory( new File( project.getBasedir() + "/jni" ).getAbsolutePath() );
894                     e.setIncludes( new String[]{ "**/*.h" } );
895                     finalHeaderFilesDirectives.add( e );
896                 }
897                 createHeaderArchive( finalHeaderFilesDirectives, ndkArchitecture );
898             }
899         }
900         catch ( Exception e )
901         {
902             throw new MojoExecutionException( "Error while processing headers to include: " + e.getMessage(), e );
903         }
904 
905     }

907     private void createHeaderArchive( List<HeaderFilesDirective> finalHeaderFilesDirectives, String ndkArchitecture )
908             throws MojoExecutionException
909     {
910         try
911         {
912             MavenArchiver mavenArchiver = new MavenArchiver();
913             mavenArchiver.setArchiver( jarArchiver );
914 
915             final File jarFile = new File( new File( project.getBuild().getDirectory() ),
916                     project.getBuild().getFinalName() + ".har" );
917 
918             mavenArchiver.setOutputFile( jarFile );
919 
920             for ( HeaderFilesDirective headerFilesDirective : finalHeaderFilesDirectives )
921             {
922                 mavenArchiver.getArchiver().addDirectory( new File( headerFilesDirective.getDirectory() ),
923                         headerFilesDirective.getIncludes(), headerFilesDirective.getExcludes() );
924             }
925 
926             final MavenArchiveConfiguration mavenArchiveConfiguration = new MavenArchiveConfiguration();
927             mavenArchiveConfiguration.setAddMavenDescriptor( false );
928 
929             mavenArchiver.createArchive( project, mavenArchiveConfiguration );
930 
931             String classifier = ndkArchitecture;
932             if ( ndkClassifier != null )
933             {
934                 classifier += "-" + ndkClassifier;
935             }
936 
937             getLog().debug( "Attaching 'har' classifier=" + classifier + " file=" + jarFile );
938             projectHelper.attachArtifact( project, Const.ArtifactType.NATIVE_HEADER_ARCHIVE, classifier, jarFile );
939 
940         }
941         catch ( Exception e )
942         {
943             throw new MojoExecutionException( e.getMessage() );
944         }

974     {
975         final NativeHelper nativeHelper = getNativeHelper();
976         final Set<Artifact> staticLibraryArtifacts = nativeHelper
977                 .getNativeDependenciesArtifacts( this, getUnpackedLibsDirectory(), false );
978         final Set<Artifact> sharedLibraryArtifacts = nativeHelper
979                 .getNativeDependenciesArtifacts( this, getUnpackedLibsDirectory(), true );

991     /**
992      * Selectively add artifacts from source to target excluding any whose groupId and artifactId match
993      * the current build.
994      * Introduced to work around an issue when the ndk-build is executed twice by maven for example when
995      * invoking maven 'install site'. In this case the artifacts attached by the first invocation are
996      * found but are not valid dependencies and must be excluded.
997      * @param target artifact Set to copy in to
998      * @param source artifact Set to filter
999      */
1000     private void filterNativeDependencies( Set<Artifact> target, Set<Artifact> source )
1001     {
1002         for ( Artifact a : source )
1003         {
1004             if ( project.getGroupId().equals( a.getGroupId() ) 
1005                     && project.getArtifactId().equals( a.getArtifactId() ) )
1006             {
1007                 getLog().warn( "Excluding native dependency attached by this build" );
1008             }
1009             else
1010             {
1011                 target.add( a );
1012             }
1013         }
1014     }

1016     /**
1017      * Resolve the artifact type from the current project and the specified file.  If the project packaging is
1018      * either 'a' or 'so' it will use the packaging, otherwise it checks the file for the extension
1019      *
1020      * @param file The file being added as an artifact
1021      * @return The artifact type (so or a)
1022      */
1023     private String resolveArtifactType( File file )
1024     {

1026                 || Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )

1029         }
1030         else
1031         {
1032             // At this point, the file (as found by our filtering previously will end with either 'so' or 'a'
1033             return file.getName().endsWith( Const.ArtifactType.NATIVE_SYMBOL_OBJECT )
1034                     ? Const.ArtifactType.NATIVE_SYMBOL_OBJECT
1035                     : Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE;
1036         }

1060         public Preparation invoke( String ndkArchitecture )
1061         {
1062             // This usually points to ${basedir}/obj/local
1063             nativeLibDirectory = new File( nativeLibrariesOutputDirectory, ndkArchitecture );
1064             libsDirectoryExists = nativeLibDirectory.exists();
1065 
1066             // Determine how much of the output directory structure (most likely obj/...) does not exist
1067             // and based on what we find, determine how much of it we delete after the build
1068             directoryToRemove = nativeLibDirectory;
1069             if ( ! libsDirectoryExists )
1070             {
1071                 getLog().info( "Creating native output directory " + nativeLibDirectory );
1072 
1073                 // This simply checks how much of the structure already exists - nothing (e.g. we make all the dirs)
1074                 // or just a partial part (the architecture part)?
1075                 if ( ! nativeLibrariesOutputDirectory.exists() )
1076                 {
1077                     if ( nativeLibrariesOutputDirectory.getParentFile().exists() )
1078                     {
1079                         nativeLibDirectory.mkdir();
1080                     }
1081                     else
1082                     {
1083                         nativeLibDirectory.mkdirs();
1084                         directoryToRemove = nativeLibrariesOutputDirectory.getParentFile();
1085                     }
1086                 }
1087                 else
1088                 {
1089                     if ( nativeLibDirectory.getParentFile().exists() )
1090                     {
1091                         nativeLibDirectory.mkdir();
1092                     }
1093                     else
1094                     {
1095                         nativeLibDirectory.mkdirs();
1096                         directoryToRemove = nativeLibDirectory.getParentFile();
1097                     }
1098                 }
1099             }
1100             return this;
1101         }
