38       * Holder for the result of creating a makefile.  This in particular keep tracks of all directories created
39       * for extracted header files.
40       */
41      public static class MakefileHolder
42      {
43          String makeFile;
44          List<File> includeDirectories;
45  
46          public MakefileHolder( List<File> includeDirectories, String makeFile )
47          {
48              this.includeDirectories = includeDirectories;
49              this.makeFile = makeFile;
50          }
51  
52          public List<File> getIncludeDirectories()
53          {
54              return includeDirectories;
55          }
56  
57          public String getMakeFile()
58          {
59              return makeFile;
60          }

84      /**
85       * Cleans up all include directories created in the temp directory during the build.
86       *
87       * @param makefileHolder The holder produced by the
88       * {@link MakefileHelper#createMakefileFromArtifacts(Set, String, String, boolean)}
89       */
90      public static void cleanupAfterBuild( MakefileHolder makefileHolder )
91      {
92  
93          if ( makefileHolder.getIncludeDirectories() != null )
94          {
95              for ( File file : makefileHolder.getIncludeDirectories() )
96              {
97                  try
98                  {
99                      FileUtils.deleteDirectory( file );
100                 }
101                 catch ( IOException e )
102                 {
103                     e.printStackTrace();
104                 }
105             }
106         }
107 
108     }

110     /**
111      * Creates an Android Makefile based on the specified set of static library dependency artifacts.
112      *
113      * @param artifacts         The list of (static library) dependency artifacts to create the Makefile from
114      * @param useHeaderArchives If true, the Makefile should include a LOCAL_EXPORT_C_INCLUDES statement, pointing to
115      *                          the location where the header archive was expanded
116      * @return The created Makefile
117      */
118     public MakefileHolder createMakefileFromArtifacts( Set<Artifact> artifacts,
119                                                               String ndkArchitecture, String defaultNDKArchitecture,
120                                                               boolean useHeaderArchives )
121             throws IOException, MojoExecutionException
122     {
123 
124         final StringBuilder makeFile = new StringBuilder( "# Generated by Android Maven Plugin\n" );
125         final List<File> includeDirectories = new ArrayList<File>();
126 
127         // Add now output - allows us to somewhat intelligently determine the include paths to use for the header
128         // archive
129         makeFile.append( "$(shell echo \"LOCAL_C_INCLUDES=$(LOCAL_C_INCLUDES)\" > $(" + MAKEFILE_CAPTURE_FILE + "))" );
130         makeFile.append( '\n' );
131         makeFile.append( "$(shell echo \"LOCAL_PATH=$(LOCAL_PATH)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
132         makeFile.append( '\n' );
133         makeFile.append( "$(shell echo \"LOCAL_MODULE_FILENAME=$(LOCAL_MODULE_FILENAME)\" >> $("
134                 + MAKEFILE_CAPTURE_FILE + "))" );
135         makeFile.append( '\n' );
136         makeFile.append( "$(shell echo \"LOCAL_MODULE=$(LOCAL_MODULE)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
137         makeFile.append( '\n' );
138         makeFile.append( "$(shell echo \"LOCAL_CFLAGS=$(LOCAL_CFLAGS)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
139         makeFile.append( '\n' );
140 
141         if ( ! artifacts.isEmpty() )
142         {
143             for ( Artifact artifact : artifacts )
144             {
145                 final String architecture = NativeHelper.extractArchitectureFromArtifact( artifact,
146                         defaultNDKArchitecture );
147 
148                 makeFile.append( '\n' );
149                 makeFile.append( "ifeq ($(TARGET_ARCH_ABI)," ).append( architecture ).append( ")\n" );
150 
151                 makeFile.append( "#\n" );
152                 makeFile.append( "# Group ID: " );
153                 makeFile.append( artifact.getGroupId() );
154                 makeFile.append( '\n' );
155                 makeFile.append( "# Artifact ID: " );
156                 makeFile.append( artifact.getArtifactId() );
157                 makeFile.append( '\n' );
158                 makeFile.append( "# Artifact Type: " );
159                 makeFile.append( artifact.getType() );
160                 makeFile.append( '\n' );
161                 makeFile.append( "# Version: " );
162                 makeFile.append( artifact.getVersion() );
163                 makeFile.append( '\n' );
164                 makeFile.append( "include $(CLEAR_VARS)" );
165                 makeFile.append( '\n' );
166                 makeFile.append( "LOCAL_MODULE    := " );
167                 makeFile.append( artifact.getArtifactId() );
168                 makeFile.append( '\n' );
169 
170                 final boolean apklibStatic = addLibraryDetails( makeFile, artifact, ndkArchitecture );
171 
172                 if ( useHeaderArchives )
173                 {
174                     try
175                     {
176                         // Fix for dealing with APKLIBs - unfortunately it does not fully work since
177                         // an APKLIB can contain any number of architectures making it somewhat to resolve the
178                         // related (HAR) artifact.
179                         //
180                         // In this case, we construct the classifier from <architecture> and the artifact classifier
181                         // if it is also present.  Only issue is that if the APKLIB contains more than the armeabi
182                         // libraries (e.g. x86 for examples) the HAR is not resolved correctly.
183                         //
184                         String classifier = artifact.getClassifier();
185                         if ( "apklib".equals( artifact.getType() ) )
186                         {
187                             classifier = ndkArchitecture;
188                             if ( artifact.getClassifier() != null )
189                             {
190                                 classifier += "-" + artifact.getClassifier();
191                             }
192                         }
193 
194                         Artifact harArtifact = new DefaultArtifact( artifact.getGroupId(), artifact.getArtifactId(),
195                                 artifact.getVersion(), artifact.getScope(),
196                                 Const.ArtifactType.NATIVE_HEADER_ARCHIVE, classifier,
197                                 harArtifactHandler );
198 
199                         File resolvedHarArtifactFile = artifactResolverHelper.resolveArtifactToFile( harArtifact );
200                         log.debug( "Resolved har artifact file : " + resolvedHarArtifactFile );
201 
202                         final File includeDir = new File( System.getProperty( "java.io.tmpdir" ),
203                                 "android_maven_plugin_native_includes" + System.currentTimeMillis() + "_"
204                                         + harArtifact.getArtifactId() );
205                         includeDir.deleteOnExit();
206                         includeDirectories.add( includeDir );
207 
208                         JarHelper.unjar( new JarFile( resolvedHarArtifactFile ), includeDir,
209                                 new JarHelper.UnjarListener()
210                                 {
211                                     @Override
212                                     public boolean include( JarEntry jarEntry )
213                                     {
214                                         return ! jarEntry.getName().startsWith( "META-INF" );
215                                     }
216                                 } );
217 
218                         makeFile.append( "LOCAL_EXPORT_C_INCLUDES := " );
219                         makeFile.append( includeDir.getAbsolutePath() );
220                         makeFile.append( '\n' );
221                         
222                         if ( log.isDebugEnabled() )
223                         {
224                             Collection<File> includes = FileUtils.listFiles( includeDir,
225                                     TrueFileFilter.INSTANCE, TrueFileFilter.INSTANCE );
226                             log.debug( "Listing LOCAL_EXPORT_C_INCLUDES for " + artifact.getId() + ": " + includes );
227                         }
228                     }
229                     catch ( RuntimeException e )
230                     {
231                         throw new MojoExecutionException(
232                                 "Error while resolving header archive file for: " + artifact.getArtifactId(), e );
233                     }
234                 }
235                 if ( Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( artifact.getType() ) || apklibStatic )
236                 {
237                     makeFile.append( "include $(PREBUILT_STATIC_LIBRARY)\n" );
238                 }
239                 else
240                 {
241                     makeFile.append( "include $(PREBUILT_SHARED_LIBRARY)\n" );
242                 }
243 
244                 makeFile.append( "endif #" ).append( artifact.getClassifier() ).append( '\n' );
245                 makeFile.append( '\n' );
246 
247             }
248         }
249         
250         return new MakefileHolder( includeDirectories, makeFile.toString() );
251     }

253     private boolean addLibraryDetails( StringBuilder makeFile,
254                                        Artifact artifact, String ndkArchitecture ) throws IOException
255     {
256         boolean apklibStatic = false;
257         if ( AndroidExtension.APKLIB.equals( artifact.getType() ) )
258         {
259             String classifier = artifact.getClassifier();
260             String architecture = ( classifier != null ) ? classifier : ndkArchitecture;
261             // 
262             // We assume that APKLIB contains a single static OR shared library
263             // that we should link against. The follow code identifies that file.
264             //
265             File[] staticLibs = NativeHelper.listNativeFiles( artifact, unpackedApkLibsDirectory, 
266                                                               architecture, true );
267             if ( staticLibs != null && staticLibs.length > 0 )
268             {
269                 int libIdx = findApklibNativeLibrary( staticLibs, artifact.getArtifactId() );
270                 apklibStatic = true;
271                 addLibraryDetails( makeFile, staticLibs[libIdx], "" );
272             }
273             else
274             {
275                 File[] sharedLibs = NativeHelper.listNativeFiles( artifact, unpackedApkLibsDirectory, 
276                                                                   architecture, false );
277                 if ( sharedLibs == null )
278                 {
279                     throw new IOException( "Failed to find any library file in APKLIB" );
280                 }
281                 int libIdx = findApklibNativeLibrary( sharedLibs, artifact.getArtifactId() );
282                 addLibraryDetails( makeFile, sharedLibs[libIdx], "" );
283             }
284         }
285         else
286         {
287             addLibraryDetails( makeFile, artifact.getFile(), artifact.getArtifactId() );
288         }
289 
290         return apklibStatic;
291     }

293     private void addLibraryDetails( StringBuilder makeFile, File libFile, String outputName )
294         throws IOException
295     {
296         makeFile.append( "LOCAL_PATH := " );
297         makeFile.append( libFile.getParentFile().getAbsolutePath() );
298         makeFile.append( '\n' );
299         makeFile.append( "LOCAL_SRC_FILES := " );
300         makeFile.append( libFile.getName() );
301         makeFile.append( '\n' );
302         makeFile.append( "LOCAL_MODULE_FILENAME := " );
303         if ( "".equals( outputName ) )
304         {
305             makeFile.append( FilenameUtils.removeExtension( libFile.getName() ) );
306         }
307         else
308         {
309             makeFile.append( outputName );
310         }
311         makeFile.append( '\n' );
312     }

314     /**
315      * @param libs the array of possible library files. Must not be null.
316      * @return the index in the array of the library to use
317      * @throws IOException if a library cannot be identified
318      */
319     private int findApklibNativeLibrary( File[] libs, String artifactName ) throws IOException
320     {
321         int libIdx = -1;
322         
323         if ( libs.length == 1 )
324         
325         {
326             libIdx = 0;
327         }
328         else
329         {
330             log.info( "Found multiple library files, looking for name match with artifact" );
331             StringBuilder sb = new StringBuilder();
332             for ( int i = 0; i < libs.length; i++ )
333             {
334                 if ( sb.length() != 0 )
335                 {
336                     sb.append( ", " );
337                 }
338                 sb.append( libs[i].getName() );
339                 if ( libs[i].getName().startsWith( "lib" + artifactName ) )
340                 {
341                     if ( libIdx != -1 )
342                     {
343                         // We have multiple matches, tell the user we can't handle this ...
344                         throw new IOException( "Found multiple libraries matching artifact name " + artifactName
345                                 + ". Please use unique artifact/library names." );
346                         
347                     }
348                     libIdx = i;
349                 }
350             }
351             if ( libIdx < 0 )
352             {
353                 throw new IOException( "Unable to determine main library from " + sb.toString()
354                         + " APKLIB should contain only 1 library or a library matching the artifact name" );
355             }
356         }
357         return libIdx;
358     }

360     /**
361      * Creates a list of artifacts suitable for use in the LOCAL_STATIC_LIBRARIES or LOCAL_SHARED_LIBRARIES 
362      * variable in an Android makefile
363      *
364      * @param resolvedLibraryList
365      * @param staticLibrary
366      * @return a list of Ids for artifacts that include static or shared libraries
367      */
368     public String createLibraryList( Set<Artifact> resolvedLibraryList,
369                                      String ndkArchitecture,
370                                      boolean staticLibrary )
371     {
372         Set<String> libraryNames = new LinkedHashSet<String>();
373 
374         for ( Artifact a : resolvedLibraryList )
375         {
376             if ( staticLibrary && Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( a.getType() ) )
377             {
378                 libraryNames.add( a.getArtifactId() );
379             }
380             if ( ! staticLibrary && Const.ArtifactType.NATIVE_SYMBOL_OBJECT.equals( a.getType() ) )
381             {
382                 libraryNames.add( a.getArtifactId() );
383             }
384             if ( AndroidExtension.APKLIB.equals( a.getType() ) || AndroidExtension.AAR.equals( a.getType() ) )
385             {
386                 File[] libFiles = NativeHelper.listNativeFiles( a, unpackedApkLibsDirectory, 
387                                                                 ndkArchitecture, staticLibrary );
388                 if ( libFiles != null && libFiles.length > 0 )
389                 {
390                     libraryNames.add( a.getArtifactId() );
391                 }
392                 
393             }
394         }
395 
396         StringBuilder sb = new StringBuilder();
397 
398         Iterator<String> iter = libraryNames.iterator();
399 
400         while ( iter.hasNext() )
401         {
402             sb.append( iter.next() );
403 
404             if ( iter.hasNext() )
405             {
406                 sb.append( " " );
407             }
408         }
409 
410         return sb.toString();
411     }
