101     {
102 
103         if ( makefileHolder.getIncludeDirectories() != null )

120     /**
121      * Creates an Android Makefile based on the specified set of static library dependency artifacts.
122      *
123      * @param outputDir         Directory to resolve artifact locations relative to.  Makefiles contain relative paths
124      * @param artifacts         The list of (static library) dependency artifacts to create the Makefile from
125      * @param useHeaderArchives If true, the Makefile should include a LOCAL_EXPORT_C_INCLUDES statement, pointing to
126      *                          the location where the header archive was expanded
127      * @param repoSession
128      * @param projectRepos
129      * @param repoSystem
130      * @return The created Makefile
131      */
132     public MakefileHolder createMakefileFromArtifacts( File outputDir, Set<Artifact> artifacts,
133                                                               String ndkArchitecture, String defaultNDKArchitecture,
134                                                               boolean useHeaderArchives )
135             throws IOException, MojoExecutionException
136     {
137 
138         final StringBuilder makeFile = new StringBuilder( "# Generated by Android Maven Plugin\n" );
139         final List<File> includeDirectories = new ArrayList<File>();
140 
141         // Add now output - allows us to somewhat intelligently determine the include paths to use for the header
142         // archive
143         makeFile.append( "$(shell echo \"LOCAL_C_INCLUDES=$(LOCAL_C_INCLUDES)\" > $(" + MAKEFILE_CAPTURE_FILE + "))" );
144         makeFile.append( '\n' );
145         makeFile.append( "$(shell echo \"LOCAL_PATH=$(LOCAL_PATH)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
146         makeFile.append( '\n' );
147         makeFile.append( "$(shell echo \"LOCAL_MODULE_FILENAME=$(LOCAL_MODULE_FILENAME)\" >> $("
148                 + MAKEFILE_CAPTURE_FILE + "))" );
149         makeFile.append( '\n' );
150         makeFile.append( "$(shell echo \"LOCAL_MODULE=$(LOCAL_MODULE)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
151         makeFile.append( '\n' );
152         makeFile.append( "$(shell echo \"LOCAL_CFLAGS=$(LOCAL_CFLAGS)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
153         makeFile.append( '\n' );
154 
155         if ( ! artifacts.isEmpty() )
156         {
157             for ( Artifact artifact : artifacts )
158             {
159                 final String architecture = NativeHelper.extractArchitectureFromArtifact( artifact,
160                         defaultNDKArchitecture );
161 
162                 makeFile.append( '\n' );
163                 makeFile.append( "ifeq ($(TARGET_ARCH_ABI)," ).append( architecture ).append( ")\n" );
164 
165                 makeFile.append( "#\n" );
166                 makeFile.append( "# Group ID: " );
167                 makeFile.append( artifact.getGroupId() );
168                 makeFile.append( '\n' );
169                 makeFile.append( "# Artifact ID: " );
170                 makeFile.append( artifact.getArtifactId() );
171                 makeFile.append( '\n' );
172                 makeFile.append( "# Artifact Type: " );
173                 makeFile.append( artifact.getType() );
174                 makeFile.append( '\n' );
175                 makeFile.append( "# Version: " );
176                 makeFile.append( artifact.getVersion() );
177                 makeFile.append( '\n' );
178                 makeFile.append( "include $(CLEAR_VARS)" );
179                 makeFile.append( '\n' );
180                 makeFile.append( "LOCAL_MODULE    := " );
181                 makeFile.append( artifact.getArtifactId() );
182                 makeFile.append( '\n' );
183 
184                 final boolean apklibStatic = addLibraryDetails( makeFile, outputDir, artifact, ndkArchitecture );
185 
186                 if ( useHeaderArchives )
187                 {
188                     try
189                     {
190                         // Fix for dealing with APKLIBs - unfortunately it does not fully work since
191                         // an APKLIB can contain any number of architectures making it somewhat to resolve the
192                         // related (HAR) artifact.
193                         //
194                         // In this case, we construct the classifier from <architecture> and the artifact classifier
195                         // if it is also present.  Only issue is that if the APKLIB contains more than the armeabi
196                         // libraries (e.g. x86 for examples) the HAR is not resolved correctly.
197                         //
198                         String classifier = artifact.getClassifier();
199                         if ( "apklib".equals( artifact.getType() ) )
200                         {
201                             classifier = ndkArchitecture;
202                             if ( artifact.getClassifier() != null )
203                             {
204                                 classifier += "-" + artifact.getClassifier();
205                             }
206                         }
207 
208                         Artifact harArtifact = new DefaultArtifact( artifact.getGroupId(), artifact.getArtifactId(),
209                                 artifact.getVersion(), artifact.getScope(), "har", classifier,
210                                 artifact.getArtifactHandler() );
211                         final Artifact resolvedHarArtifact = AetherHelper
212                                 .resolveArtifact( harArtifact, repoSystem, repoSession, projectRepos );
213 
214                         File includeDir = new File( System.getProperty( "java.io.tmpdir" ),
215                                 "android_maven_plugin_native_includes" + System.currentTimeMillis() + "_"
216                                         + resolvedHarArtifact.getArtifactId() );
217                         includeDir.deleteOnExit();
218                         includeDirectories.add( includeDir );
219 
220                         JarHelper.unjar( new JarFile( resolvedHarArtifact.getFile() ), includeDir,
221                                 new JarHelper.UnjarListener()
222                                 {
223                                     @Override
224                                     public boolean include( JarEntry jarEntry )
225                                     {
226                                         return ! jarEntry.getName().startsWith( "META-INF" );
227                                     }
228                                 } );
229 
230                         makeFile.append( "LOCAL_EXPORT_C_INCLUDES := " );
231                         final String str = includeDir.getAbsolutePath();
232                         makeFile.append( str );
233                         makeFile.append( '\n' );
234                         
235                         if ( log.isDebugEnabled() )
236                         {
237                             Collection<File> includes = FileUtils.listFiles( includeDir,
238                                     TrueFileFilter.INSTANCE, TrueFileFilter.INSTANCE );
239                             log.debug( "Listing LOCAL_EXPORT_C_INCLUDES for " + artifact.getId() + ": " + includes );
240                         }
241                     }
242                     catch ( Exception e )
243                     {
244                         throw new MojoExecutionException(
245                                 "Error while resolving header archive file for: " + artifact.getArtifactId(), e );
246                     }
247                 }
248                 if ( "a".equals( artifact.getType() ) || apklibStatic )
249                 {
250                     makeFile.append( "include $(PREBUILT_STATIC_LIBRARY)\n" );
251                 }
252                 else
253                 {
254                     makeFile.append( "include $(PREBUILT_SHARED_LIBRARY)\n" );
255                 }
256 
257                 makeFile.append( "endif #" ).append( artifact.getClassifier() ).append( '\n' );
258                 makeFile.append( '\n' );
259 
260             }
261         }
262         
263         return new MakefileHolder( includeDirectories, makeFile.toString() );
264     }

266     private boolean addLibraryDetails( StringBuilder makeFile, File outputDir,
267                                        Artifact artifact, String ndkArchitecture ) throws IOException
268     {
269         boolean apklibStatic = false;
270         if ( AndroidExtension.APKLIB.equals( artifact.getType() ) )
271         {
272             String classifier = artifact.getClassifier();
273             String architecture = ( classifier != null ) ? classifier : ndkArchitecture;
274             // 
275             // We assume that APKLIB contains a single static OR shared library
276             // that we should link against. The follow code identifies that file.
277             //
278             File[] staticLibs = NativeHelper.listNativeFiles( artifact, unpackedApkLibsDirectory, 
279                                                               architecture, true );
280             if ( staticLibs != null && staticLibs.length > 0 )
281             {
282                 int libIdx = findApklibNativeLibrary( staticLibs, artifact.getArtifactId() );
283                 apklibStatic = true;
284                 addLibraryDetails( makeFile, outputDir, staticLibs[libIdx], "" );
285             }
286             else
287             {
288                 File[] sharedLibs = NativeHelper.listNativeFiles( artifact, unpackedApkLibsDirectory, 
289                                                                   architecture, false );
290                 if ( sharedLibs == null )
291                 {
292                     throw new IOException( "Failed to find any library file in APKLIB" );
293                 }
294                 int libIdx = findApklibNativeLibrary( sharedLibs, artifact.getArtifactId() );
295                 addLibraryDetails( makeFile, outputDir, sharedLibs[libIdx], "" );
296             }
297         }
298         else
299         {
300             addLibraryDetails( makeFile, outputDir, artifact.getFile(), artifact.getArtifactId() );
301         }
302 
303         return apklibStatic;
304     }

306     private void addLibraryDetails( StringBuilder makeFile, File outputDir, File libFile, String outputName )
307         throws IOException
308     {
309         String localPath = resolveRelativePath( outputDir, libFile );
310         localPath = localPath.substring( 0, localPath.indexOf( libFile.getName() ) - 1 );
311 
312         makeFile.append( "LOCAL_PATH := " );
313         makeFile.append( localPath );
314         makeFile.append( '\n' );
315         makeFile.append( "LOCAL_SRC_FILES := " );
316         makeFile.append( libFile.getName() );
317         makeFile.append( '\n' );
318         makeFile.append( "LOCAL_MODULE_FILENAME := " );
319         if ( "".equals( outputName ) )
320         {
321             makeFile.append( FilenameUtils.removeExtension( libFile.getName() ) );
322         }
323         else
324         {
325             makeFile.append( outputName );
326         }
327         makeFile.append( '\n' );
328     }

330     /**
331      * @param libs the array of possible library files. Must not be null.
332      * @return the index in the array of the library to use
333      * @throws IOException if a library cannot be identified
334      */
335     private int findApklibNativeLibrary( File[] libs, String artifactName ) throws IOException
336     {
337         int libIdx = -1;
338         
339         if ( libs.length == 1 )
340         
341         {
342             libIdx = 0;
343         }
344         else
345         {
346             log.info( "Found multiple library files, looking for name match with artifact" );
347             StringBuilder sb = new StringBuilder();
348             for ( int i = 0; i < libs.length; i++ )
349             {
350                 if ( sb.length() != 0 )
351                 {
352                     sb.append( ", " );
353                 }
354                 sb.append( libs[i].getName() );
355                 if ( libs[i].getName().startsWith( "lib" + artifactName ) )
356                 {
357                     if ( libIdx != -1 )
358                     {
359                         // We have multiple matches, tell the user we can't handle this ...
360                         throw new IOException( "Found multiple libraries matching artifact name " + artifactName
361                                 + ". Please use unique artifact/library names." );
362                         
363                     }
364                     libIdx = i;
365                 }
366             }
367             if ( libIdx < 0 )
368             {
369                 throw new IOException( "Unable to determine main library from " + sb.toString()
370                         + " APKLIB should contain only 1 library or a library matching the artifact name" );
371             }
372         }
373         return libIdx;
374     }

376     /**
377      * Resolves the relative path of the specified artifact
378      *
379      * @param outputDirectory typically the parent directory of the directory containing the makefile
380      * @param file
381      * @return
382      */
383     protected static String resolveRelativePath( File outputDirectory, File file ) throws IOException
384     {
385         String resolvedPath = file.getCanonicalPath();
386         
387         String strOutputDirectoryPath = outputDirectory.getCanonicalPath();
388         String strFilePath = file.getCanonicalPath();
389         //System.out.println( "Resolving " + strFilePath + " against " + strOutputDirectoryPath );
390 
391         if ( strFilePath.startsWith( strOutputDirectoryPath ) )
392         {
393             // Simple case where file is in a subdirectory of outputDirectory
394             resolvedPath =  strFilePath.substring( strOutputDirectoryPath.length() + 1 );
395         }
396         else
397         {
398             // Look for commonality in paths
399             List<String> outputDirectoryPathParts = splitPath( outputDirectory.getCanonicalFile() );
400             List<String> filePathParts = splitPath( file.getCanonicalFile() );
401             int commonDepth = 0;
402             int maxCommonDepth = Math.min( outputDirectoryPathParts.size(), filePathParts.size() );
403             for ( int i = 0; 
404                     ( i < maxCommonDepth ) 
405                     && outputDirectoryPathParts.get( i ).equals( filePathParts.get( i ) ); 
406                     i++ )
407             {
408                 commonDepth++;
409             }
410             // If there is a common root build a relative path between the common roots
411             if ( commonDepth > 0 )
412             {
413                 final StringBuilder stringBuilder = new StringBuilder();
414                 stringBuilder.append( ".." );
415                 for ( int i = 0; i < outputDirectoryPathParts.size() - commonDepth - 1; i++ )
416                 {
417                     stringBuilder.append( File.separator );
418                     stringBuilder.append( ".." );
419                 }
420                 for ( int i = commonDepth; i < filePathParts.size(); i++ )
421                 {
422                     stringBuilder.append( File.separator );
423                     stringBuilder.append( filePathParts.get( i ) );
424                 }
425                 resolvedPath = stringBuilder.toString();
426             }
427             else
428             {
429                 if ( IS_WINDOWS )
430                 {
431                     // Windows has no common root directory, cannot resolve a path
432                     // across drives so ...
433                     throw new IOException( "Unable to resolve relative path across windows drives" );
434                 }
435 
436                 // no intersection between paths so calculate a path via the root directory
437                 final StringBuilder stringBuilder = new StringBuilder();
438                 
439                 File depthCheck = outputDirectory.getParentFile();
440                 while ( depthCheck != null )
441                 {
442                     if ( stringBuilder.length() > 0 )
443                     {
444                         stringBuilder.append( File.separator );
445                     }
446                     stringBuilder.append( ".." );
447                     depthCheck = depthCheck.getParentFile();
448                 }
449     
450                 resolvedPath = stringBuilder.toString() + strFilePath;
451             }
452         }
453 
454         //System.out.println( "resolvedPath = " + resolvedPath );
455         return resolvedPath;
456     }

458     /**
459      * Method to split the path components of a file into a List
460      * @param f the file to split
461      * @return a new list containing the components of the path as strings
462      */
463     protected static List<String> splitPath( File f )
464     {
465         List<String> result;
466         File parent = f.getParentFile();
467         if ( parent == null )
468         {
469             result = new ArrayList<String>();
470             if ( f.getName().length() > 0 )
471             {
472                 // We're at the root but have a name so we have a relative path
473                 // for which we need to add the first component to the list
474                 result.add( f.getName() );
475             }
476             else if ( IS_WINDOWS ) 
477             {
478                 String strF = f.toString();
479                 if ( strF.matches( WINDOWS_DRIVE_ROOT_REGEX ) )
480                 {
481                     // We're on windows and the path is <Drive>:\ so we
482                     // add the <Drive>: to the list
483                     result.add( strF.substring( 0, strF.length() - 1 ).toUpperCase() );
484                 }
485             }
486         }
487         else
488         {
489             result = splitPath( parent );
490             result.add( f.getName() );
491         }
492         return result;
493     }

495     /**
496      * Creates a list of artifacts suitable for use in the LOCAL_STATIC_LIBRARIES or LOCAL_SHARED_LIBRARIES 
497      * variable in an Android makefile
498      *
499      * @param resolvedLibraryList
500      * @param staticLibrary
501      * @return a list of Ids for artifacts that include static or shared libraries
502      */
503     public String createLibraryList( Set<Artifact> resolvedLibraryList,
504                                      String ndkArchitecture,
505                                      boolean staticLibrary )
506     {
507         Set<String> libraryNames = new LinkedHashSet<String>();
508 
509         for ( Artifact a : resolvedLibraryList )
510         {
511             if ( staticLibrary && "a".equals( a.getType() ) )
512             {
513                 libraryNames.add( a.getArtifactId() );
514             }
515             if ( ! staticLibrary && "so".equals( a.getType() ) )
516             {
517                 libraryNames.add( a.getArtifactId() );
518             }
519             if ( AndroidExtension.APKLIB.equals( a.getType() ) || AndroidExtension.AAR.equals( a.getType() ) )
520             {
521                 File[] libFiles = NativeHelper.listNativeFiles( a, unpackedApkLibsDirectory, 
522                                                                 ndkArchitecture, staticLibrary );
523                 if ( libFiles != null && libFiles.length > 0 )
524                 {
525                     libraryNames.add( a.getArtifactId() );
526                 }
527                 
528             }
529         }
530 
531         StringBuilder sb = new StringBuilder();
532 
533         Iterator<String> iter = libraryNames.iterator();
534 
535         while ( iter.hasNext() )
536         {
537             sb.append( iter.next() );
538 
539             if ( iter.hasNext() )
540             {
541                 sb.append( " " );
542             }
543         }
544 
545         return sb.toString();
546     }
