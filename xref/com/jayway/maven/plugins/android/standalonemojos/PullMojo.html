44  /**
45   * Copy file or directory from all the attached (or specified)
46   * devices/emulators.
47   *
48   * @author Manfred Moser <manfred@simpligility.com>
49   */
50  @Mojo( name = "pull", requiresProject = false )
51  public class PullMojo extends AbstractAndroidMojo
52  {
53  
54      /**
55       * <p>The configuration for the pull goal can be set up in the plugin configuration in the pom file as:</p>
56       * <pre>
57       * &lt;pull&gt;
58       *     &lt;source&gt;path&lt;/source&gt;
59       *     &lt;destination&gt;path&lt;/destination&gt;
60       * &lt;/pull&gt;
61       * </pre>
62       * <p>The parameters can also be configured as property in the pom or settings file
63       * <pre>
64       * &lt;properties&gt;
65       *     &lt;android.pull.source&gt;pathondevice&lt;/android.pull.source&gt;
66       *     &lt;android.pull.destination&gt;path&lt;/android.pull.destination&gt;
67       * &lt;/properties&gt;
68       * </pre>
69       * or from command-line with parameter <code>-Dandroid.pull.source=path</code>
70       * and <code>-Dandroid.pull.destination=path</code>.</p>
71       */
72      @Parameter
73      @ConfigPojo
74      private Pull pull;
75  
76      /**
77       * The path of the source file or directory on the emulator/device.
78       */
79      @Parameter( property = "android.pull.source" )
80      private String pullSource;
81  
82      @PullParameter( required = true )
83      private String parsedSource;
84  
85      /**
86       * The path of the destination to copy the file to.
87       * <p/>
88       * If destination ends with {@link File#separator}, it is supposed to be a
89       * directory. Therefore the source - whether it refers to a file or
90       * directory - will be copied into the destination directory.
91       * <p/>
92       * If destination does not end with {@link File#separator}, the last path
93       * segment will be assumed as the new file or directory name (depending on
94       * the type of source).
95       * <p/>
96       * Any missing directories will be created.
97       */
98      @Parameter( property = "android.pull.destination" )
99      private String pullDestination;
100 
101     @PullParameter( required = true )
102     private String parsedDestination;
103 
104     public void execute() throws MojoExecutionException, MojoFailureException
105     {
106 
107         ConfigHandler configHandler = new ConfigHandler( this, this.session, this.execution );
108         configHandler.parseConfiguration();
109 
110         doWithDevices( new DeviceCallback()
111         {
112             public void doWithDevice( final IDevice device ) throws MojoExecutionException
113             {
114                 String deviceLogLinePrefix = DeviceHelper.getDeviceLogLinePrefix( device );
115 
116                 // message will be set later according to the processed files
117                 String message = "";
118                 try
119                 {
120                     SyncService syncService = device.getSyncService();
121                     FileListingService fileListingService = device.getFileListingService();
122 
123                     FileEntry sourceFileEntry = getFileEntry( parsedSource, fileListingService );
124 
125                     if ( sourceFileEntry.isDirectory() )
126                     {
127                         // pulling directory
128                         File destinationDir = new File( parsedDestination );
129                         if ( ! destinationDir.exists() )
130                         {
131                             getLog().info( "Creating destination directory " + destinationDir );
132                             destinationDir.mkdirs();
133                             destinationDir.mkdir();
134                         }
135                         String destinationDirPath = destinationDir.getAbsolutePath();
136 
137                         FileEntry[] fileEntries;
138                         if ( parsedDestination.endsWith( File.separator ) )
139                         {
140                             // pull source directory directly
141                             fileEntries = new FileEntry[]{ sourceFileEntry };
142                         }
143                         else
144                         {
145                             // pull the children of source directory only
146                             fileEntries = fileListingService.getChildren( sourceFileEntry, true, null );
147                         }
148 
149                         message = deviceLogLinePrefix + "Pull of " + parsedSource + " to " + destinationDirPath 
150                                 + " from ";
151 
152                         syncService.pull( fileEntries, destinationDirPath, new LogSyncProgressMonitor( getLog() ) );
153                     }
154                     else
155                     {
156                         // pulling file
157                         File parentDir = new File( FilenameUtils.getFullPath( parsedDestination ) );
158                         if ( ! parentDir.exists() )
159                         {
160                             getLog().info( deviceLogLinePrefix + "Creating destination directory " + parentDir );
161                             parentDir.mkdirs();
162                         }
163 
164                         String destinationFileName;
165                         if ( parsedDestination.endsWith( File.separator ) )
166                         {
167                             // keep original filename
168                             destinationFileName = FilenameUtils.getName( parsedSource );
169                         }
170                         else
171                         {
172                             // rename filename
173                             destinationFileName = FilenameUtils.getName( parsedDestination );
174                         }
175 
176                         File destinationFile = new File( parentDir, destinationFileName );
177                         String destinationFilePath = destinationFile.getAbsolutePath();
178                         message = deviceLogLinePrefix + "Pull of " + parsedSource + " to " + destinationFilePath 
179                                 + " from " + DeviceHelper.getDescriptiveName( device );
180 
181                         syncService.pullFile( sourceFileEntry, destinationFilePath,
182                                 new LogSyncProgressMonitor( getLog() ) );
183                     }
184 
185                     getLog().info( message + " successful." );
186                 }
187                 catch ( SyncException e )
188                 {
189                     throw new MojoExecutionException( message + " failed.", e );
190                 }
191                 catch ( IOException e )
192                 {
193                     throw new MojoExecutionException( message + " failed.", e );
194                 }
195                 catch ( TimeoutException e )
196                 {
197                     throw new MojoExecutionException( message + " failed.", e );
198                 }
199                 catch ( AdbCommandRejectedException e )
200                 {
201                     throw new MojoExecutionException( message + " failed.", e );
202                 }
203             }
204         } );
205     }
206 
207     /**
208      * Retrieves the corresponding {@link FileEntry} on the emulator/device for
209      * a given file path.
210      * <p/>
211      * If the file path starts with the symlink "sdcard", it will be resolved
212      * statically to "/mnt/sdcard".
213      *
214      * @param filePath           path to file or directory on device or emulator
215      * @param fileListingService {@link FileListingService} for retrieving the
216      *                           {@link FileEntry}
217      * @return a {@link FileEntry} object for the given file path
218      * @throws MojoExecutionException if the file path could not be found on the device
219      */
220     private FileEntry getFileEntry( String filePath, FileListingService fileListingService )
221             throws MojoExecutionException
222     {
223         // static resolution of symlink
224         if ( filePath.startsWith( "/sdcard" ) )
225         {
226             filePath = "/mnt" + filePath;
227         }
228 
229         String[] destinationPathSegments = StringUtils.split( filePath, "/" );
230 
231         FileEntry fileEntry = fileListingService.getRoot();
232         for ( String destinationPathSegment : destinationPathSegments )
233         {
234             // build up file listing cache
235             fileListingService.getChildren( fileEntry, true, null );
236 
237             fileEntry = fileEntry.findChild( destinationPathSegment );
238             if ( fileEntry == null )
239             {
240                 throw new MojoExecutionException( "Cannot execute goal: " + filePath + " does not exist on device." );
241             }
242         }
243         return fileEntry;
244     }
