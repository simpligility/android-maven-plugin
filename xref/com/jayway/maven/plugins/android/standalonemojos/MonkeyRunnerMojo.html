297     /**
298      * Actually plays tests.
299      * 
300      * @param device
301      *            the device on which tests are going to be executed.
302      * @param iTestRunListeners
303      *            test run listeners.
304      * @throws MojoExecutionException
305      *             if exercising app threw an exception and isIgnoreTestFailures is false..
306      * @throws MojoFailureException
307      *             if exercising app failed and isIgnoreTestFailures is false.
308      */
309     protected void run( IDevice device, ITestRunListener... iTestRunListeners ) throws MojoExecutionException,
310             MojoFailureException
311     {
312 
313         this.mTestListeners = iTestRunListeners;
314 
315         getLog().debug( "Parsed values for Android Monkey Runner invocation: " );
316 
317         CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
318         if ( !Os.isFamily( Os.FAMILY_WINDOWS ) )
319         {
320             executor.setCustomShell( new CustomBourneShell() );
321         }
322         executor.setLogger( this.getLog() );
323 
324         String command = getAndroidSdk().getMonkeyRunnerPath();
325 
326         List< String > pluginParameters = new ArrayList< String >();
327 
328         if ( parsedPlugins != null && parsedPlugins.length != 0 )
329         {
330             for ( String plugin : parsedPlugins )
331             {
332                 String pluginFilePath = new File( project.getBasedir(), plugin ).getAbsolutePath();
333                 pluginParameters.add( "-plugin " + pluginFilePath );
334             }
335         }
336 
337         if ( parsedPrograms != null && !parsedPrograms.isEmpty() )
338         {
339             handleTestRunStarted();
340             errorListener = new MonkeyRunnerErrorListener();
341             executor.setErrorListener( errorListener );
342 
343             for ( Program program : parsedPrograms )
344             {
345                 List< String > parameters = new ArrayList< String >( pluginParameters );
346 
347                 String programFileName = new File( project.getBasedir(), program.getFilename() ).getAbsolutePath();
348                 parameters.add( programFileName );
349                 String testName = programFileName;
350                 if ( testName.contains( "/" ) )
351                 {
352                     testName.substring( testName.indexOf( '/' ) + 1 );
353                 }
354                 mCurrentTestIndentifier = new TestIdentifier( "MonkeyTest ", testName );
355 
356                 String programOptions = program.getOptions();
357                 if ( parsedInjectDeviceSerialNumberIntoScript != null && parsedInjectDeviceSerialNumberIntoScript )
358                 {
359                     parameters.add( device.getSerialNumber() );
360                 }
361                 if ( programOptions != null && !StringUtils.isEmpty( programOptions ) )
362                 {
363                     parameters.add( programOptions );
364                 }
365 
366                 try
367                 {
368                     getLog().info( "Running command: " + command );
369                     getLog().info( "with parameters: " + parameters );
370                     handleTestStarted();
371                     executor.executeCommand( command, parameters, true );
372                     handleTestEnded();
373                 }
374                 catch ( ExecutionException e )
375                 {
376                     getLog().info( "Monkey runner produced errors" );
377                     handleTestRunFailed( e.getMessage() );
378 
379                     if ( !isIgnoreTestFailures() )
380                     {
381                         getLog().info( "Project is configured to fail on error." );
382                         getLog().info(
383                                 "Inspect monkey runner reports or re-run with -X to see monkey runner errors in log" );
384                         getLog().info( "Failing build as configured. Ignore following error message." );
385                         if ( errorListener.hasError )
386                         {
387                             getLog().info( "Stack trace is:" );
388                             getLog().info( errorListener.getStackTrace() );
389                         }
390                         throw new MojoExecutionException( "", e );
391                     }
392                 }
393 
394                 if ( errorListener.hasError() )
395                 {
396                     handleCrash();
397                 }
398             }
399             handleTestRunEnded();
400         }
401 
402         getLog().info( "Monkey runner test runs completed successfully." );
403     }
404 
405     private void handleTestRunStarted()
406     {
407         runMetrics = new HashMap< String, String >();
408         elapsedTime = System.currentTimeMillis();
409         for ( ITestRunListener listener : mTestListeners )
410         {
411             listener.testRunStarted( mRunName, eventCount );
412         }
413     }
414 
415     private void handleTestRunFailed( String error )
416     {
417         for ( ITestRunListener listener : mTestListeners )
418         {
419             listener.testRunFailed( error );
420         }
421     }
422 
423     private void handleTestRunEnded()
424     {
425         elapsedTime = System.currentTimeMillis() - elapsedTime;
426 
427         for ( ITestRunListener listener : mTestListeners )
428         {
429             listener.testRunEnded( elapsedTime, runMetrics );
430         }
431     }
432 
433     private void handleTestStarted()
434     {
435         System.out.println( "TEST START " + mTestListeners.length );
436         for ( ITestRunListener listener : mTestListeners )
437         {
438             listener.testStarted( mCurrentTestIndentifier );
439         }
440     }
441 
442     private void handleTestEnded()
443     {
444         if ( mCurrentTestIndentifier != null )
445         {
446             for ( ITestRunListener listener : mTestListeners )
447             {
448                 listener.testEnded( mCurrentTestIndentifier, new HashMap< String, String >() );
449             }
450             mCurrentTestIndentifier = null;
451         }
452     }
453 
454     private void handleCrash()
455     {
456 
457         String trace = errorListener.getStackTrace();
458 
459         for ( ITestRunListener listener : mTestListeners )
460         {
461             listener.testFailed( TestFailure.ERROR, mCurrentTestIndentifier, trace );
462         }
463         mCurrentTestIndentifier = null;

467     private final class MonkeyRunnerErrorListener implements CommandExecutor.ErrorListener
468     {
469         private StringBuilder stackTraceBuilder = new StringBuilder();
470         private boolean hasError = false;
471 
472         @Override
473         public boolean isError( String error )
474         {
475 
476             // Unconditionally ignore *All* build warning if configured to
477             if ( isIgnoreTestFailures() )
478             {
479                 return false;
480             }
481 
482             if ( hasError )
483             {
484                 stackTraceBuilder.append( error ).append( '\n' );
485             }
486 
487             final Pattern pattern = Pattern.compile( ".*error.*|.*exception.*", Pattern.CASE_INSENSITIVE );
488             final Matcher matcher = pattern.matcher( error );
489 
490             // If the the reg.exp actually matches, we can safely say this is not an error
491             // since in theory the user told us so
492             if ( matcher.matches() )
493             {
494                 hasError = true;
495                 stackTraceBuilder.append( error ).append( '\n' );
496                 return true;
497             }
498 
499             // Otherwise, it is just another error
500             return false;
501         }
502 
503         public String getStackTrace()
504         {
505             if ( hasError )
506             {
507                 return stackTraceBuilder.toString();
508             }
509             else
510             {
511                 return null;
512             }
513         }
514 
515         public boolean hasError()
516         {
517             return hasError;
518         }

621         @Override
622         public void testRunStarted( String runName, int tCount )
623         {
624 
625             this.testCount = tCount;
626             getLog().info( deviceLogLinePrefix + INDENT + "Run started: " + runName + ", " + testCount + " tests:" );
627 
628             if ( parsedCreateReport )
629             {
630                 try
631                 {
632                     DocumentBuilderFactory fact = DocumentBuilderFactory.newInstance();
633                     DocumentBuilder parser = null;
634                     parser = fact.newDocumentBuilder();
635                     junitReport = parser.newDocument();
636                     Node testSuitesNode = junitReport.createElement( TAG_TESTSUITES );
637                     junitReport.appendChild( testSuitesNode );
638                     testSuiteNode = junitReport.createElement( TAG_TESTSUITE );
639                     NamedNodeMap testSuiteAttributes = testSuiteNode.getAttributes();
640                     Attr nameAttr = junitReport.createAttribute( ATTR_TESTSUITE_NAME );
641                     nameAttr.setValue( runName );
642                     testSuiteAttributes.setNamedItem( nameAttr );
643                     Attr hostnameAttr = junitReport.createAttribute( ATTR_TESTSUITE_HOSTNAME );
644                     hostnameAttr.setValue( DeviceHelper.getDescriptiveName( device ) );
645                     testSuiteAttributes.setNamedItem( hostnameAttr );
646                     Node propertiesNode = junitReport.createElement( TAG_PROPERTIES );
647                     Node propertyNode;
648                     NamedNodeMap propertyAttributes;
649                     Attr propNameAttr;
650                     Attr propValueAttr;
651                     for ( Map.Entry< Object, Object > systemProperty : System.getProperties().entrySet() )
652                     {
653                         propertyNode = junitReport.createElement( TAG_PROPERTY );
654                         propertyAttributes = propertyNode.getAttributes();
655                         propNameAttr = junitReport.createAttribute( ATTR_PROPERTY_NAME );
656                         propNameAttr.setValue( systemProperty.getKey().toString() );
657                         propertyAttributes.setNamedItem( propNameAttr );
658                         propValueAttr = junitReport.createAttribute( ATTR_PROPERTY_VALUE );
659                         propValueAttr.setValue( systemProperty.getValue().toString() );
660                         propertyAttributes.setNamedItem( propValueAttr );
661                         propertiesNode.appendChild( propertyNode );
662                     }
663                     Map< String, String > deviceProperties = device.getProperties();
664                     for ( Map.Entry< String, String > deviceProperty : deviceProperties.entrySet() )
665                     {
666                         propertyNode = junitReport.createElement( TAG_PROPERTY );
667                         propertyAttributes = propertyNode.getAttributes();
668                         propNameAttr = junitReport.createAttribute( ATTR_PROPERTY_NAME );
669                         propNameAttr.setValue( deviceProperty.getKey() );
670                         propertyAttributes.setNamedItem( propNameAttr );
671                         propValueAttr = junitReport.createAttribute( ATTR_PROPERTY_VALUE );
672                         propValueAttr.setValue( deviceProperty.getValue() );
673                         propertyAttributes.setNamedItem( propValueAttr );
674                         propertiesNode.appendChild( propertyNode );
675                     }
676                     testSuiteNode.appendChild( propertiesNode );
677                     testSuitesNode.appendChild( testSuiteNode );
678                 }
679                 catch ( ParserConfigurationException e )
680                 {
681                     threwException = true;
682                     exceptionMessages.append( "Failed to create document" );
683                     exceptionMessages.append( e.getMessage() );
684                 }
685             }
686         }

866         /**
867          * Parse a trace string for the message in it. Assumes that the message is located after ":" and before "\r\n".
868          * 
869          * @param trace
870          * @return message or empty string
871          */
872         private String parseForMessage( String trace )
873         {
874             if ( StringUtils.isNotBlank( trace ) )
875             {
876                 String newline = "\r\n";
877                 // if there is message like
878                 // junit.junit.framework.AssertionFailedError ... there is no
879                 // message
880                 int messageEnd = trace.indexOf( newline );
881                 boolean hasMessage = !trace.startsWith( "junit." ) && messageEnd > 0;
882                 if ( hasMessage )
883                 {
884                     int messageStart = trace.indexOf( ":" ) + 2;
885                     if ( messageStart > messageEnd )
886                     {
887                         messageEnd = trace.indexOf( newline + "at" );
888                         // match start of stack trace "\r\nat org.junit....."
889                         if ( messageStart > messageEnd )
890                         {
891                             // ':' wasn't found in message but in stack trace
892                             messageStart = 0;
893                         }
894                     }
895                     return trace.substring( messageStart, messageEnd );
896                 }
897                 else
898                 {
899                     return StringUtils.EMPTY;
900                 }
901             }
902             else
903             {
904                 return StringUtils.EMPTY;
905             }
906         }
907 
908         /**
909          * Parse a trace string for the exception class. Assumes that it is the start of the trace and ends at the first
910          * ":".
911          * 
912          * @param trace
913          * @return Exception class as string or empty string
914          */
915         private String parseForException( String trace )
916         {
917             if ( StringUtils.isNotBlank( trace ) )
918             {
919                 return trace.substring( 0, trace.indexOf( ":" ) );
920             }
921             else
922             {
923                 return StringUtils.EMPTY;
924             }
925         }

954                 File reportFile = new File( fileName );
