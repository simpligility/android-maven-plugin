50  /**
51   * Runs the first Activity shown in the top-level launcher as determined by its Intent filters.
52   * <p>
53   * Android provides a component-based architecture, which means that there is no "main" function which serves as an
54   * entry point to the APK. There's an homogeneous collection of Activity(es), Service(s), Receiver(s), etc.
55   * </p>
56   * <p>
57   * The Android top-level launcher (whose purpose is to allow users to launch other applications) uses the Intent
58   * resolution mechanism to determine which Activity(es) to show to the end user. Such activities are identified by at
59   * least:
60   * <ul>
61   * <li>Action type: <code>android.intent.action.MAIN</code></li>
62   * <li>Category: <code>android.intent.category.LAUNCHER</code></li>
63   * </ul>
64   * </p>
65   * <p>And are declared in <code>AndroidManifest.xml</code> as such:</p>
66   * <pre>
67   * &lt;activity android:name=".ExampleActivity"&gt;
68   *     &lt;intent-filter&gt;
69   *         &lt;action android:name="android.intent.action.MAIN" /&gt;
70   *         &lt;category android:name="android.intent.category.LAUNCHER" /&gt;
71   *     &lt;/intent-filter&gt;
72   * &lt;/activity&gt;
73   * </pre>
74   * <p>
75   * This {@link Mojo} will try to to launch the first activity of this kind found in <code>AndroidManifest.xml</code>. In
76   * case multiple activities satisfy the requirements listed above only the first declared one is run. In case there are
77   * no "Launcher activities" declared in the manifest or no activities declared at all, this goal aborts throwing an
78   * error.
79   * </p>
80   * <p>
81   * The device parameter is taken into consideration so potentially the Activity found is started on all attached
82   * devices. The application will NOT be deployed and running will silently fail if the application is not deployed.
83   * </p>
84   *
85   * @author Lorenzo Villani <lorenzo@villani.me>
86   * @author Manfred Mosr <manfred@simpligility.com>
87   * @see "http://developer.android.com/guide/topics/fundamentals.html"
88   * @see "http://developer.android.com/guide/topics/intents/intents-filters.html"
89   */
90  @Mojo( name = "run" )
91  public class RunMojo extends AbstractAndroidMojo
92  {
93  
94      /**
95       * <p>The configuration for the run goal can be set up in the plugin configuration in the pom file as:</p>
96       * <pre>
97       * &lt;run&gt;
98       *     &lt;debug&gt;true&lt;/debug&gt;
99       * &lt;/run&gt;
100      * </pre>
101      * <p>The <code>&lt;debug&gt;</code> parameter is optional and defaults to false.
102      * <p>The debug parameter can also be configured as property in the pom or settings file
103      * <pre>
104      * &lt;properties&gt;
105      *     &lt;android.run.debug&gt;true&lt;/android.run.debug&gt;
106      * &lt;/properties&gt;
107      * </pre>
108      * or from command-line with parameter <code>-Dandroid.run.debug=true</code>.</p>
109      */
110     @Parameter
111     @ConfigPojo
112     private Run run;
113

118     @Parameter( property = "android.run.debug" )

161     /**
162      * {@inheritDoc}
163      */
164     @Override
165     public void execute() throws MojoExecutionException, MojoFailureException
166     {
167         if ( project.getPackaging().equals( APK ) )
168         {
169             try
170             {
171                 LauncherInfo launcherInfo;
172 
173                 launcherInfo = getLauncherActivity();
174 
175                 ConfigHandler configHandler = new ConfigHandler( this, this.session, this.execution );
176                 configHandler.parseConfiguration();
177 
178                 launch( launcherInfo );
179             }
180             catch ( Exception ex )
181             {
182                 getLog().info( "Unable to run launcher Activity" );
183                 getLog().debug( ex );
184             }
185         }
186         else
187         {
188             getLog().info( "Project packaging is not apk, skipping run action." );
189         }
190     }

196     /**
197      * Gets the first "Launcher" Activity by running an XPath query on <code>AndroidManifest.xml</code>.
198      *
199      * @return A {@link LauncherInfo}
200      * @throws MojoFailureException
201      * @throws ParserConfigurationException
202      * @throws IOException
203      * @throws SAXException
204      * @throws XPathExpressionException
205      */
206     private LauncherInfo getLauncherActivity()
207             throws ParserConfigurationException, SAXException, IOException, XPathExpressionException,
208             MojoFailureException
209     {
210         Document document;
211         DocumentBuilder documentBuilder;
212         DocumentBuilderFactory documentBuilderFactory;
213         Object result;
214         XPath xPath;
215         XPathExpression xPathExpression;
216         XPathFactory xPathFactory;
217 
218         //
219         // Setup JAXP stuff
220         //
221         documentBuilderFactory = DocumentBuilderFactory.newInstance();
222 
223         documentBuilder = documentBuilderFactory.newDocumentBuilder();
224 
225         document = documentBuilder.parse( destinationManifestFile );
226 
227         xPathFactory = XPathFactory.newInstance();
228 
229         xPath = xPathFactory.newXPath();
230 
231         xPathExpression = xPath.compile(
232                 "//manifest/application/activity/intent-filter[action[@name=\"android.intent.action.MAIN\"] "
233                 + "and category[@name=\"android.intent.category.LAUNCHER\"]]/.." );
234 
235         //
236         // Run XPath query
237         //
238         result = xPathExpression.evaluate( document, XPathConstants.NODESET );
239 
240         if ( result instanceof NodeList )
241         {
242             NodeList activities;
243 
244             activities = ( NodeList ) result;
245 
246             if ( activities.getLength() > 0 )
247             {
248                 // Grab the first declared Activity
249                 LauncherInfo launcherInfo;
250 
251                 launcherInfo = new LauncherInfo();
252                 String activityName = activities.item( 0 ).getAttributes().getNamedItem( "android:name" )
253                         .getNodeValue();
254 
255                 if ( ! activityName.contains( "." ) )
256                 {
257                     activityName = "." + activityName;
258                 }
259 
260                 if ( activityName.startsWith( "." ) )
261                 {
262                     String packageName = document.getElementsByTagName( "manifest" ).item( 0 ).getAttributes()
263                             .getNamedItem( "package" ).getNodeValue();
264                     activityName = packageName + activityName;
265                 }
266 
267                 launcherInfo.activity = activityName;
268 
269                 launcherInfo.packageName = renameManifestPackage != null
270                     ? renameManifestPackage
271                     : document.getDocumentElement().getAttribute( "package" ).toString();
272 
273                 return launcherInfo;
274             }
275             else
276             {
277                 // If we get here, we couldn't find a launcher activity.
278                 throw new MojoFailureException( "Could not find a launcher activity in manifest" );
279             }
280         }
281         else
282         {
283             // If we get here we couldn't find any Activity
284             throw new MojoFailureException( "Could not find any activity in manifest" );
285         }
286     }

288     /**
289      * Executes the "Launcher activity".
290      *
291      * @param info A {@link LauncherInfo}.
292      * @throws MojoFailureException
293      * @throws MojoExecutionException
294      */
295     private void launch( final LauncherInfo info ) throws MojoExecutionException, MojoFailureException
296     {
297         final String command;
298 
299         command = String.format( "am start %s-n %s/%s", parsedDebug ? "-D " : "", info.packageName, info.activity );
300 
301         doWithDevices( new DeviceCallback()
302         {
303             @Override
304             public void doWithDevice( IDevice device ) throws MojoExecutionException, MojoFailureException
305             {
306                 String deviceLogLinePrefix = DeviceHelper.getDeviceLogLinePrefix( device );
307 
308                 try
309                 {
310                     getLog().info( deviceLogLinePrefix + "Attempting to start " + info.packageName + "/" 
311                             + info.activity );
312 
313                     CollectingOutputReceiver shellOutput = new CollectingOutputReceiver();
314                     device.executeShellCommand( command, shellOutput );
315                     if ( shellOutput.getOutput().contains( "Error" ) )
316                     {
317                         throw new MojoFailureException( shellOutput.getOutput() );
318                     }
319                 }
320                 catch ( IOException ex )
321                 {
322                     throw new MojoFailureException( deviceLogLinePrefix + "Input/Output error", ex );
323                 }
324                 catch ( TimeoutException ex )
325                 {
326                     throw new MojoFailureException( deviceLogLinePrefix + "Command timeout", ex );
327                 }
328                 catch ( AdbCommandRejectedException ex )
329                 {
330                     throw new MojoFailureException( deviceLogLinePrefix + "ADB rejected the command", ex );
331                 }
332                 catch ( ShellCommandUnresponsiveException ex )
333                 {
334                     throw new MojoFailureException( deviceLogLinePrefix + "Unresponsive command", ex );
335                 }
336             }
337         } );
338     }
