56  /**
57   * Runs the first Activity shown in the top-level launcher as determined by its Intent filters.
58   * <p>
59   * Android provides a component-based architecture, which means that there is no "main" function which serves as an
60   * entry point to the APK. There's an homogeneous collection of Activity(es), Service(s), Receiver(s), etc.
61   * </p>
62   * <p>
63   * The Android top-level launcher (whose purpose is to allow users to launch other applications) uses the Intent
64   * resolution mechanism to determine which Activity(es) to show to the end user. Such activities are identified by at
65   * least:
66   * <ul>
67   * <li>Action type: <code>android.intent.action.MAIN</code></li>
68   * <li>Category: <code>android.intent.category.LAUNCHER</code></li>
69   * </ul>
70   * </p>
71   * <p>And are declared in <code>AndroidManifest.xml</code> as such:</p>
72   * <pre>
73   * &lt;activity android:name=".ExampleActivity"&gt;
74   *     &lt;intent-filter&gt;
75   *         &lt;action android:name="android.intent.action.MAIN" /&gt;
76   *         &lt;category android:name="android.intent.category.LAUNCHER" /&gt;
77   *     &lt;/intent-filter&gt;
78   * &lt;/activity&gt;
79   * </pre>
80   * <p>
81   * This {@link Mojo} will try to to launch the first activity of this kind found in <code>AndroidManifest.xml</code>. In
82   * case multiple activities satisfy the requirements listed above only the first declared one is run. In case there are
83   * no "Launcher activities" declared in the manifest or no activities declared at all, this goal aborts throwing an
84   * error.
85   * </p>
86   * <p>
87   * The device parameter is taken into consideration so potentially the Activity found is started on all attached
88   * devices. The application will NOT be deployed and running will silently fail if the application is not deployed.
89   * </p>
90   *
91   * @author Lorenzo Villani <lorenzo@villani.me>
92   * @author Manfred Mosr <manfred@simpligility.com>
93   * @see "http://developer.android.com/guide/topics/fundamentals.html"
94   * @see "http://developer.android.com/guide/topics/intents/intents-filters.html"
95   */
96  @Mojo( name = "run" )
97  public class RunMojo extends AbstractAndroidMojo
98  {
99  
100     /**
101      * <p>The configuration for the run goal can be set up in the plugin configuration in the pom file as:</p>
102      * <pre>
103      * &lt;run&gt;
104      *     &lt;debug&gt;true|false|portnumber&lt;/debug&gt;
105      * &lt;/run&gt;
106      * </pre>
107      * <p>The <code>&lt;debug&gt;</code> parameter is optional and defaults to false. Numeric values like 5432 are 
108      * parsed as port number. 
109      * <p>The debug parameter can also be configured as property in the pom or settings file
110      * <pre>
111      * &lt;properties&gt;
112      *     &lt;android.run.debug&gt;true&lt;/android.run.debug&gt;
113      * &lt;/properties&gt;
114      * </pre>
115      * or from command-line with parameter <code>-Dandroid.run.debug=true</code>.</p>
116      */
117     @Parameter
118     @ConfigPojo
119     private Run run;
120 
121     /**
122      * Debug parameter for the the run goal. If true, the device or emulator will pause execution of the process at
123      * startup to wait for a debugger to connect. Also see the "run" parameter documentation. Default value is false.
124      * If the value is numeric, it is treated as a port number to forward the JDWP protocol
125      * of the launched process to.
126      */
127     @Parameter( property = "android.run.debug" )
128     protected String runDebug;

170     /**
171      * {@inheritDoc}
172      */
173     @Override
174     public void execute() throws MojoExecutionException, MojoFailureException
175     {
176         if ( project.getPackaging().equals( APK ) )
177         {
178             try
179             {
180                 LauncherInfo launcherInfo;
181 
182                 launcherInfo = getLauncherActivity();
183 
184                 ConfigHandler configHandler = new ConfigHandler( this, this.session, this.execution );
185                 configHandler.parseConfiguration();
186 
187                 launch( launcherInfo );
188             }
189             catch ( Exception ex )
190             {
191                 getLog().info( "Unable to run launcher Activity" );
192                 getLog().debug( ex );
193             }
194         }
195         else
196         {
197             getLog().info( "Project packaging is not apk, skipping run action." );
198         }
199     }

205     /**
206      * Gets the first "Launcher" Activity by running an XPath query on <code>AndroidManifest.xml</code>.
207      *
208      * @return A {@link LauncherInfo}
209      * @throws MojoFailureException
210      * @throws ParserConfigurationException
211      * @throws IOException
212      * @throws SAXException
213      * @throws XPathExpressionException
214      */
215     private LauncherInfo getLauncherActivity()
216             throws ParserConfigurationException, SAXException, IOException, XPathExpressionException,
217             MojoFailureException
218     {
219         Document document;
220         DocumentBuilder documentBuilder;
221         DocumentBuilderFactory documentBuilderFactory;
222         Object result;
223         XPath xPath;
224         XPathExpression xPathExpression;
225         XPathFactory xPathFactory;
226 
227         //
228         // Setup JAXP stuff
229         //
230         documentBuilderFactory = DocumentBuilderFactory.newInstance();
231 
232         documentBuilder = documentBuilderFactory.newDocumentBuilder();
233 
234         document = documentBuilder.parse( destinationManifestFile );
235 
236         xPathFactory = XPathFactory.newInstance();
237 
238         xPath = xPathFactory.newXPath();
239 
240         xPathExpression = xPath.compile(
241                 "//manifest/application/activity/intent-filter[action[@name=\"android.intent.action.MAIN\"] "
242                 + "and category[@name=\"android.intent.category.LAUNCHER\"]]/.." );
243 
244         //
245         // Run XPath query
246         //
247         result = xPathExpression.evaluate( document, XPathConstants.NODESET );
248 
249         if ( result instanceof NodeList )
250         {
251             NodeList activities;
252 
253             activities = ( NodeList ) result;
254 
255             if ( activities.getLength() > 0 )
256             {
257                 // Grab the first declared Activity
258                 LauncherInfo launcherInfo;
259 
260                 launcherInfo = new LauncherInfo();
261                 String activityName = activities.item( 0 ).getAttributes().getNamedItem( "android:name" )
262                         .getNodeValue();
263 
264                 if ( ! activityName.contains( "." ) )
265                 {
266                     activityName = "." + activityName;
267                 }
268 
269                 if ( activityName.startsWith( "." ) )
270                 {
271                     String packageName = document.getElementsByTagName( "manifest" ).item( 0 ).getAttributes()
272                             .getNamedItem( "package" ).getNodeValue();
273                     activityName = packageName + activityName;
274                 }
275 
276                 launcherInfo.activity = activityName;
277 
278                 launcherInfo.packageName = renameManifestPackage != null
279                     ? renameManifestPackage
280                     : document.getDocumentElement().getAttribute( "package" ).toString();
281 
282                 return launcherInfo;
283             }
284             else
285             {
286                 // If we get here, we couldn't find a launcher activity.
287                 throw new MojoFailureException( "Could not find a launcher activity in manifest" );
288             }
289         }
290         else
291         {
292             // If we get here we couldn't find any Activity
293             throw new MojoFailureException( "Could not find any activity in manifest" );
294         }
295     }

297     /**
298      * Executes the "Launcher activity".
299      *
300      * @param info A {@link LauncherInfo}.
301      * @throws MojoFailureException
302      * @throws MojoExecutionException
303      */
304     private void launch( final LauncherInfo info ) throws MojoExecutionException, MojoFailureException
305     {
306         final String command;
307         
308         final int debugPort = findDebugPort();
309 
310         command = String.format( "am start %s-n %s/%s", debugPort >= 0 ? "-D " : "", info.packageName, info.activity );
311 
312         doWithDevices( new DeviceCallback()
313         {
314             @Override
315             public void doWithDevice( IDevice device ) throws MojoExecutionException, MojoFailureException
316             {
317                 String deviceLogLinePrefix = DeviceHelper.getDeviceLogLinePrefix( device );
318 
319                 try
320                 {
321                     getLog().info( deviceLogLinePrefix + "Attempting to start " + info.packageName + "/" 
322                             + info.activity );
323 
324                     CollectingOutputReceiver shellOutput = new CollectingOutputReceiver();
325                     device.executeShellCommand( command, shellOutput );
326                     if ( shellOutput.getOutput().contains( "Error" ) )
327                     {
328                         throw new MojoFailureException( shellOutput.getOutput() );
329                     }
330                     if ( debugPort > 0 ) 
331                     {
332                         CollectingOutputReceiver processOutput = new CollectingOutputReceiver();
333                         device.executeShellCommand( "ps", processOutput );
334                         BufferedReader r = new BufferedReader( new StringReader( processOutput.getOutput() ) );
335                         int pid = -1;
336                         for ( ;; ) 
337                         {
338                             String line = r.readLine();
339                             if ( line == null )
340                             {
341                                 break;
342                             }
343                             if ( line.endsWith( info.packageName ) )
344                             {
345                                 String[] values = line.split( " +" );
346                                 if ( values.length > 2 )
347                                 {
348                                     pid = Integer.valueOf( values[1] );
349                                     break;
350                                 }
351                             }
352                         }
353                         r.close();
354                         if ( pid == -1 )
355                         {
356                             throw new MojoFailureException( "Cannot find stated process " + info.packageName );
357                         }
358                         getLog().info(
359                             deviceLogLinePrefix + "Process " + debugPort + " launched"
360                         );
361                         try 
362                         {
363                             createForward( device, debugPort, pid );
364                             getLog().info(
365                                 deviceLogLinePrefix + "Debugger listening on " + debugPort
366                             );
367                         } 
368                         catch ( Exception ex ) 
369                         {
370                             throw new MojoFailureException( 
371                                 "Cannot create forward tcp: " + debugPort 
372                                     + " jdwp: " + pid, ex 
373                             );
374                         }
375                     }
376                 }
377                 catch ( IOException ex )
378                 {
379                     throw new MojoFailureException( deviceLogLinePrefix + "Input/Output error", ex );
380                 }
381                 catch ( TimeoutException ex )
382                 {
383                     throw new MojoFailureException( deviceLogLinePrefix + "Command timeout", ex );
384                 }
385                 catch ( AdbCommandRejectedException ex )
386                 {
387                     throw new MojoFailureException( deviceLogLinePrefix + "ADB rejected the command", ex );
388                 }
389                 catch ( ShellCommandUnresponsiveException ex )
390                 {
391                     throw new MojoFailureException( deviceLogLinePrefix + "Unresponsive command", ex );
392                 }
393             }
394         } );
395     }
396     
397     private static void createForward( IDevice device, int debugPort, int pid ) 
398     throws ClassNotFoundException, NoSuchMethodException, IllegalAccessException, InvocationTargetException
399     {
400         Method m = Class.forName( "com.android.ddmlib.AdbHelper" ).
401             getDeclaredMethod(
402                 "createForward", InetSocketAddress.class, 
403                 device.getClass(), String.class, String.class
404             );
405         m.setAccessible( true );
406         m.invoke(
407             null, AndroidDebugBridge.getSocketAddress(), device,
408             String.format( "tcp:%d", debugPort ), String.format( "jdwp:%d", pid )
409         );
410     }
411 
412     private int findDebugPort()
413     {
414         int debugPort;
415         if ( "true".equals( parsedDebug ) )
416         {
417             debugPort = 0;
418         }
419         else
420         {
421             try
422             {
423                 debugPort = Integer.parseInt( parsedDebug );
424             }
425             catch ( NumberFormatException ex )
426             {
427                 debugPort = -1;
428             }
429         }
430         return debugPort;
431     }
