106     /**
107      * Execute the mojo by parsing the confign and actually doing the zipalign.
108      *
109      * @throws MojoExecutionException
110      */
111     public void execute() throws MojoExecutionException, MojoFailureException
112     {
113 
114         // If we're not on a supported packaging with just skip (Issue 87)
115         // http://code.google.com/p/maven-android-plugin/issues/detail?id=87
116         if ( ! SUPPORTED_PACKAGING_TYPES.contains( project.getPackaging() ) )
117         {
118             getLog().info( "Skipping zipalign on " + project.getPackaging() );
119             return;
120         }
121 
122         ConfigHandler configHandler = new ConfigHandler( this );
123         configHandler.parseConfiguration();
124 
125         parsedInputApk = FilenameUtils.separatorsToSystem( parsedInputApk );
126         parsedOutputApk = FilenameUtils.separatorsToSystem( parsedOutputApk );
127 
128         getLog().debug( "skip:" + parsedSkip );
129         getLog().debug( "verbose:" + parsedVerbose );
130         getLog().debug( "inputApk:" + parsedInputApk );
131         getLog().debug( "outputApk:" + parsedOutputApk );
132 
133         if ( parsedSkip )
134         {
135             getLog().info( "Skipping zipalign" );
136         }
137         else
138         {
139             boolean outputToSameFile = sameOutputAsInput();
140 
141             CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
142             executor.setLogger( this.getLog() );
143 
144             String command = getAndroidSdk().getZipalignPath();
145 
146             List<String> parameters = new ArrayList<String>();
147             if ( parsedVerbose )
148             {
149                 parameters.add( "-v" );
150             }
151             parameters.add( "-f" ); // force overwriting existing output file
152             parameters.add( "4" ); // byte alignment has to be 4!
153             parameters.add( parsedInputApk );
154             String outputApk = outputToSameFile ? getTemporaryOutputApkFilename() : parsedOutputApk;
155             parameters.add( outputApk );
156 
157             try
158             {
159                 getLog().info( "Running command: " + command );
160                 getLog().info( "with parameters: " + parameters );
161                 executor.executeCommand( command, parameters );
162 
163                 if ( FileUtils.fileExists( outputApk ) )
164                 {
165                     if ( outputToSameFile )
166                     {
167                         // No needs to attach zipaligned apk to artifacts
168                         try
169                         {
170                             FileUtils.rename( new File( outputApk ),  new File( parsedInputApk ) );
171                         }
172                         catch ( IOException e )
173                         {
174                             getLog().error( "Failed to replace original apk with aligned "
175                                     + getFullPathWithName( outputApk ), e );
176                         }
177                     }
178                     else
179                     {
180                         // Attach the resulting artifact (Issue 88)
181                         // http://code.google.com/p/maven-android-plugin/issues/detail?id=88
182                         projectHelper.attachArtifact( project, APK, "aligned", new File( outputApk ) );
183                         getLog().info( "Attach " + getFullPathWithName( outputApk ) + " to the project" );
184                     }
185                 }
186                 else
187                 {
188                     getLog().error( "Cannot attach " + getFullPathWithName( outputApk ) + " to the project"
189                             + " - The file does not exist" );
190                 }
191             }
192             catch ( ExecutionException e )
193             {
194                 throw new MojoExecutionException( "", e );
195             }
196         }
197     }

199     private String getFullPathWithName( String filename )

204     private boolean sameOutputAsInput()
205     {
206         return getFullPathWithName( parsedInputApk ).equals( getFullPathWithName( parsedOutputApk ) );
207     }

209     // zipalign doesn't allow output file to be same as input
210     private String getTemporaryOutputApkFilename()
