17  package com.jayway.maven.plugins.android.standalonemojos;
18  
19  import com.jayway.maven.plugins.android.AbstractAndroidMojo;
20  import org.apache.maven.plugin.MojoExecutionException;
21  import org.apache.maven.plugin.MojoFailureException;
22  import org.apache.maven.plugins.annotations.Mojo;
23  import org.apache.maven.plugins.annotations.ResolutionScope;
24  
25  /**
26   * Deploys all directly declared dependencies of <code>&lt;type&gt;apk&lt;/type&gt;</code> in this project's pom.<br/>
27   * Usually used in a project with instrumentation tests, to deploy the apk to test onto the device before running the
28   * deploying and running the instrumentation tests apk.<br/>
29   * Automatically performed when running <code>mvn integration-test</code> (or <code>mvn install</code>) on a project
30   * with instrumentation tests.
31   *
32   * @author hugo.josefson@jayway.com
33   */
34  @Mojo( name = "deploy-dependencies", requiresDependencyResolution = ResolutionScope.RUNTIME )
35  public class DeployDependenciesMojo extends AbstractAndroidMojo
36  {
37  
38      /**
39       *
40       * @throws MojoExecutionException
41       * @throws MojoFailureException
42       */
43      public void execute() throws MojoExecutionException, MojoFailureException
44      {
45          deployDependencies();
46      }
47  
48  
49  }
