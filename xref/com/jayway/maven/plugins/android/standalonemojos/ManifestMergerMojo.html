39      /**
40       * Configuration for the manifest-update goal.
41       * <p>
42       * You can configure this mojo to update the following basic manifestMerger attributes:

46       * <code>android:versionCode</code> on the <code>manifestMerger</code> element.

48       * <p>
49       * You can configure attributes in the plugin configuration like so
50       * <p/>
51       * <pre>
52       *   &lt;plugin&gt;
53       *     &lt;groupId&gt;com.jayway.maven.plugins.android.generation2&lt;/groupId&gt;
54       *     &lt;artifactId&gt;android-maven-plugin&lt;/artifactId&gt;
55       *     &lt;executions&gt;
56       *       &lt;execution&gt;
57       *         &lt;id&gt;merge-manifest&lt;/id&gt;
58       *         &lt;goals&gt;

102     /**
103      * Update the <code>android:versionCode</code> attribute automatically from the project version
104      * e.g 3.2.1 will become version code 3002001. As described in this blog post
105      * http://www.simpligility.com/2010/11/release-version-management-for-your-android-application/
106      * but done without using resource filtering. The value is exposed via the project property
107      * property <code>android.manifest.versionCodeUpdateFromVersion</code> and the resulting value
108      * as <code>android.manifest.versionCode</code>.
109      * For the purpose of generating the versionCode, if a version element is missing it is presumed to be 0.
110      * The maximum values for the version increment and version minor values are 999,
111      * the version major should be no larger than 2000.  Any other suffixes do not
112      * participate in the version code generation.
113      */
114     @Parameter( property = "android.manifest.versionCodeUpdateFromVersion", defaultValue = "false" )
115     protected Boolean manifestVersionCodeUpdateFromVersion = false;

143     /**
144      * @throws org.apache.maven.plugin.MojoExecutionException
145      * @throws org.apache.maven.plugin.MojoFailureException
146      */
147     public void execute() throws MojoExecutionException, MojoFailureException
148     {
149         if ( ! AndroidExtension.isAndroidPackaging( project.getPackaging() ) )
150         {
151             return; // skip, not an android project.
152         }
153 
154         if ( androidManifestFile == null )
155         {
156             getLog().debug( "skip, no androidmanifest.xml defined (androidManifestFile rare case)" );
157             return; // skip, no androidmanifest.xml defined (rare case)
158         }
159 
160         parseConfiguration();
161 
162         getLog().info( "Attempting to update manifest " + androidManifestFile );
163         getLog().debug( "    usesSdk=" + parsedUsesSdk );
164         getLog().debug( "    versionName=" + parsedVersionName );
165         getLog().debug( "    versionCode=" + parsedVersionCode );
166         getLog().debug( "    usesSdk=" + parsedUsesSdk );
167         getLog().debug( "    versionCodeUpdateFromVersion=" + parsedVersionCodeUpdateFromVersion );
168         getLog().debug( "    mergeLibraries=" + parsedMergeLibraries );
169         getLog().debug( "    mergeReportFile=" + parsedMergeReportFile );
170 
171         if ( ! androidManifestFile.exists() )
172         {
173             return; // skip, no AndroidManifest.xml file found.
174         }
175 
176         getLog().debug( "Using manifest merger V2" );
177         manifestMergerV2();
178     }

180     private void parseConfiguration()
181     {
182         // manifestMerger element found in plugin config in pom
183         if ( manifestMerger != null )
184         {
185             if ( StringUtils.isNotEmpty( manifestMerger.getVersionName() ) )
186             {
187                 parsedVersionName = manifestMerger.getVersionName();
188             }
189             else
190             {
191                 parsedVersionName = manifestVersionName;
192             }
193             if ( manifestMerger.getVersionCode() != null )
194             {
195                 parsedVersionCode = manifestMerger.getVersionCode();
196             }
197             else
198             {
199                 parsedVersionCode = manifestVersionCode;
200             }
201             if ( manifestMerger.getVersionCodeUpdateFromVersion() != null )
202             {
203                 parsedVersionCodeUpdateFromVersion = manifestMerger.getVersionCodeUpdateFromVersion();
204             }
205             else
206             {
207                 parsedVersionCodeUpdateFromVersion = manifestVersionCodeUpdateFromVersion;
208             }
209             if ( manifestMerger.getUsesSdk() != null )
210             {
211                 parsedUsesSdk = manifestMerger.getUsesSdk();
212             }
213             else
214             {
215                 parsedUsesSdk = manifestUsesSdk;
216             }
217             if ( manifestMerger.getMergeLibraries() != null )
218             {
219                 parsedMergeLibraries = manifestMerger.getMergeLibraries();
220             }
221             else
222             {
223                 parsedMergeLibraries = manifestMergeLibraries;
224             }
225             if ( manifestMerger.getMergeReportFile() != null )
226             {
227                 parsedMergeReportFile = manifestMerger.getMergeReportFile();
228             }
229             else
230             {
231                 parsedMergeReportFile = manifestMergeReportFile;
232             }
233         }
234         else
235         {
236             parsedVersionName = manifestVersionName;
237             parsedVersionCode = manifestVersionCode;
238             parsedUsesSdk = manifestUsesSdk;
239             parsedVersionCodeUpdateFromVersion = manifestVersionCodeUpdateFromVersion;
240             parsedMergeLibraries = manifestMergeLibraries;
241             parsedMergeReportFile = manifestMergeReportFile;
242         }
243     }

246     public void manifestMergerV2() throws MojoExecutionException, MojoFailureException
247     {
248         AndroidBuilder builder = new AndroidBuilder( project.toString(), "created by Android Maven Plugin",
249                 new MavenILogger( getLog() ), false );
250 
251         String minSdkVersion = null;
252         String targetSdkVersion = null;
253         int versionCode;
254         if ( parsedUsesSdk != null )
255         {
256             minSdkVersion = parsedUsesSdk.getMinSdkVersion();
257             targetSdkVersion = parsedUsesSdk.getTargetSdkVersion();
258         }
259         if ( parsedVersionCodeUpdateFromVersion )
260         {
261             versionCode = generateVersionCodeFromVersionName( parsedVersionName );
262         }
263         else
264         {
265             versionCode = parsedVersionCode;
266         }
267         List<MavenManifestDependency> manifestDependencies = new ArrayList<MavenManifestDependency>();
268 
269         if ( parsedMergeLibraries )
270         {
271             final Set<Artifact> allArtifacts = project.getDependencyArtifacts();
272             Set<Artifact> dependencyArtifacts = getArtifactResolverHelper().getFilteredArtifacts( allArtifacts );
273 
274             for ( Artifact dependency : dependencyArtifacts )
275             {
276                 final File unpackedLibFolder = getUnpackedLibFolder( dependency );
277                 final File manifestFile = new File( unpackedLibFolder, SdkConstants.FN_ANDROID_MANIFEST_XML );
278                 if ( manifestFile.exists() )
279                 {
280                     manifestDependencies.add( new MavenManifestDependency( manifestFile,
281                             manifestFile.getAbsolutePath(), new ArrayList<MavenManifestDependency>() ) );
282                 }
283             }
284         }
285 
286         builder.mergeManifests(
287                 androidManifestFile, new ArrayList<File>(), manifestDependencies, "",
288                 versionCode, parsedVersionName,
289                 minSdkVersion, targetSdkVersion, null,
290                 destinationManifestFile.getPath(), ManifestMerger2.MergeType.APPLICATION,
291                 new HashMap<String, String>(), parsedMergeReportFile );
292 
293     }

295     private int generateVersionCodeFromVersionName( String versionName )
296     {
297         String[] versionNameDigits = versionName.replaceAll( "[^0-9.]", "" ).split( "\\." );
298 
299         int versionCode = 0;
300         for ( int i = 0; i < versionNameDigits.length; i++ )
301         {
302             double digitMultiplayer = Math.pow( 10, i * NUMBER_OF_DIGITS_FOR_VERSION_POSITION );
303             String versionDigit = versionNameDigits[versionNameDigits.length - i - 1 ];
304             versionCode += Integer.valueOf( versionDigit ) * digitMultiplayer;
305         }
306         return versionCode;
307     }
