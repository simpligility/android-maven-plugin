494     /**
495      * Actually plays tests.
496      * 
497      * @throws MojoExecutionException
498      *             if exercising app threw an exception and isIgnoreTestFailures is false..
499      * @throws MojoFailureException
500      *             if exercising app failed and isIgnoreTestFailures is false.
501      */
502     protected void exerciseApp() throws MojoExecutionException, MojoFailureException
503     {
504 
505         getLog().debug( "Parsed values for Android Monkey invocation: " );
506         getLog().debug( "seed:" + parsedSeed );
507 
508         DeviceCallback instrumentationTestExecutor = new DeviceCallback()
509         {
510             @Override
511             public void doWithDevice( final IDevice device ) throws MojoExecutionException, MojoFailureException
512             {
513                 String deviceLogLinePrefix = DeviceHelper.getDeviceLogLinePrefix( device );
514 
515                 MonkeyTestRunner monkeyTestRunner = new MonkeyTestRunner( parsedEventCount, device );
516 
517                 monkeyTestRunner.setRunName( "ui monkey tests" );
518                 if ( parsedSeed != null )
519                 {
520                     monkeyTestRunner.setSeed( parsedSeed );
521                 }
522                 if ( parsedPercentTouch != null )
523                 {
524                     monkeyTestRunner.setPercentTouch( parsedPercentTouch );
525                 }
526                 if ( parsedPercentMotion != null )
527                 {
528                     monkeyTestRunner.setPercentTouch( parsedPercentMotion );
529                 }
530                 if ( parsedPercentTrackball != null )
531                 {
532                     monkeyTestRunner.setPercentTrackball( parsedPercentTrackball );
533                 }
534                 if ( parsedPercentNav != null )
535                 {
536                     monkeyTestRunner.setPercentNav( parsedPercentNav );
537                 }
538                 if ( parsedPercentMajorNav != null )
539                 {
540                     monkeyTestRunner.setPercentMajorNav( parsedPercentMajorNav );
541                 }
542                 if ( parsedPercentSyskeys != null )
543                 {
544                     monkeyTestRunner.setPercentSyskeys( parsedPercentSyskeys );
545                 }
546                 if ( parsedPercentAppswitch != null )
547                 {
548                     monkeyTestRunner.setPercentAppswitch( parsedPercentAppswitch );
549                 }
550                 if ( parsedPercentAnyevent != null )
551                 {
552                     monkeyTestRunner.setPercentAnyEvent( parsedPercentAnyevent );
553                 }
554                 if ( parsedPackages != null )
555                 {
556                     monkeyTestRunner.setPackages( parsedPackages );
557                 }
558                 if ( parsedCategories != null )
559                 {
560                     monkeyTestRunner.setCategories( parsedCategories );
561                 }
562                 monkeyTestRunner.setDebugNoEvents( parsedDebugNoEvents );
563                 monkeyTestRunner.setHprof( parsedHprof );
564                 monkeyTestRunner.setIgnoreCrashes( parsedIgnoreCrashes );
565                 monkeyTestRunner.setIgnoreTimeouts( parsedIgnoreTimeouts );
566                 monkeyTestRunner.setIgnoreSecurityExceptions( parsedIgnoreSecurityExceptions );
567                 monkeyTestRunner.setKillProcessAfterError( parsedKillProcessAfterError );
568                 monkeyTestRunner.setMonitorNativeCrash( parsedMonitorNativeCrashes );
569 
570                 getLog().info( deviceLogLinePrefix + "Running ui monkey tests" );
571                 try
572                 {
573                     AndroidTestRunListener testRunListener = new AndroidTestRunListener( project, device );
574                     monkeyTestRunner.run( testRunListener );
575                     if ( testRunListener.hasFailuresOrErrors() && !isIgnoreTestFailures() )
576                     {
577                         throw new MojoFailureException( deviceLogLinePrefix + "Tests failed on device." );
578                     }
579                     if ( testRunListener.testRunFailed() )
580                     {
581                         throw new MojoFailureException( deviceLogLinePrefix + "Test run failed to complete: "
582                                 + testRunListener.getTestRunFailureCause() );
583                     }
584                     if ( testRunListener.threwException() && !isIgnoreTestFailures() )
585                     {
586                         throw new MojoFailureException( deviceLogLinePrefix + testRunListener.getExceptionMessages() );
587                     }
588                 }
589                 catch ( TimeoutException e )
590                 {
591                     throw new MojoExecutionException( deviceLogLinePrefix + "timeout", e );
592                 }
593                 catch ( AdbCommandRejectedException e )
594                 {
595                     throw new MojoExecutionException( deviceLogLinePrefix + "adb command rejected", e );
596                 }
597                 catch ( ShellCommandUnresponsiveException e )
598                 {
599                     throw new MojoExecutionException( deviceLogLinePrefix + "shell command " + "unresponsive", e );
600                 }
601                 catch ( IOException e )
602                 {
603                     throw new MojoExecutionException( deviceLogLinePrefix + "IO problem", e );
604                 }
605             }
606         };
607 
608         doWithDevices( instrumentationTestExecutor );
609     }

711         @Override
712         public void testRunStarted( String runName, int tCount )
713         {
714 
715             this.testCount = tCount;
716             getLog().info( deviceLogLinePrefix + INDENT + "Run started: " + runName + ", " + testCount + " tests:" );
717 
718             if ( parsedCreateReport )
719             {
720                 try
721                 {
722                     DocumentBuilderFactory fact = DocumentBuilderFactory.newInstance();
723                     DocumentBuilder parser = null;
724                     parser = fact.newDocumentBuilder();
725                     junitReport = parser.newDocument();
726                     Node testSuitesNode = junitReport.createElement( TAG_TESTSUITES );
727                     junitReport.appendChild( testSuitesNode );
728                     testSuiteNode = junitReport.createElement( TAG_TESTSUITE );
729                     NamedNodeMap testSuiteAttributes = testSuiteNode.getAttributes();
730                     Attr nameAttr = junitReport.createAttribute( ATTR_TESTSUITE_NAME );
731                     nameAttr.setValue( runName );
732                     testSuiteAttributes.setNamedItem( nameAttr );
733                     Attr hostnameAttr = junitReport.createAttribute( ATTR_TESTSUITE_HOSTNAME );
734                     hostnameAttr.setValue( DeviceHelper.getDescriptiveName( device ) );
735                     testSuiteAttributes.setNamedItem( hostnameAttr );
736                     Node propertiesNode = junitReport.createElement( TAG_PROPERTIES );
737                     Node propertyNode;
738                     NamedNodeMap propertyAttributes;
739                     Attr propNameAttr;
740                     Attr propValueAttr;
741                     for ( Map.Entry< Object, Object > systemProperty : System.getProperties().entrySet() )
742                     {
743                         propertyNode = junitReport.createElement( TAG_PROPERTY );
744                         propertyAttributes = propertyNode.getAttributes();
745                         propNameAttr = junitReport.createAttribute( ATTR_PROPERTY_NAME );
746                         propNameAttr.setValue( systemProperty.getKey().toString() );
747                         propertyAttributes.setNamedItem( propNameAttr );
748                         propValueAttr = junitReport.createAttribute( ATTR_PROPERTY_VALUE );
749                         propValueAttr.setValue( systemProperty.getValue().toString() );
750                         propertyAttributes.setNamedItem( propValueAttr );
751                         propertiesNode.appendChild( propertyNode );
752                     }
753                     Map< String, String > deviceProperties = device.getProperties();
754                     for ( Map.Entry< String, String > deviceProperty : deviceProperties.entrySet() )
755                     {
756                         propertyNode = junitReport.createElement( TAG_PROPERTY );
757                         propertyAttributes = propertyNode.getAttributes();
758                         propNameAttr = junitReport.createAttribute( ATTR_PROPERTY_NAME );
759                         propNameAttr.setValue( deviceProperty.getKey() );
760                         propertyAttributes.setNamedItem( propNameAttr );
761                         propValueAttr = junitReport.createAttribute( ATTR_PROPERTY_VALUE );
762                         propValueAttr.setValue( deviceProperty.getValue() );
763                         propertyAttributes.setNamedItem( propValueAttr );
764                         propertiesNode.appendChild( propertyNode );
765                     }
766                     testSuiteNode.appendChild( propertiesNode );
767                     testSuitesNode.appendChild( testSuiteNode );
768                 }
769                 catch ( ParserConfigurationException e )
770                 {
771                     threwException = true;
772                     exceptionMessages.append( "Failed to create document" );
773                     exceptionMessages.append( e.getMessage() );
774                 }
775             }

956         /**
957          * Parse a trace string for the message in it. Assumes that the message is located after ":" and before "\r\n".
958          * 
959          * @param trace
960          * @return message or empty string
961          */
962         private String parseForMessage( String trace )
963         {
964             if ( StringUtils.isNotBlank( trace ) )
965             {
966                 String newline = "\r\n";
967                 // if there is message like
968                 // junit.junit.framework.AssertionFailedError ... there is no
969                 // message
970                 int messageEnd = trace.indexOf( newline );
971                 boolean hasMessage = !trace.startsWith( "junit." ) && messageEnd > 0;
972                 if ( hasMessage )
973                 {
974                     int messageStart = trace.indexOf( ":" ) + 2;
975                     if ( messageStart > messageEnd )
976                     {
977                         messageEnd = trace.indexOf( newline + "at" );
978                         // match start of stack trace "\r\nat org.junit....."
979                         if ( messageStart > messageEnd )
980                         {
981                             // ':' wasn't found in message but in stack trace
982                             messageStart = 0;
983                         }
984                     }
985                     return trace.substring( messageStart, messageEnd );
986                 }
987                 else
988                 {
989                     return StringUtils.EMPTY;
990                 }
991             }
992             else
993             {
994                 return StringUtils.EMPTY;
995             }
996         }

998         /**
999          * Parse a trace string for the exception class. Assumes that it is the start of the trace and ends at the first
1000          * ":".
1001          * 
1002          * @param trace
1003          * @return Exception class as string or empty string
1004          */
1005         private String parseForException( String trace )
1006         {
1007             if ( StringUtils.isNotBlank( trace ) )
1008             {
1009                 return trace.substring( 0, trace.indexOf( ":" ) );
1010             }
1011             else
1012             {
1013                 return StringUtils.EMPTY;
1014             }

1018          * Write the junit report xml file.
1019          */
1020         private void writeJunitReportToFile()
1021         {
1022             TransformerFactory xfactory = TransformerFactory.newInstance();
1023             Transformer xformer = null;
1024             try
1025             {
1026                 xformer = xfactory.newTransformer();
1027             }
1028             catch ( TransformerConfigurationException e )
1029             {
1030                 e.printStackTrace();
1031             }
1032             Source source = new DOMSource( junitReport );
1033 
1034             FileWriter writer = null;
1035             try
1036             {
1037                 String directory = new StringBuilder().append( project.getBuild().getDirectory() )
1038                         .append( "/surefire-reports" ).toString();
1039 
1040                 FileUtils.forceMkdir( new File( directory ) );
1041 
1042                 String fileName = new StringBuilder().append( directory ).append( "/TEST-" )
1043                         .append( DeviceHelper.getDescriptiveName( device ) ).append( ".xml" ).toString();
1044                 File reportFile = new File( fileName );
1045                 writer = new FileWriter( reportFile );
1046                 Result result = new StreamResult( writer );
1047 
1048                 xformer.transform( source, result );
1049                 getLog().info( deviceLogLinePrefix + "Report file written to " + reportFile.getAbsolutePath() );
1050             }
1051             catch ( IOException e )
1052             {
1053                 threwException = true;
1054                 exceptionMessages.append( "Failed to write test report file" );
1055                 exceptionMessages.append( e.getMessage() );
1056             }
1057             catch ( TransformerException e )
1058             {
1059                 threwException = true;
1060                 exceptionMessages.append( "Failed to transform document to write to test report file" );
1061                 exceptionMessages.append( e.getMessage() );
1062             }
1063             finally
1064             {
1065                 IOUtils.closeQuietly( writer );
1066             }
1067         }
