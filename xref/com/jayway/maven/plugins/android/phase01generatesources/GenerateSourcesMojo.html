270     /**
271      * Generates the sources.
272      *
273      * @throws MojoExecutionException if it fails.
274      * @throws MojoFailureException if it fails.
275      */
276     public void execute() throws MojoExecutionException, MojoFailureException
277     {
278         // If the current POM isn't an Android-related POM, then don't do
279         // anything. This helps work with multi-module projects.
280         if ( ! isCurrentProjectAndroid() )
281         {
282             return;
283         }
284 
285         validateStandardLocations();
286 
287         try
288         {
289             targetDirectory.mkdirs();
290             copyManifest();
291 
292             // Check for a deprecated AAR > APKLIB artifact combination
293             if ( warnOnApklibDependencies )
294             {
295                 checkForApklibDependencies();
296             }
297 
298             // TODO Do we really want to continue supporting APKSOURCES? How long has it been deprecated
299             extractSourceDependencies();
300 
301             // Extract the apklib and aar dependencies into unpacked-libs so that they can be referenced in the build.
302             extractLibraryDependencies();
303 
304             // Copy project assets to combinedAssets so that aapt has a single assets folder to load.
305             copyFolder( assetsDirectory, combinedAssets );
306 
307             final String[] relativeAidlFileNames1 = findRelativeAidlFileNames( aidlSourceDirectory );
308             final String[] relativeAidlFileNames2 = findRelativeAidlFileNames( extractedDependenciesJavaSources );
309             final Map<String, String[]> relativeApklibAidlFileNames = new HashMap<String, String[]>();
310 
311             if ( !isInstrumentationTest() )
312             {
313                 // Only add transitive APKLIB deps if we are building an APK and not an instrumentation test apk.
314                 for ( Artifact artifact : getTransitiveDependencyArtifacts( APKLIB ) )
315                 {
316                     final File libSourceFolder = getUnpackedApkLibSourceFolder( artifact );
317                     final String[] apklibAidlFiles = findRelativeAidlFileNames( libSourceFolder );
318                     relativeApklibAidlFileNames.put( artifact.getId(), apklibAidlFiles );
319                 }
320             }
321 
322             mergeManifests();
323 
324             checkPackagesForDuplicates();
325             checkForConflictingLayouts();
326             generateR();
327             generateBuildConfig();
328 
329             // When compiling AIDL for this project,
330             // make sure we compile AIDL for dependencies as well.
331             // This is so project A, which depends on project B, can
332             // use AIDL info from project B in its own AIDL
333             final Map<File, String[]> files = new HashMap<File, String[]>();
334             files.put( aidlSourceDirectory, relativeAidlFileNames1 );
335             files.put( extractedDependenciesJavaSources, relativeAidlFileNames2 );
336 
337             if ( !isInstrumentationTest() )
338             {
339                 // Only add transitive APKLIB deps if we are building an APK and not an instrumentation test apk.
340                 for ( Artifact artifact : getTransitiveDependencyArtifacts( APKLIB ) )
341                 {
342                     final File unpackedLibSourceFolder = getUnpackedApkLibSourceFolder( artifact );
343                     files.put( unpackedLibSourceFolder, relativeApklibAidlFileNames.get( artifact.getId() ) );
344                 }
345             }
346             generateAidlFiles( files );
347         }
348         catch ( MojoExecutionException e )
349         {
350             getLog().error( "Error when generating sources.", e );
351             throw e;
352         }
353     }

355     /**
356      * Performs some level of validation on the configured files and folders in light of the change
357      * to AndroidStudio/Gradle style folder structures instead of the original Ant/Eclipse structures.
358      *
359      * Essentially we will be noisy if we see an artifact that looks like before 
360      * Android Maven Plugin 4 and is not explicitly configured.
361      */
362     private void validateStandardLocations() throws MojoExecutionException
363     {
364         boolean hasNonStandardStructure = false;
365         if ( androidManifestFilePre4.exists() && !androidManifestFilePre4.equals( androidManifestFile ) )
366         {
367             getLog().warn( "Non-standard location of AndroidManifest.xml file found, but not configured:\n " 
368                 + androidManifestFilePre4 + "\nMove to the standard location src/main/AndroidManifest.xml\n"
369                 + "Or configure androidManifestFile. \n" );
370             hasNonStandardStructure = true;
371         }
372         if ( resourceDirectoryPre4.exists() && !resourceDirectoryPre4.equals( resourceDirectory ) )
373         {
374             getLog().warn( "Non-standard location of Android res folder found, but not configured:\n " 
375                 + resourceDirectoryPre4 + "\nMove to the standard location src/main/res/\n"
376                 + "Or configure resourceDirectory. \n" );
377             hasNonStandardStructure = true;
378         }
379         if ( assetsDirectoryPre4.exists() && !assetsDirectoryPre4.equals( assetsDirectory ) )
380         {
381             getLog().warn( "Non-standard location assets folder found, but not configured:\n " 
382                 + assetsDirectoryPre4 + "\nMove to the standard location src/main/assets/\n"
383                 + "Or configure assetsDirectory. \n" );
384             hasNonStandardStructure = true;
385         }
386         if ( nativeLibrariesDirectoryPre4.exists() && !nativeLibrariesDirectoryPre4.equals( nativeLibrariesDirectory ) )
387         {
388             getLog().warn( "Non-standard location native libs folder found, but not configured:\n " 
389                 + nativeLibrariesDirectoryPre4 + "\nMove to the standard location src/main/libs/\n"
390                 + "Or configure nativeLibrariesDirectory. \n" );
391             hasNonStandardStructure = true;
392         }
393 
394         if ( hasNonStandardStructure && failOnNonStandardStructure )
395         {
396             throw new MojoExecutionException(
397                 "\n\nFound files or folders in non-standard locations in the project!\n"
398                 + "....This might be a side-effect of a migration to Android Maven Plugin 4+.\n"
399                 + "....Please observe the warnings for specific files and folders above.\n"
400                 + "....Ideally you should restructure your project.\n"
401                 + "....Alternatively add explicit configuration overrides for files or folders.\n"
402                 + "....Finally you could set failOnNonStandardStructure to false, potentially "
403                 + "resulting in other failures.\n\n\n"
404             );
405         }
406     }

408     /**
409      * Copy the AndroidManifest.xml from androidManifestFile to destinationManifestFile
410      *
411      * @throws MojoExecutionException
412      */
413     protected void copyManifest() throws MojoExecutionException
414     {
415         getLog().debug( "copyManifest: " + androidManifestFile + " -> " + destinationManifestFile );
416         if ( androidManifestFile == null )
417         {
418             getLog().debug( "Manifest copying disabled. Using default manifest only" );
419             return;
420         }
421 
422         try
423         {
424             DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
425             DocumentBuilder db = dbf.newDocumentBuilder();
426             Document doc = db.parse( androidManifestFile );
427             Source source = new DOMSource( doc );
428 
429             TransformerFactory xfactory = TransformerFactory.newInstance();
430             Transformer xformer = xfactory.newTransformer();
431             xformer.setOutputProperty( OutputKeys.OMIT_XML_DECLARATION, "yes" );
432 
433             FileWriter writer = null;
434             try
435             {
436                 destinationManifestFile.getParentFile().mkdirs();
437 
438                 writer = new FileWriter( destinationManifestFile, false );
439                 if ( doc.getXmlEncoding() != null && doc.getXmlVersion() != null )
440                 {
441                     String xmldecl = String.format( "<?xml version=\"%s\" encoding=\"%s\"?>%n",
442                             doc.getXmlVersion(), doc.getXmlEncoding() );
443 
444                     writer.write( xmldecl );
445                 }
446                 Result result = new StreamResult( writer );
447                 xformer.transform( source, result );
448                 getLog().info( "Manifest copied from " + androidManifestFile + " to " + destinationManifestFile );
449             }
450             finally
451             {
452                 IOUtils.closeQuietly( writer );
453             }
454         }
455         catch ( Exception e )
456         {
457             getLog().error( "Error during copyManifest" );
458             throw new MojoExecutionException( "Error during copyManifest", e );
459         }
460     }

462     /**
463      * Extract the source dependencies.
464      *
465      * @throws MojoExecutionException if it fails.
466      */
467     protected void extractSourceDependencies() throws MojoExecutionException
468     {
469         for ( Artifact artifact : getDirectDependencyArtifacts() )
470         {
471             String type = artifact.getType();
472             if ( type.equals( APKSOURCES ) )
473             {
474                 getLog().debug( "Detected apksources dependency " + artifact + " with file " + artifact.getFile()
475                         + ". Will resolve and extract..." );
476 
477                 final File apksourcesFile = resolveArtifactToFile( artifact );
478                 getLog().debug( "Extracting " + apksourcesFile + "..." );
479                 extractApksources( apksourcesFile );
480             }
481         }
482 
483         if ( extractedDependenciesJavaResources.exists() )
484         {
485             projectHelper.addResource( project, extractedDependenciesJavaResources.getAbsolutePath(), null, null );
486             project.addCompileSourceRoot( extractedDependenciesJavaSources.getAbsolutePath() );
487         }
488     }

491      * @deprecated Support <code>APKSOURCES</code> artifacts has been deprecated. Use APKLIB instead.

515         }
516         catch ( ArchiverException e )
517         {
518             throw new MojoExecutionException( "ArchiverException while extracting " + apksourcesFile.getAbsolutePath()
519                     + ". Message: " + e.getLocalizedMessage(), e );
520         }
521     }

523     private void extractLibraryDependencies() throws MojoExecutionException
524     {
525         final Collection<Artifact> artifacts = getTransitiveDependencyArtifacts(
526                 EXCLUDED_DEPENDENCY_SCOPES_FOR_EXTRACTION );
527 
528         getLog().info( "Extracting libs" );
529 
530         // The only library dependencies we want included in an instrumentation test are APK and AARs.
531         // Any APKLIB classes have already been compiled into the APK.
532         final boolean instrumentationTest = isInstrumentationTest();
533 
534         for ( Artifact artifact : artifacts )
535         {
536             final String type = artifact.getType();
537             if ( type.equals( APKLIB ) && !instrumentationTest )
538             {
539                 getLog().info( "Extracting apklib " + artifact.getArtifactId() + "..." );
540                 extractApklib( artifact );
541             }
542             else if ( type.equals( AAR ) )
543             {
544                 getLog().info( "Extracting aar " + artifact.getArtifactId() + "..." );
545                 extractAarLib( artifact );
546             }
547             else if ( type.equals( APK ) )
548             {
549                 getLog().info( "Extracting apk " + artifact.getArtifactId() + "..." );
550                 extractApkClassesJar( artifact );
551             }
552             else
553             {
554                 getLog().debug( "Not extracting " + artifact.getArtifactId() + "..." );
555             }
556         }
557     }

559     /**
560      * Extracts ApkLib and adds the assets and apklib sources and resources to the build.
561      */
562     private void extractApklib( Artifact apklibArtifact ) throws MojoExecutionException
563     {
564         getUnpackedLibHelper().extractApklib( apklibArtifact );
565 
566         // Copy the assets to the the combinedAssets folder.
567         // Add the apklib source and resource to the compile.
568         // NB apklib sources are added to compileSourceRoot because we may need to compile against them.
569         //    This means the apklib classes will be compiled into target/classes and packaged with this build.
570         copyFolder( getUnpackedLibAssetsFolder( apklibArtifact ), combinedAssets );
571 
572         final File apklibSourceFolder = getUnpackedApkLibSourceFolder( apklibArtifact );
573         final List<String> resourceExclusions = Arrays.asList( "**/*.java", "**/*.aidl" );
574         projectHelper.addResource( project, apklibSourceFolder.getAbsolutePath(), null, resourceExclusions );
575         project.addCompileSourceRoot( apklibSourceFolder.getAbsolutePath() );
576     }

578     /**
579      * Extracts AarLib and if this is an APK build then adds the assets and resources to the build.
580      */
581     private void extractAarLib( Artifact aarArtifact ) throws MojoExecutionException
582     {
583         getUnpackedLibHelper().extractAarLib( aarArtifact );
584 
585         // Copy the assets to the the combinedAssets folder, but only if an APK build.
586         // Ie we only want to package assets that we own.
587         // Assets should only live within their owners or the final APK.
588         if ( isAPKBuild() )
589         {
590             copyFolder( getUnpackedLibAssetsFolder( aarArtifact ), combinedAssets );
591         }
592 
593         // Aar lib resources should only be included if we are building an apk.
594         // So we need to extract them into a folder that we then add to the resource classpath.
595         if ( isAPKBuild() )
596         {
597             // (If we are including SYSTEM_SCOPE artifacts when considering resources for APK packaging)
598             // then adding the AAR resources to the project would have them added twice.
599             getLog().debug( "Not adding AAR resources to resource classpath : " + aarArtifact );
600         }
601     }

603     /**
604      * Copies a dependent APK jar over the top of the placeholder created for it in AarMavenLifeCycleParticipant.
605      *
606      * This is necessary because we need the classes of the APK added to the compile classpath.
607      * NB APK dependencies are uncommon as they should really only be used in a project that tests an apk.
608      *
609      * @param artifact  APK dependency for this project whose classes will be copied over.
610      */
611     private void extractApkClassesJar( Artifact artifact ) throws MojoExecutionException
612     {
613         final File apkClassesJar = getUnpackedLibHelper().getJarFileForApk( artifact );
614         final File unpackedClassesJar = getUnpackedLibHelper().getUnpackedClassesJar( artifact );
615         try
616         {
617             FileUtils.copyFile( apkClassesJar, unpackedClassesJar );
618         }
619         catch ( IOException e )
620         {
621             throw new MojoExecutionException(
622                     "Could not copy APK classes jar " + apkClassesJar + " to " + unpackedClassesJar, e );

854     private void generateR() throws MojoExecutionException
855     {
856         getLog().info( "Generating R file for " + project.getArtifact() );
857 
858         genDirectory.mkdirs();
859 
860         final AaptPackageCommandBuilder commandBuilder = AaptCommandBuilder
861                 .packageResources( getLog() )
862                 .makePackageDirectories()
863                 .setResourceConstantsFolder( genDirectory )
864                 .forceOverwriteExistingFiles()
865                 .disablePngCrunching()
866                 .generateRIntoPackage( customPackage )
867                 .setPathToAndroidManifest( destinationManifestFile )
868                 .addResourceDirectoryIfExists( resourceDirectory )
869                 .addResourceDirectoriesIfExists( getResourceOverlayDirectories() )
870                     // Need to include any AAR or APKLIB dependencies when generating R because if any local
871                     // resources directly reference dependent resources then R generation will crash.
872                 .addResourceDirectoriesIfExists( getLibraryResourceFolders() )
873                 .autoAddOverlay()
874                 .addRawAssetsDirectoryIfExists( combinedAssets )
875                 .addExistingPackageToBaseIncludeSet( getAndroidSdk().getAndroidJar() )
876                 .addConfigurations( configurations )
877                 .addExtraArguments( aaptExtraArgs )
878                 .setVerbose( aaptVerbose )
879                     // We need to generate R.txt for all projects as it needs to be consumed when generating R class.
880                     // It also needs to be consumed when packaging aar.
881                 .generateRTextFile( targetDirectory )
882                 // If a proguard file is defined then output Proguard options to it.
883                 .setProguardOptionsOutputFile( proguardFile )
884                 .makeResourcesNonConstant( AAR.equals( project.getArtifact().getType() ) );
885 
886         getLog().debug( getAndroidSdk().getAaptPath() + " " + commandBuilder.toString() );
887         try
888         {
889             final CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
890             executor.setLogger( getLog() );
891             executor.setCaptureStdOut( true );
892             final List<String> commands = commandBuilder.build();
893             executor.executeCommand( getAndroidSdk().getAaptPath(), commands, project.getBasedir(), false );
894         }
895         catch ( ExecutionException e )
896         {
897             throw new MojoExecutionException( "", e );
898         }
899 
900         final ResourceClassGenerator resGenerator = new ResourceClassGenerator( this, targetDirectory, genDirectory );
901         generateCorrectRJavaForApklibDependencies( resGenerator );
902         generateCorrectRJavaForAarDependencies( resGenerator );
903 
904         getLog().info( "Adding R gen folder to compile classpath: " + genDirectory );
905         project.addCompileSourceRoot( genDirectory.getAbsolutePath() );
906     }

974      * Executes aapt to generate the R class for the given apklib.
975      *
976      * @param apklibArtifact    apklib for which to generate the R class.
977      * @throws MojoExecutionException if it fails.
978      */
979     private void generateRForApkLibDependency( Artifact apklibArtifact ) throws MojoExecutionException
980     {
981         final File unpackDir = getUnpackedLibFolder( apklibArtifact );
982         getLog().debug( "Generating incomplete R file for apklib: " + apklibArtifact.getGroupId()
983                 + ":" + apklibArtifact.getArtifactId() );
984         final File apklibManifest = new File( unpackDir, "AndroidManifest.xml" );
985         final File apklibResDir = new File( unpackDir, "res" );

988         final Set<Artifact> apklibDeps = getDependencyResolver()
989                 .getLibraryDependenciesFor( this.session, this.repositorySystem, apklibArtifact );
990         getLog().debug( "apklib=" + apklibArtifact + "  dependencies=" + apklibDeps );
991         for ( Artifact dependency : apklibDeps )
992         {
993             // Add in the resources that are dependencies of the apklib.
994             final String extension = dependency.getType();
995             final File dependencyResDir = getUnpackedLibResourceFolder( dependency );
996             if ( ( extension.equals( APKLIB ) || extension.equals( AAR ) ) && dependencyResDir.exists() )
997             {
998                 dependenciesResDirectories.add( dependencyResDir );
999             }
1000         }
1001 
1002         // Create combinedAssets for this apklib dependency - can't have multiple -A args
1003         final File apklibCombAssets = new File( getUnpackedLibFolder( apklibArtifact ), "combined-assets" );
1004         for ( Artifact dependency : apklibDeps )
1005         {
1006             // Accumulate assets for dependencies of the apklib (if they exist).
1007             final String extension = dependency.getType();
1008             final File dependencyAssetsDir = getUnpackedLibAssetsFolder( dependency );
1009             if ( ( extension.equals( APKLIB ) || extension.equals( AAR ) ) )
1010             {
1011                 copyFolder( dependencyAssetsDir, apklibCombAssets );
1012             }
1013         }
1014         // Overlay the apklib dependency assets (if they exist)
1015         final File apkLibAssetsDir = getUnpackedLibAssetsFolder( apklibArtifact );
1016         copyFolder( apkLibAssetsDir, apklibCombAssets );
1017 
1018         final CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
1019         executor.setLogger( getLog() );
1020 
1021         final AaptCommandBuilder commandBuilder = AaptCommandBuilder
1022                 .packageResources( getLog() )
1023                 .makeResourcesNonConstant()
1024                 .makePackageDirectories()
1025                 .setResourceConstantsFolder( genDirectory )
1026                 .generateRIntoPackage( extractPackageNameFromAndroidManifest( apklibManifest ) )
1027                 .setPathToAndroidManifest( apklibManifest )
1028                 .addResourceDirectoryIfExists( apklibResDir )
1029                 .addResourceDirectoriesIfExists( dependenciesResDirectories )
1030                 .autoAddOverlay()
1031                 .addRawAssetsDirectoryIfExists( apklibCombAssets )
1032                 .addExistingPackageToBaseIncludeSet( getAndroidSdk().getAndroidJar() )
1033                 .addConfigurations( configurations )
1034                 .addExtraArguments( aaptExtraArgs )
1035                 .setVerbose( aaptVerbose )
1036                 // We need to generate R.txt for all projects as it needs to be consumed when generating R class.
1037                 // It also needs to be consumed when packaging aar.
1038                 .generateRTextFile( unpackDir );
1039 
1040         getLog().debug( getAndroidSdk().getAaptPath() + " " + commandBuilder.toString() );
1041         try
1042         {
1043             executor.setCaptureStdOut( true );
1044             final List<String> commands = commandBuilder.build();

1249     /**
1250      * Given a map of source directories to list of AIDL (relative) filenames within each,
1251      * runs the AIDL compiler for each, such that all source directories are available to
1252      * the AIDL compiler.
1253      *
1254      * @param files Map of source directory File instances to the relative paths to all AIDL files within
1255      * @throws MojoExecutionException If the AIDL compiler fails
1256      */
1257     private void generateAidlFiles( Map<File /*sourceDirectory*/, String[] /*relativeAidlFileNames*/> files )
1258             throws MojoExecutionException
1259     {
1260         List<String> protoCommands = new ArrayList<String>();
1261         protoCommands.add( "-p" + getAndroidSdk().getPathForFrameworkAidl() );
1262 
1263         genDirectoryAidl.mkdirs();
1264         getLog().info( "Adding AIDL gen folder to compile classpath: " + genDirectoryAidl );
1265         project.addCompileSourceRoot( genDirectoryAidl.getPath() );
1266         Set<File> sourceDirs = files.keySet();
1267         for ( File sourceDir : sourceDirs )
1268         {
1269             protoCommands.add( "-I" + sourceDir );
1270         }
1271         for ( File sourceDir : sourceDirs )
1272         {
1273             for ( String relativeAidlFileName : files.get( sourceDir ) )
1274             {
1275                 File targetDirectory = new File( genDirectoryAidl, new File( relativeAidlFileName ).getParent() );
1276                 targetDirectory.mkdirs();
1277 
1278                 final String shortAidlFileName = new File( relativeAidlFileName ).getName();
1279                 final String shortJavaFileName = shortAidlFileName.substring( 0, shortAidlFileName.lastIndexOf( "." ) )
1280                                                  + ".java";
1281                 final File aidlFileInSourceDirectory = new File( sourceDir, relativeAidlFileName );
1282 
1283                 List<String> commands = new ArrayList<String>( protoCommands );
1284                 commands.add( aidlFileInSourceDirectory.getAbsolutePath() );
1285                 commands.add( new File( targetDirectory, shortJavaFileName ).getAbsolutePath() );
1286                 try
1287                 {
1288                     CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
1289                     executor.setLogger( this.getLog() );
1290                     executor.setCaptureStdOut( true );
1291                     executor.executeCommand( getAndroidSdk().getAidlPath(), commands, project.getBasedir(),
1292                             false );
1293                 }
1294                 catch ( ExecutionException e )
1295                 {
1296                     throw new MojoExecutionException( "", e );
1297                 }
1298             }
1299         }
1300     }

1303     {
1304         final FileRetriever retriever = new FileRetriever( "**/*.aidl" );
1305         final String[] relativeAidlFileNames = retriever.getFileNames( sourceDirectory );

1313         }
