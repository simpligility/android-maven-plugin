85       * <p>
86       * Override default merging. You must have SDK Tools r20+
87       * </p>
88       * 
89       * <p>
90       * <b>IMPORTANT:</b> The resource plugin needs to be disabled for the
91       * <code>process-resources</code> phase, so the "default-resources"
92       * execution must be added. Without this the non-merged manifest will get
93       * re-copied to the build directory.
94       * </p>
95       * 
96       * <p>
97       * The <code>androidManifestFile</code> should also be configured to pull
98       * from the build directory so that later phases will pull the merged
99       * manifest file.
100      * </p>
101      * <p>
102      * Example POM Setup:
103      * </p>
104      * 
105      * <pre>
106      * &lt;build&gt;
107      *     ...
108      *     &lt;plugins&gt;
109      *         ...
110      *         &lt;plugin&gt;
111      *             &lt;artifactId&gt;maven-resources-plugin&lt;/artifactId&gt;
112      *             &lt;version&gt;2.6&lt;/version&gt;
113      *             &lt;executions&gt;
114      *                 &lt;execution&gt;
115      *                     &lt;phase&gt;initialize&lt;/phase&gt;
116      *                     &lt;goals&gt;
117      *                         &lt;goal&gt;resources&lt;/goal&gt;
118      *                     &lt;/goals&gt;
119      *                 &lt;/execution&gt;
120      *                 <b>&lt;execution&gt;
121      *                     &lt;id&gt;default-resources&lt;/id&gt;
122      *                     &lt;phase&gt;DISABLED&lt;/phase&gt;
123      *                 &lt;/execution&gt;</b>
124      *             &lt;/executions&gt;
125      *         &lt;/plugin&gt;
126      *         &lt;plugin&gt;
127      *             &lt;groupId&gt;com.jayway.maven.plugins.android.generation2&lt;/groupId&gt;
128      *             &lt;artifactId&gt;android-maven-plugin&lt;/artifactId&gt;
129      *             &lt;configuration&gt;
130      *                 <b>&lt;androidManifestFile&gt;
131      *                     ${project.build.directory}/AndroidManifest.xml
132      *                 &lt;/androidManifestFile&gt;
133      *                 &lt;mergeManifests&gt;true&lt;/mergeManifests&gt;</b>
134      *             &lt;/configuration&gt;
135      *             &lt;extensions&gt;true&lt;/extensions&gt;
136      *         &lt;/plugin&gt;
137      *         ...
138      *     &lt;/plugins&gt;
139      *     ...
140      * &lt;/build&gt;
141      * </pre>
142      * <p>
143      * You can filter the pre-merged APK manifest. One important note about Eclipse, Eclipse will
144      * replace the merged manifest with a filtered pre-merged version when the project is refreshed.
145      * If you want to review the filtered merged version then you will need to open it outside Eclipse
146      * without refreshing the project in Eclipse. 
147      * </p>
148      * <pre>
149      * &lt;resources&gt;
150      *     &lt;resource&gt;
151      *         &lt;targetPath&gt;${project.build.directory}&lt;/targetPath&gt;
152      *         &lt;filtering&gt;true&lt;/filtering&gt;
153      *         &lt;directory&gt;${basedir}&lt;/directory&gt;
154      *         &lt;includes&gt;
155      *             &lt;include&gt;AndroidManifest.xml&lt;/include&gt;
156      *         &lt;/includes&gt;
157      *     &lt;/resource&gt;
158      * &lt;/resources&gt;
159      * </pre>
160      * 
161      * @parameter expression="${android.mergeManifests}" default-value="false"
162      */
163     protected boolean mergeManifests;

188     /**
189      * Generates the source.
190      *
191      * @throws MojoExecutionException if it fails.
192      * @throws MojoFailureException if it fails.
193      */
194     public void execute() throws MojoExecutionException, MojoFailureException
195     {
196 
197         // If the current POM isn't an Android-related POM, then don't do
198         // anything.  This helps work with multi-module projects.
199         if ( ! isCurrentProjectAndroid() )
200         {
201             return;
202         }
203 
204         try
205         {
206             targetDirectory.mkdirs();
207             copyManifest();
208             // TODO Do we really want to continue supporting APKSOURCES? How long has it been deprecated
209             extractSourceDependencies();
210 
211             // Extract the apklib and aar dependencies into unpacked-libs so that they can be referenced in the build.
212             extractLibraryDependencies();
213 
214             // Copy project assets to combinedAssets so that aapt has a single assets folder to load.
215             copyFolder( assetsDirectory, combinedAssets );
216 
217             final String[] relativeAidlFileNames1 = findRelativeAidlFileNames( sourceDirectory );
218             final String[] relativeAidlFileNames2 = findRelativeAidlFileNames( extractedDependenciesJavaSources );
219             final Map<String, String[]> relativeApklibAidlFileNames = new HashMap<String, String[]>();
220 
221             for ( Artifact artifact : getTransitiveDependencyArtifacts( APKLIB ) )
222             {
223                 final File libSourceFolder = getUnpackedApkLibSourceFolder( artifact );
224                 final String[] apklibAidlFiles = findRelativeAidlFileNames( libSourceFolder );
225                 relativeApklibAidlFileNames.put( artifact.getId(), apklibAidlFiles );
226             }
227 
228             mergeManifests();
229             generateR();
230             generateBuildConfig();
231 
232             // When compiling AIDL for this project,
233             // make sure we compile AIDL for dependencies as well.
234             // This is so project A, which depends on project B, can
235             // use AIDL info from project B in its own AIDL
236             Map<File, String[]> files = new HashMap<File, String[]>();
237             files.put( sourceDirectory, relativeAidlFileNames1 );
238             files.put( extractedDependenciesJavaSources, relativeAidlFileNames2 );
239             for ( Artifact artifact : getTransitiveDependencyArtifacts( APKLIB ) )
240             {
241                 final File unpackedLibSourceFolder = getUnpackedApkLibSourceFolder( artifact );
242                 files.put( unpackedLibSourceFolder, relativeApklibAidlFileNames.get( artifact.getId() ) );
243             }
244             generateAidlFiles( files );
245         }
246         catch ( MojoExecutionException e )
247         {
248             getLog().error( "Error when generating sources.", e );
249             throw e;
250         }
251     }

253     /**
254      * Copy the AndroidManifest.xml from sourceManifestFile to androidManifestFile
255      *
256      * @throws MojoExecutionException
257      */
258     protected void copyManifest() throws MojoExecutionException
259     {
260         getLog().debug( "copyManifest: " + sourceManifestFile + " -> " + androidManifestFile );
261         if ( sourceManifestFile == null )
262         {
263             getLog().info( "Manifest copying disabled. Using default manifest only" );
264             return;
265         }
266 
267         try
268         {
269             DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
270             DocumentBuilder db = dbf.newDocumentBuilder();
271             Document doc = db.parse( sourceManifestFile );
272             Source source = new DOMSource( doc );
273 
274             TransformerFactory xfactory = TransformerFactory.newInstance();
275             Transformer xformer = xfactory.newTransformer();
276             xformer.setOutputProperty( OutputKeys.OMIT_XML_DECLARATION, "yes" );
277 
278             FileWriter writer = null;
279             try
280             {
281                 androidManifestFile.getParentFile().mkdirs();
282 
283                 writer = new FileWriter( androidManifestFile, false );
284                 if ( doc.getXmlEncoding() != null && doc.getXmlVersion() != null )
285                 {
286                     String xmldecl = String.format( "<?xml version=\"%s\" encoding=\"%s\"?>%n",
287                             doc.getXmlVersion(), doc.getXmlEncoding() );
288 
289                     writer.write( xmldecl );
290                 }
291                 Result result = new StreamResult( writer );
292                 xformer.transform( source, result );
293                 getLog().info( "Manifest copied from " + sourceManifestFile + " to " + androidManifestFile );
294             }
295             finally
296             {
297                 IOUtils.closeQuietly( writer );
298             }
299         }
300         catch ( Exception e )
301         {
302             getLog().error( "Error during copyManifest" );
303             throw new MojoExecutionException( "Error during copyManifest", e );
304         }
305     }

307     /**
308      * Extract the source dependencies.
309      *
310      * @throws MojoExecutionException if it fails.
311      */
312     protected void extractSourceDependencies() throws MojoExecutionException
313     {
314         for ( Artifact artifact : getDirectDependencyArtifacts() )
315         {
316             String type = artifact.getType();
317             if ( type.equals( APKSOURCES ) )
318             {
319                 getLog().debug( "Detected apksources dependency " + artifact + " with file " + artifact.getFile()
320                         + ". Will resolve and extract..." );
321 
322                 Artifact resolvedArtifact = AetherHelper
323                         .resolveArtifact( artifact, repoSystem, repoSession, projectRepos );
324 
325                 File apksourcesFile = resolvedArtifact.getFile();
326 
327                 // When the artifact is not installed in local repository, but rather part of the current reactor,
328                 // resolve from within the reactor. (i.e. ../someothermodule/target/*)
329                 if ( ! apksourcesFile.exists() )
330                 {
331                     apksourcesFile = resolveArtifactToFile( artifact );
332                 }
333 
334                 // When using maven under eclipse the artifact will by default point to a directory, which isn't
335                 // correct. To work around this we'll first try to get the archive from the local repo, and only if it
336                 // isn't found there we'll do a normal resolve.
337 
338                 if ( apksourcesFile.isDirectory() )
339                 {
340                     apksourcesFile = resolveArtifactToFile( artifact );
341                 }
342                 getLog().debug( "Extracting " + apksourcesFile + "..." );
343                 extractApksources( apksourcesFile );
344             }
345         }
346 
347         if ( extractedDependenciesJavaResources.exists() )
348         {
349             projectHelper.addResource( project, extractedDependenciesJavaResources.getAbsolutePath(), null, null );
350             project.addCompileSourceRoot( extractedDependenciesJavaSources.getAbsolutePath() );
351         }
352     }

354     private void extractApksources( File apksourcesFile ) throws MojoExecutionException
355     {
356         if ( apksourcesFile.isDirectory() )
357         {
358             getLog().warn( "The apksources artifact points to '" + apksourcesFile
359                     + "' which is a directory; skipping unpacking it." );
360             return;
361         }
362         final UnArchiver unArchiver = new ZipUnArchiver( apksourcesFile )
363         {
364             @Override
365             protected Logger getLogger()
366             {
367                 return new ConsoleLogger( Logger.LEVEL_DEBUG, "dependencies-unarchiver" );
368             }
369         };
370         extractedDependenciesDirectory.mkdirs();
371         unArchiver.setDestDirectory( extractedDependenciesDirectory );
372         try
373         {
374             unArchiver.extract();
375         }
376         catch ( ArchiverException e )
377         {
378             throw new MojoExecutionException( "ArchiverException while extracting " + apksourcesFile.getAbsolutePath()
379                     + ". Message: " + e.getLocalizedMessage(), e );
380         }
381     }

383     private void extractLibraryDependencies() throws MojoExecutionException
384     {
385         final Collection<Artifact> artifacts = getTransitiveDependencyArtifacts();
386         getLog().info( "Extracting libs" );
387 
388         for ( Artifact artifact : artifacts )
389         {
390             final String type = artifact.getType();
391             if ( type.equals( APKLIB ) )
392             {
393                 getLog().info( "Extracting apklib " + artifact.getArtifactId() + "..." );
394                 extractApklib( artifact );
395             }
396             else if ( type.equals( AAR ) )
397             {
398                 getLog().info( "Extracting aar " + artifact.getArtifactId() + "..." );
399                 extractAarLib( artifact );
400             }
401             else if ( type.equals( APK ) )
402             {
403                 getLog().info( "Extracting apk " + artifact.getArtifactId() + "..." );
404                 extractApkClassesJar( artifact );
405             }
406             else
407             {
408                 getLog().debug( "Not extracting " + artifact.getArtifactId() + "..." );
409             }
410         }
411     }

413     /**
414      * Extracts ApkLib and adds the assets and apklib sources and resources to the build.
415      */
416     private void extractApklib( Artifact apklibArtifact ) throws MojoExecutionException
417     {
418         getBuildHelper().extractApklib( apklibArtifact );
419 
420         // Copy the assets to the the combinedAssets folder.
421         // Add the apklib source and resource to the compile.
422         // NB apklib sources are added to compileSourceRoot because we may need to compile against them.
423         //    This means the apklib classes will be compiled into target/classes and packaged with this build.
424         copyFolder( getUnpackedLibAssetsFolder( apklibArtifact ), combinedAssets );
425 
426         final File apklibSourceFolder = getUnpackedApkLibSourceFolder( apklibArtifact );
427         final List<String> resourceExclusions = Arrays.asList( "**/*.java", "**/*.aidl" );
428         projectHelper.addResource( project, apklibSourceFolder.getAbsolutePath(), null, resourceExclusions );
429         project.addCompileSourceRoot( apklibSourceFolder.getAbsolutePath() );
430     }

432     /**
433      * Extracts AarLib and if this is an APK build then adds the assets and resources to the build.
434      */
435     private void extractAarLib( Artifact aarArtifact ) throws MojoExecutionException
436     {
437         getBuildHelper().extractAarLib( aarArtifact );
438 
439         // Copy the assets to the the combinedAssets folder, but only if an APK build.
440         // Ie we only want to package assets that we own.
441         // Assets should only live within their owners or the final APK.
442         if ( isAPKBuild() )
443         {
444             copyFolder( getUnpackedLibAssetsFolder( aarArtifact ), combinedAssets );
445         }
446 
447         // Aar lib resources should only be included if we are building an apk.
448         // So we need to extract them into a folder that we then add to the resource classpath.
449         if ( isAPKBuild() )
450         {
451             // Extract the resources from the AAR classes and add them as a resource path to the project.
452             final File aarClassesJar = getUnpackedAarClassesJar( aarArtifact );
453             final ZipExtractor extractor = new ZipExtractor( getLog() );
454             final File javaResourcesFolder = getUnpackedAarJavaResourcesFolder( aarArtifact );
455             extractor.extract( aarClassesJar, javaResourcesFolder, ".class" );
456             projectHelper.addResource( project, javaResourcesFolder.getAbsolutePath(), null, null );
457             getLog().debug( "Added AAR resources to resource classpath : " + javaResourcesFolder );
458         }
459     }

461     /**
462      * Copies a dependent APK jar over the top of the placeholder created for it in AarMavenLifeCycleParticipant.
463      *
464      * This is necessary because we need the classes of the APK added to the compile classpath.
465      * NB APK dependencies are uncommon as they should really only be used in a project that tests an apk.
466      *
467      * @param artifact  APK dependency for this project whose classes will be copied over.
468      */
469     private void extractApkClassesJar( Artifact artifact ) throws MojoExecutionException
470     {
471         final File apkClassesJar = getBuildHelper().getJarFileForApk( artifact );
472         final File unpackedClassesJar = getBuildHelper().getUnpackedClassesJar( artifact );
473         try
474         {
475             FileUtils.copyFile( apkClassesJar, unpackedClassesJar );
476         }
477         catch ( IOException e )
478         {
479             throw new MojoExecutionException(
480                     "Could not copy APK classes jar " + apkClassesJar + " to " + unpackedClassesJar, e );
481         }
482         getLog().debug( "Copied APK classes jar into compile classpath : " + unpackedClassesJar );
483 
484     }

486     private void generateR() throws MojoExecutionException
487     {
488         getLog().info( "Generating R file for " + project.getArtifact() );
489 
490         genDirectory.mkdirs();
491 
492         final File[] overlayDirectories = getResourceOverlayDirectories();
493         getLog().debug( "Resource overlay folders : " + Arrays.asList( overlayDirectories ) );
494 
495         final CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
496         executor.setLogger( this.getLog() );
497 
498         final List<String> commands = new ArrayList<String>();
499         commands.add( "package" );
500 
501         commands.add( "-f" );
502         commands.add( "--no-crunch" );
503 
504         // inputs
505         commands.add( "-I" );
506         commands.add( getAndroidSdk().getAndroidJar().getAbsolutePath() );
507 
508         commands.add( "-M" );
509         commands.add( androidManifestFile.getAbsolutePath() );
510 
511         // NB AndroidBuilder only adds a single folder - presumably it contains a merge of all resources.
512         for ( File resOverlayDir : overlayDirectories )
513         {
514             if ( resOverlayDir != null && resOverlayDir.exists() )
515             {
516                 getLog().debug( "Adding resource overlay folder : " + resOverlayDir );
517                 commands.add( "-S" );
518                 commands.add( resOverlayDir.getAbsolutePath() );
519             }
520         }
521         if ( resourceDirectory.exists() )
522         {
523             getLog().debug( "Adding resource folder : " + resourceDirectory );
524             commands.add( "-S" );
525             commands.add( resourceDirectory.getAbsolutePath() );
526         }
527 
528         // Need to include any AAR or APKLIB dependencies when generating R because if any local
529         // resources directly reference dependent resources then R generation will crash.
530         addLibraryResourceFolders( commands );
531 
532         // NB aapt only accepts a single assets parameter - combinedAssets is a merge of all assets
533         if ( combinedAssets.exists() )
534         {
535             getLog().debug( "Adding assets folder : " + combinedAssets );
536             commands.add( "-A" );
537             commands.add( combinedAssets.getAbsolutePath() );
538         }
539 
540         // outputs
541         commands.add( "-m" );
542         commands.add( "-J" );
543         commands.add( genDirectory.getAbsolutePath() );
544 
545         // Write the output to an optional location
546         // Used by AndroidBuilder but not by us.
547         // final File optionalOutputLocation = some file;
548         // getLog().debug( "Using default package : " + optionalOutputLocation );
549         // commands.add( "-F" );
550         // commands.add( optionalOutputLocation.getAbsolutePath() );
551 
552         // If a proguard file is defined then output Proguard options to it.
553         if ( proguardFile != null )
554         {
555             final File parentFolder = proguardFile.getParentFile();
556             if ( parentFolder != null )
557             {
558                 parentFolder.mkdirs();
559             }
560             getLog().debug( "Adding proguard file : " + proguardFile );
561             commands.add( "-G" );
562             commands.add( proguardFile.getAbsolutePath() );
563         }
564 
565         if ( StringUtils.isNotBlank( customPackage ) )
566         {
567             getLog().debug( "Adding custom-package : " + customPackage );
568             commands.add( "--custom-package" );
569             commands.add( customPackage );
570         }
571 
572         if ( AAR.equals( project.getArtifact().getType() ) )
573         {
574             getLog().debug( "Adding non-constant-id" );
575             commands.add( "--non-constant-id" );
576         }
577 
578         for ( String aaptExtraArg : aaptExtraArgs )
579         {
580             getLog().debug( "Adding aapt arg : " + aaptExtraArg );
581             commands.add( aaptExtraArg );
582         }
583 
584         if ( StringUtils.isNotBlank( configurations ) )
585         {
586             // Should be comma separated list of locales etc.
587             getLog().debug( "Adding resource configurations : " + configurations );
588             commands.add( "-c" );
589             commands.add( configurations );
590         }
591 
592         // We need to generate R.txt for all projects as it needs to be consumed when generating R class.
593         // It also needs to be consumed when packaging aar.
594         commands.add( "--output-text-symbols" );
595         commands.add( targetDirectory.getAbsolutePath() );
596 
597         // Allows us to supply multiple -S arguments.
598         commands.add( "--auto-add-overlay" );
599 
600         getLog().debug( getAndroidSdk().getAaptPath() + " " + commands.toString() );
601         try
602         {
603             targetDirectory.mkdirs();
604             executor.executeCommand( getAndroidSdk().getAaptPath(), commands, project.getBasedir(), false );
605         }
606         catch ( ExecutionException e )
607         {
608             throw new MojoExecutionException( "", e );
609         }
610 
611         // Generate R.java for apklibs
612         // Compatibility with Apklib which isn't present in AndroidBuilder
613         generateApkLibRs();
614 
615         // Generate error corrected R.java for APKLIB dependencies, but only if this is an APK build.
616         final Set<Artifact> apklibDependencies = getTransitiveDependencyArtifacts( APKLIB );
617         if ( !apklibDependencies.isEmpty() && APK.equals( project.getArtifact().getType() ) )
618         {
619             // Generate R.java for each APKLIB based on R.txt
620             getLog().debug( "Generating R file for APKLIB dependencies" );
621             generateRJavaFromRTxt( apklibDependencies, "apklib" );
622         }
623 
624         // Generate error corrected R.java for AAR dependencies.
625         final Set<Artifact> aarLibraries = getTransitiveDependencyArtifacts( AAR );
626         if ( !aarLibraries.isEmpty() )
627         {
628             // Generate R.java for each AAR based on R.txt
629             getLog().debug( "Generating R file for AAR dependencies" );
630             generateRJavaFromRTxt( aarLibraries, "aar" );
631         }
632 
633         getLog().info( "Adding R gen folder to compile classpath: " + genDirectory );
634         project.addCompileSourceRoot( genDirectory.getAbsolutePath() );
635     }

671      /**
672      * Executes aapt to generate the R class for the given apklib.
673      *
674      * @param apklibArtifact    apklib for which to generate the R class.
675      * @throws MojoExecutionException if it fails.
676      */
677     private void generateRForApkLibDependency( Artifact apklibArtifact ) throws MojoExecutionException
678     {
679         final File unpackDir = getUnpackedLibFolder( apklibArtifact );
680         getLog().info( "Generating R file for apklibrary: " + apklibArtifact.getGroupId()
681                 + ":" + apklibArtifact.getArtifactId() );
682         final File apklibManifest = new File( unpackDir, "AndroidManifest.xml" );
683         final File apklibResDir = new File( unpackDir, "res" );
684 
685         final CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
686         executor.setLogger( getLog() );
687 
688         List<String> commands = new ArrayList<String>();
689         commands.add( "package" );
690         commands.add( "--non-constant-id" );
691         commands.add( "-m" );
692         commands.add( "-J" );
693         commands.add( genDirectory.getAbsolutePath() );
694         commands.add( "--custom-package" );
695         commands.add( extractPackageNameFromAndroidManifest( apklibManifest ) );
696         commands.add( "-M" );
697         commands.add( apklibManifest.getAbsolutePath() );
698 
699         if ( apklibResDir.exists() )
700         {
701             // Add the APKLIB resources first so they overlay any of it's dependencies.
702             commands.add( "-S" );
703             commands.add( apklibResDir.getAbsolutePath() );
704         }
705 
706         final List<Artifact> apklibDependencies = getDependencyResolver().getDependenciesFor( apklibArtifact );
707         getLog().debug( "apklib dependencies = " + apklibDependencies );
708         for ( Artifact dependency : apklibDependencies )
709         {
710             // Add in the resources that are dependencies of the apklib.
711             final String extension = dependency.getType();
712             final File dependencyResDir = getUnpackedLibResourceFolder( dependency );
713             if ( ( extension.equals( APKLIB ) || extension.equals( AAR ) ) && dependencyResDir.exists() )
714             {
715                 commands.add( "-S" );
716                 commands.add( dependencyResDir.getAbsolutePath() );
717             }
718         }
719 
720         commands.add( "--auto-add-overlay" );
721 
722         // Create combinedAssets for this apklib dependency - can't have multiple -A args
723         final File apklibCombAssets = new File( getUnpackedLibFolder( apklibArtifact ), "combined-assets" );
724         for ( Artifact dependency : apklibDependencies )
725         {
726             // Accumulate assets for dependencies of the apklib (if they exist).
727             final String extension = dependency.getType();
728             final File dependencyAssetsDir = getUnpackedLibAssetsFolder( dependency );
729             if ( ( extension.equals( APKLIB ) || extension.equals( AAR ) ) )
730             {
731                 copyFolder( dependencyAssetsDir, apklibCombAssets );
732             }
733         }
734         // Overlay the apklib dependency assets (if they exist)
735         final File apkLibAssetsDir = getUnpackedLibAssetsFolder( apklibArtifact );
736         copyFolder( apkLibAssetsDir, apklibCombAssets );
737 
738         // If there are any combined assets for the apklib dependency then provide them to aapt.
739         if ( apklibCombAssets.exists() )
740         {
741             commands.add( "-A" );
742             commands.add( apklibCombAssets.getAbsolutePath() );
743         }
744 
745         commands.add( "-I" );
746         commands.add( getAndroidSdk().getAndroidJar().getAbsolutePath() );
747         if ( StringUtils.isNotBlank( configurations ) )
748         {
749             commands.add( "-c" );
750             commands.add( configurations );
751         }
752 
753         for ( String aaptExtraArg : aaptExtraArgs )
754         {
755             commands.add( aaptExtraArg );
756         }
757 
758         // We need to generate R.txt for all projects as it needs to be consumed when generating R class.
759         // It also needs to be consumed when packaging aar.
760         commands.add( "--output-text-symbols" );
761         commands.add( unpackDir.getAbsolutePath() );
762 
763         getLog().debug( getAndroidSdk().getAaptPath() + " " + commands.toString() );
764         try
765         {
766             executor.executeCommand( getAndroidSdk().getAaptPath(), commands, project.getBasedir(), false );
767         }
768         catch ( ExecutionException e )
769         {
770             throw new MojoExecutionException( "", e );
771         }
772     }

774     private void mergeManifests() throws MojoExecutionException
775     {
776         getLog().debug( "mergeManifests: " + mergeManifests );
777 
778         if ( !mergeManifests )
779         {
780             getLog().info( "Manifest merging disabled. Using project manifest only" );
781             return;
782         }
783 
784         getLog().info( "Getting manifests of dependent apklibs" );
785         List<File> libManifests = new ArrayList<File>();
786         for ( Artifact artifact : getTransitiveDependencyArtifacts( APKLIB, AAR ) )
787         {
788             final File libManifest = new File( getUnpackedLibFolder( artifact ), "AndroidManifest.xml" );
789             if ( !libManifest.exists() )
790             {
791                 throw new MojoExecutionException( artifact.getArtifactId() + " is missing AndroidManifest.xml" );
792             }
793 
794             libManifests.add( libManifest );
795         }
796 
797         if ( !libManifests.isEmpty() )
798         {
799             final File mergedManifest = new File( androidManifestFile.getParent(), "AndroidManifest-merged.xml" );
800             final StdLogger stdLogger = new StdLogger( StdLogger.Level.VERBOSE );
801             final ManifestMerger merger = new ManifestMerger( MergerLog.wrapSdkLog( stdLogger ), null );
802 
803             getLog().info( "Merging manifests of dependent apklibs" );
804 
805             final boolean mergeSuccess = merger.process( mergedManifest, androidManifestFile,
806                 libManifests.toArray( new File[libManifests.size()] ),  null, null );
807 
808             if ( mergeSuccess )
809             {
810                 // Replace the original manifest file with the merged one so that
811                 // the rest of the build will pick it up.
812                 androidManifestFile.delete();
813                 mergedManifest.renameTo( androidManifestFile );
814                 getLog().info( "Done Merging Manifests of APKLIBs" );
815             }
816             else
817             {
818                 getLog().error( "Manifests were not merged!" );
819                 throw new MojoExecutionException( "Manifests were not merged!" );
820             }
821         }
822         else
823         {
824             getLog().info( "No APKLIB manifests found. Using project manifest only." );
825         }
826     }

828     private void generateBuildConfig() throws MojoExecutionException
829     {
830         getLog().debug( "Generating BuildConfig file" );
831 
832         // Create the BuildConfig for our package.
833         String packageName = extractPackageNameFromAndroidManifest( androidManifestFile );
834         if ( StringUtils.isNotBlank( customPackage ) )
835         {
836             packageName = customPackage;
837         }
838         generateBuildConfigForPackage( packageName );
839 
840         // Generate the BuildConfig for any APKLIB and AAR dependencies.
841         // Need to generate for AAR, because some old AARs like ActionBarSherlock do not have BuildConfig (or R)
842         for ( Artifact artifact : getTransitiveDependencyArtifacts( APKLIB, AAR ) )
843         {
844             final File manifest = new File( getUnpackedLibFolder( artifact ), "AndroidManifest.xml" );
845             final String depPackageName = extractPackageNameFromAndroidManifest( manifest );
846             generateBuildConfigForPackage( depPackageName );
847         }
848     }

850     private void generateBuildConfigForPackage( String packageName ) throws MojoExecutionException
851     {
852         File outputFolder = new File( genDirectory, packageName.replace( ".", File.separator ) );
853         outputFolder.mkdirs();
854 
855         StringBuilder buildConfig = new StringBuilder();
856         buildConfig.append( "package " ).append( packageName ).append( ";\n\n" );
857         buildConfig.append( "public final class BuildConfig {\n" );
858         buildConfig.append( "  public static final boolean DEBUG = " ).append( !release ).append( ";\n" );
859         for ( BuildConfigConstant constant : buildConfigConstants )
860         {
861             String value = constant.getValue();
862             if ( "String".equals( constant.getType() ) )
863             {
864                 value = "\"" + value + "\"";
865             }
866 
867             buildConfig.append( "  public static final " )
868                        .append( constant.getType() )
869                        .append( " " )
870                        .append( constant.getName() )
871                        .append( " = " )
872                        .append( value )
873                        .append( ";\n" );
874         }
875         buildConfig.append( "}\n" );
876 
877 
878         File outputFile = new File( outputFolder, "BuildConfig.java" );
879         try
880         {
881             FileUtils.writeStringToFile( outputFile, buildConfig.toString() );
882         }
883         catch ( IOException e )
884         {
885             getLog().error( "Error generating BuildConfig ", e );
886             throw new MojoExecutionException( "Error generating BuildConfig", e );
887         }
888     }

890     /**
891      * Given a map of source directories to list of AIDL (relative) filenames within each,
892      * runs the AIDL compiler for each, such that all source directories are available to
893      * the AIDL compiler.
894      *
895      * @param files Map of source directory File instances to the relative paths to all AIDL files within
896      * @throws MojoExecutionException If the AIDL compiler fails
897      */
898     private void generateAidlFiles( Map<File /*sourceDirectory*/, String[] /*relativeAidlFileNames*/> files )
899             throws MojoExecutionException
900     {
901         List<String> protoCommands = new ArrayList<String>();
902         protoCommands.add( "-p" + getAndroidSdk().getPathForFrameworkAidl() );
903 
904         genDirectoryAidl.mkdirs();
905         getLog().warn( "Adding AIDL gen folder to compile classpath: " + genDirectoryAidl );
906         project.addCompileSourceRoot( genDirectoryAidl.getPath() );
907         Set<File> sourceDirs = files.keySet();
908         for ( File sourceDir : sourceDirs )
909         {
910             protoCommands.add( "-I" + sourceDir );
911         }
912         for ( File sourceDir : sourceDirs )
913         {
914             for ( String relativeAidlFileName : files.get( sourceDir ) )
915             {
916                 File targetDirectory = new File( genDirectoryAidl, new File( relativeAidlFileName ).getParent() );
917                 targetDirectory.mkdirs();
918 
919                 final String shortAidlFileName = new File( relativeAidlFileName ).getName();
920                 final String shortJavaFileName = shortAidlFileName.substring( 0, shortAidlFileName.lastIndexOf( "." ) )
921                                                  + ".java";
922                 final File aidlFileInSourceDirectory = new File( sourceDir, relativeAidlFileName );
923 
924                 List<String> commands = new ArrayList<String>( protoCommands );
925                 commands.add( aidlFileInSourceDirectory.getAbsolutePath() );
926                 commands.add( new File( targetDirectory, shortJavaFileName ).getAbsolutePath() );
927                 try
928                 {
929                     CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
930                     executor.setLogger( this.getLog() );
931 
932                     executor.executeCommand( getAndroidSdk().getAidlPath(), commands, project.getBasedir(),
933                             false );
934                 }
935                 catch ( ExecutionException e )
936                 {
937                     throw new MojoExecutionException( "", e );
938                 }
939             }
940         }
941     }
