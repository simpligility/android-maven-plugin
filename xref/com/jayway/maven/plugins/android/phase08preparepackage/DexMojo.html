142     @Override
143     public void execute() throws MojoExecutionException, MojoFailureException
144     {
145 
146         CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
147         executor.setLogger( this.getLog() );
148 
149         File outputFile = new File( project.getBuild().getDirectory() + File.separator + "classes.dex" );
150 
151         Set< File > inputFiles = getDexInputFiles();
152 
153         parseConfiguration();
154 
155         if ( generateApk )
156         {
157             runDex( executor, outputFile, inputFiles );
158         }
159 
160         if ( attachJar )
161         {
162             File jarFile = new File( project.getBuild().getDirectory() + File.separator
163                     + project.getBuild().getFinalName() + ".jar" );
164             projectHelper.attachArtifact( project, "jar", project.getArtifact().getClassifier(), jarFile );
165         }
166 
167         if ( attachSources )
168         {
169             // Also attach an .apksources, containing sources from this project.
170             final File apksources = createApkSourcesFile();
171             projectHelper.attachArtifact( project, "apksources", apksources );
172         }
173     }

175     /**
176      * Gets the input files for dex. This is a combination of directories and jar files.
177      * 
178      * @return
179      */
180     private Set< File > getDexInputFiles()
181     {
182 
183         Set< File > inputs = new HashSet< File >();
184 
185         // ugly, don't know a better way to get this in mvn
186         File proguardJar = new File( project.getBuild().getDirectory(), ProguardMojo.PROGUARD_OBFUSCATED_JAR );
187 
188         getLog().debug( "Checking for existence of: " + proguardJar.toString() );
189 
190         if ( proguardJar.exists() )
191         {
192             // progurad has been run, use this jar
193             getLog().debug( "Obfuscated jar exists, using that as input" );
194             inputs.add( proguardJar );
195         }
196         else
197         {
198             getLog().debug( "Using non-obfuscated input" );
199             // no proguard, use original config
200             inputs.add( new File( project.getBuild().getOutputDirectory() ) );
201             for ( Artifact artifact : getAllRelevantDependencyArtifacts() )
202             {
203                 if ( artifact.getType().equals( AAR ) )
204                 {
205                     final String apkLibResDir = getLibraryUnpackDirectory( artifact ) + "/classes.jar";
206                     if ( new File( apkLibResDir ).exists() )
207                     {
208                         inputs.add( new File( apkLibResDir ) );
209                     }
210                 }
211                 else if ( artifact.getType().equals( "so" ) || artifact.getType().equals( "a" ) )
212                 {
213                     // Ignore native dependencies - no need for dexer to see those
214                     continue;
215                 }
216                 else
217                 {
218                     inputs.add( artifact.getFile().getAbsoluteFile() );
219                 }
220             }
221         }
222 
223         return inputs;
224     }

226     private void parseConfiguration()
227     {
228         // config in pom found
229         if ( dex != null )
230         {
231             // the if statements make sure that properties/command line
232             // parameter overrides configuration
233             // and that the dafaults apply in all cases;
234             if ( dex.getJvmArguments() == null )
235             {
236                 parsedJvmArguments = dexJvmArguments;
237             }
238             else
239             {
240                 parsedJvmArguments = dex.getJvmArguments();
241             }
242             if ( dex.isCoreLibrary() == null )
243             {
244                 parsedCoreLibrary = dexCoreLibrary;
245             }
246             else
247             {
248                 parsedCoreLibrary = dex.isCoreLibrary();
249             }
250             if ( dex.isNoLocals() == null )
251             {
252                 parsedNoLocals = dexNoLocals;
253             }
254             else
255             {
256                 parsedNoLocals = dex.isNoLocals();
257             }
258             if ( dex.isOptimize() == null )
259             {
260                 parsedOptimize = dexOptimize;
261             }
262             else
263             {
264                 parsedOptimize = dex.isOptimize();
265             }
266             if ( dex.isPreDex() == null )
267             {
268                 parsedPreDex = dexPreDex;
269             }
270             else
271             {
272                 parsedPreDex = dex.isPreDex();
273             }
274             if ( dex.getPreDexLibLocation() == null )
275             {
276                 parsedPreDexLibLocation = dexPreDexLibLocation;
277             }
278             else
279             {
280                 parsedPreDexLibLocation = dex.getPreDexLibLocation();
281             }

292         {
293             parsedJvmArguments = dexJvmArguments;
294             parsedCoreLibrary = dexCoreLibrary;
295             parsedNoLocals = dexNoLocals;
296             parsedOptimize = dexOptimize;

303     private Set< File > preDex( CommandExecutor executor, Set< File > inputFiles ) throws MojoExecutionException
304     {
305         Set< File > filtered = new HashSet< File >();
306         getLog().info( "Pre dex-ing libraries for faster dex-ing of the final application." );
307 
308         for ( File inputFile : inputFiles )
309         {
310             if ( inputFile.getName().matches( ".*\\.jar$" ) )
311             {
312                 List< String > commands = dexDefaultCommands();
313 
314                 File predexJar = predexJarPath( inputFile );
315                 commands.add( "--output=" + predexJar.getAbsolutePath() );
316                 commands.add( inputFile.getAbsolutePath() );
317                 filtered.add( predexJar );
318 
319                 if ( !predexJar.isFile() || predexJar.lastModified() < inputFile.lastModified() )
320                 {
321                     getLog().info( "Pre-dex ing jar: " + inputFile.getAbsolutePath() );
322 
323                     final String javaExecutable = getJavaExecutable().getAbsolutePath();
324                     getLog().info( javaExecutable + " " + commands.toString() );
325                     try
326                     {
327                         executor.executeCommand( javaExecutable, commands, project.getBasedir(), false );
328                     }
329                     catch ( ExecutionException e )
330                     {
331                         throw new MojoExecutionException( "", e );
332                     }
333                 }
334 
335             }
336             else
337             {
338                 filtered.add( inputFile );
339             }
340         }
341 
342         return filtered;
343     }

345     private File predexJarPath( File inputFile )
346     {
347         final String slash = File.separator;
348         final File predexLibsDirectory = new File( parsedPreDexLibLocation.trim() );
349         predexLibsDirectory.mkdirs();
350         return new File( predexLibsDirectory.getAbsolutePath() + slash + inputFile.getName() );
351     }

353     private List< String > dexDefaultCommands() throws MojoExecutionException
354     {
355 
356         List< String > commands = new ArrayList< String >();
357         if ( parsedJvmArguments != null )
358         {
359             for ( String jvmArgument : parsedJvmArguments )
360             {
361                 // preserve backward compatibility allowing argument with or
362                 // without dash (e.g. Xmx512m as well as
363                 // -Xmx512m should work) (see
364                 // http://code.google.com/p/maven-android-plugin/issues/detail?id=153)
365                 if ( !jvmArgument.startsWith( "-" ) )
366                 {
367                     jvmArgument = "-" + jvmArgument;
368                 }
369                 getLog().debug( "Adding jvm argument " + jvmArgument );
370                 commands.add( jvmArgument );
371             }
372         }
373         commands.add( "-jar" );
374         commands.add( getAndroidSdk().getDxJarPath() );
375         commands.add( "--dex" );
376 
377         return commands;
378 
379     }

381     private void runDex( CommandExecutor executor, File outputFile, Set< File > inputFiles )
382             throws MojoExecutionException
383     {
384         List< String > commands = dexDefaultCommands();
385         Set< File > filteredFiles = inputFiles;
386 
387         if ( parsedPreDex )
388         {
389             filteredFiles = preDex( executor, inputFiles );
390         }
391         if ( !parsedOptimize )
392         {
393             commands.add( "--no-optimize" );
394         }
395         if ( parsedCoreLibrary )
396         {
397             commands.add( "--core-library" );
398         }
399         if ( parsedIncremental )
400         {
401             commands.add( "--incremental" );
402         }
403         commands.add( "--output=" + outputFile.getAbsolutePath() );
404         if ( parsedNoLocals )
405         {
406             commands.add( "--no-locals" );
407         }
408 
409         for ( File inputFile : filteredFiles )
410         {
411             getLog().debug( "Adding dex input: " + inputFile.getAbsolutePath() );
412             commands.add( inputFile.getAbsolutePath() );
413         }
414 
415         final String javaExecutable = getJavaExecutable().getAbsolutePath();
416         getLog().info( javaExecutable + " " + commands.toString() );
417         try
418         {
419             executor.executeCommand( javaExecutable, commands, project.getBasedir(), false );
420         }
421         catch ( ExecutionException e )
422         {
423             throw new MojoExecutionException( "", e );
424         }
425     }

430      * @return the full path to the current java executable.

432     private static File getJavaExecutable()
433     {
434         final String javaHome = System.getProperty( "java.home" );

439     /**
440      * @return
441      * @throws MojoExecutionException
442      */
443     protected File createApkSourcesFile() throws MojoExecutionException
444     {
445         final File apksources = new File( project.getBuild().getDirectory(), project.getBuild().getFinalName()
446                 + ".apksources" );
447         FileUtils.deleteQuietly( apksources );
448 
449         try
450         {
451             JarArchiver jarArchiver = new JarArchiver();
452             jarArchiver.setDestFile( apksources );
453 
454             addDirectory( jarArchiver, assetsDirectory, "assets" );
455             addDirectory( jarArchiver, resourceDirectory, "res" );
456             addDirectory( jarArchiver, sourceDirectory, "src/main/java" );
457             addJavaResources( jarArchiver, project.getBuild().getResources() );
458 
459             jarArchiver.createArchive();
460         }
461         catch ( ArchiverException e )
462         {
463             throw new MojoExecutionException( "ArchiverException while creating .apksource file.", e );
464         }
465         catch ( IOException e )
466         {
467             throw new MojoExecutionException( "IOException while creating .apksource file.", e );
468         }
469 
470         return apksources;
471     }

473     /**
474      * Makes sure the string ends with "/"
475      * 
476      * @param prefix
477      *            any string, or null.
478      * @return the prefix with a "/" at the end, never null.
479      */
480     protected String endWithSlash( String prefix )
481     {
482         prefix = StringUtils.defaultIfEmpty( prefix, "/" );
483         if ( !prefix.endsWith( "/" ) )
484         {
485             prefix = prefix + "/";
486         }
487         return prefix;
488     }

490     /**
491      * Adds a directory to a {@link JarArchiver} with a directory prefix.
492      * 
493      * @param jarArchiver
494      * @param directory
495      *            The directory to add.
496      * @param prefix
497      *            An optional prefix for where in the Jar file the directory's contents should go.
498      */
499     protected void addDirectory( JarArchiver jarArchiver, File directory, String prefix )
500     {
501         if ( directory != null && directory.exists() )
502         {
503             final DefaultFileSet fileSet = new DefaultFileSet();
504             fileSet.setPrefix( endWithSlash( prefix ) );
505             fileSet.setDirectory( directory );
506             jarArchiver.addFileSet( fileSet );
507         }
508     }

522     /**
523      * Adds a Java Resources directory (typically "src/main/resources") to a {@link JarArchiver}.
524      * 
525      * @param jarArchiver
526      * @param javaResource
527      *            The Java resource to add.
528      */
529     protected void addJavaResource( JarArchiver jarArchiver, Resource javaResource )
530     {
531         if ( javaResource != null )
532         {
533             final File javaResourceDirectory = new File( javaResource.getDirectory() );
534             if ( javaResourceDirectory.exists() )
535             {
536                 final DefaultFileSet javaResourceFileSet = new DefaultFileSet();
537                 javaResourceFileSet.setDirectory( javaResourceDirectory );
538                 javaResourceFileSet.setPrefix( endWithSlash( "src/main/resources" ) );
539                 jarArchiver.addFileSet( javaResourceFileSet );
540             }
541         }
