37                                 ArtifactHandler artifactHandler )

39          this.repoSystem = repoSystem;

45      /**
46       * Returns the list of transitive APKLIB or AAR dependencies of the supplied artifact.
47       *
48       * @param artifact  Artifact for whom to get the dependencies.
49       * @return List of APKLIB and AAR dependencies.
50       * @throws MojoExecutionException if it couldn't resolve any of the dependencies.
51       */
52      public List<Artifact> getDependenciesFor( Artifact artifact ) throws MojoExecutionException
53      {
54          final List<Artifact> results = new ArrayList<Artifact>();
55  
56          final org.eclipse.aether.artifact.Artifact artifactToResolve =
57                  new DefaultArtifact(
58                          artifact.getGroupId(),
59                          artifact.getArtifactId(),
60                          artifact.getType(),
61                          artifact.getVersion()
62                  );
63  
64          final List<Dependency> transitiveDeps = getDependenciesFor( artifactToResolve );
65          for ( Dependency dependency : transitiveDeps )
66          {
67              final Artifact apklibDep = new org.apache.maven.artifact.DefaultArtifact(
68                      dependency.getArtifact().getGroupId(),
69                      dependency.getArtifact().getArtifactId(),
70                      dependency.getArtifact().getVersion(),
71                      dependency.getScope(),
72                      dependency.getArtifact().getExtension(),
73                      dependency.getArtifact().getClassifier(),
74                      artifactHandler
75              );
76              results.add( apklibDep );
77          }
78  
79          return results;
80      }

82      /**
83       * Returns the list of transitive APKLIB or AAR dependencies of the supplied artifact.
84       *
85       * @param artifact  Artifact for whom to get the dependencies.
86       * @return List of APKLIB and AAR dependencies.
87       * @throws MojoExecutionException if it couldn't resolve any of the dependencies.
88       */
89      private List<Dependency> getDependenciesFor( org.eclipse.aether.artifact.Artifact artifact )
90              throws MojoExecutionException
91      {
92          final List<Dependency> results = new ArrayList<Dependency>();
93  
94          final ArtifactDescriptorRequest descriptorRequest = new ArtifactDescriptorRequest();
95          descriptorRequest.setArtifact( artifact );
96          descriptorRequest.setRepositories( remoteRepos );
97  
98          final ArtifactDescriptorResult descriptorResult;
99          try
100         {
101             descriptorResult = repoSystem.readArtifactDescriptor( repoSession, descriptorRequest );
102         }
103         catch ( ArtifactDescriptorException e )
104         {
105             throw new MojoExecutionException( "Could not resolve dependencies for " + artifact, e );
106         }
107 
108         for ( Dependency dependency : descriptorResult.getDependencies() )
109         {
110             final String extension = dependency.getArtifact().getExtension();
111             if ( extension.equals( APKLIB ) || extension.equals( AAR ) )
112             {
113                 results.add( dependency );
114                 results.addAll( getDependenciesFor( dependency.getArtifact() ) );
115             }
116         }
117 
118         return results;
119     }
