71      public static boolean hasStaticNativeLibraryArtifact( Set<Artifact> resolveNativeLibraryArtifacts, 
72                                                            File unpackDirectory,
73                                                            String ndkArchitecture )
74      {
75          for ( Artifact resolveNativeLibraryArtifact : resolveNativeLibraryArtifacts )
76          {
77              if ( "a".equals( resolveNativeLibraryArtifact.getType() ) )
78              {
79                  return true;
80              }
81              if ( APKLIB.equals( resolveNativeLibraryArtifact.getType() ) )
82              {
83                  File[] aFiles = listNativeFiles( resolveNativeLibraryArtifact, unpackDirectory, 
84                                                   ndkArchitecture, true );
85                  if ( aFiles != null && aFiles.length > 0 )
86                  {
87                      return true;
88                  }
89              }
90          }
91          return false;
92      }

117     public static File[] listNativeFiles( Artifact a, File unpackDirectory, 
118                                           final String ndkArchitecture, final boolean staticLibrary )
119     {
120         File libsFolder = new File(
121                 AbstractAndroidMojo
122                     .getLibraryUnpackDirectory( unpackDirectory, a ), 
123                 File.separator + ApklibMojo.NATIVE_LIBRARIES_FOLDER + File.separator + ndkArchitecture );
124         if ( libsFolder.exists() )

126             File[] libFiles = libsFolder.listFiles( new FilenameFilter()
127             {
128                 public boolean accept( final File dir, final String name )

130                     return name.startsWith( "lib" ) && name.endsWith( ( staticLibrary ? ".a" : ".so" ) );
131                 }
132             } );
133             return libFiles;
134         }
135         return null;

138     public Set<Artifact> getNativeDependenciesArtifacts( File unpackDirectory, boolean sharedLibraries )
139             throws MojoExecutionException
140     {
141         final Set<Artifact> filteredArtifacts = new LinkedHashSet<Artifact>();
142         final Set<Artifact> allArtifacts = new LinkedHashSet<Artifact>();
143         
144         // Add all dependent artifacts declared in the pom file
145         // Note: The result of project.getDependencyArtifacts() can be an UnmodifiableSet so we 
146         //       have created our own above and add to that.
147         allArtifacts.addAll( project.getDependencyArtifacts() );
148 
149         // Add all attached artifacts as well - this could come from the NDK mojo for example
150         allArtifacts.addAll( project.getAttachedArtifacts() );
151 
152         for ( Artifact artifact : allArtifacts )
153         {
154             // A null value in the scope indicates that the artifact has been attached
155             // as part of a previous build step (NDK mojo)
156             if ( isNativeLibrary( sharedLibraries, artifact.getType() ) && artifact.getScope() == null )
157             {
158                 // Including attached artifact
159                 log.debug( "Including attached artifact: " + artifact.getArtifactId() 
160                         + "(" + artifact.getGroupId() + "). Artifact scope is not set." );
161                 filteredArtifacts.add( artifact );
162             }
163             else
164             {
165                 if ( isNativeLibrary( sharedLibraries, artifact.getType() ) && (
166                         Artifact.SCOPE_COMPILE.equals( artifact.getScope() ) || Artifact.SCOPE_RUNTIME
167                                 .equals( artifact.getScope() ) ) )
168                 {
169                     log.debug( "Including attached artifact: " + artifact.getArtifactId() 
170                             + "(" + artifact.getGroupId() + "). Artifact scope is Compile or Runtime." );
171                     filteredArtifacts.add( artifact );
172                 }
173                 else
174                 {
175                     if ( APKLIB.equals( artifact.getType() ) )
176                     {
177                         // Check if the artifact contains a libs folder - if so, include it in the list
178                         File libsFolder = new File(
179                                 AbstractAndroidMojo.getLibraryUnpackDirectory( unpackDirectory, artifact ) + "/libs" );
180                         // make sure we ignore libs folders that only contain JARs
181                         if ( libsFolder.exists()
182                                 && libsFolder.list( new PatternFilenameFilter( ".+[^.jar]$" ) ).length > 0 )
183                         {
184                             log.debug( "Including attached artifact: " + artifact.getArtifactId() 
185                                     + "(" + artifact.getGroupId() + "). Artifact is APKLIB." );
186                             filteredArtifacts.add( artifact );
187                         }
188                     }
189                 }
190             }
191         }
192 
193         Set<Artifact> transientArtifacts = processTransientDependencies( project.getDependencies(), sharedLibraries );
194 
195         filteredArtifacts.addAll( transientArtifacts );
196 
197         return filteredArtifacts;
198     }

205     private Set<Artifact> processTransientDependencies( List<org.apache.maven.model.Dependency> dependencies,
206                                                         boolean sharedLibraries ) throws MojoExecutionException
207     {
208 
209         Set<Artifact> transientArtifacts = new LinkedHashSet<Artifact>();
210         for ( org.apache.maven.model.Dependency dependency : dependencies )
211         {
212             if ( ! "provided".equals( dependency.getScope() ) && ! dependency.isOptional() )
213             {
214                 transientArtifacts.addAll(
215                         processTransientDependencies( toDependency( dependency, repoSession.getArtifactTypeRegistry() ),
216                                 sharedLibraries ) );
217             }
218         }
219 
220         return transientArtifacts;

224     private Set<Artifact> processTransientDependencies( Dependency dependency, boolean sharedLibraries )
225             throws MojoExecutionException
226     {
227         try
228         {
229             final Set<Artifact> artifacts = new LinkedHashSet<Artifact>();
230 
231             final CollectRequest collectRequest = new CollectRequest();
232 
233             collectRequest.setRoot( dependency );
234             collectRequest.setRepositories( projectRepos );
235             final DependencyNode node = repoSystem.collectDependencies( repoSession, collectRequest ).getRoot();
236 
237             Collection<String> exclusionPatterns = new ArrayList<String>();
238             if ( dependency.getExclusions() != null && ! dependency.getExclusions().isEmpty() )
239             {
240                 for ( Exclusion exclusion : dependency.getExclusions() )
241                 {
242                     exclusionPatterns.add( exclusion.getGroupId() + ":" + exclusion.getArtifactId() );
243                 }
244             }
245 
246             final DependencyRequest dependencyRequest = new DependencyRequest( node,
247                     new AndDependencyFilter( new ExclusionsDependencyFilter( exclusionPatterns ),
248                             new AndDependencyFilter( new ScopeDependencyFilter( Arrays.asList( "compile", "runtime" ),
249                                     Arrays.asList( "test" ) ),
250                                     // Also exclude any optional dependencies
251                                     new DependencyFilter()
252                                     {
253                                         @Override
254                                         public boolean accept( DependencyNode dependencyNode,
255                                                                List<DependencyNode> dependencyNodes )
256                                         {
257                                             return ! dependencyNode.getDependency().isOptional();
258                                         }
259                                     } ) ) );
260 
261 
262             repoSystem.resolveDependencies( repoSession, dependencyRequest );
263 
264             PreorderNodeListGenerator nlg = new PreorderNodeListGenerator();
265             node.accept( nlg );
266 
267             final List<Dependency> dependencies = nlg.getDependencies( false );
268 
269             for ( Dependency dep : dependencies )
270             {
271                 final org.eclipse.aether.artifact.Artifact depAetherArtifact = dep.getArtifact();
272                 if ( isNativeLibrary( sharedLibraries, depAetherArtifact.getExtension() ) )
273                 {
274                     final Artifact mavenArtifact = artifactFactory
275                             .createDependencyArtifact( depAetherArtifact.getGroupId(),
276                                     depAetherArtifact.getArtifactId(),
277                                     VersionRange.createFromVersion( depAetherArtifact.getVersion() ),
278                                     depAetherArtifact.getExtension(), depAetherArtifact.getClassifier(),
279                                     dep.getScope() );
280                     mavenArtifact.setFile( depAetherArtifact.getFile() );
281                     artifacts.add( mavenArtifact );
282                 }
283             }
284 
285             return artifacts;
286         }
287         catch ( Exception e )
288         {
289             throw new MojoExecutionException( "Error while processing transient dependencies", e );
290         }

293     public static void validateNDKVersion( File ndkHomeDir ) throws MojoExecutionException
294     {
295         final File ndkVersionFile = new File( ndkHomeDir, "RELEASE.TXT" );
296

302         }
303

305         {
306             String versionStr = FileUtils.readFileToString( ndkVersionFile );
307             validateNDKVersion( NDK_REQUIRED_VERSION, versionStr );

312                     + ndkVersionFile.getAbsolutePath() + "'. Please verify your setup! "
313                     + AndroidNdk.PROPER_NDK_HOME_DIRECTORY_MESSAGE );

317     public static void validateNDKVersion( int desiredVersion, String versionStr ) throws MojoExecutionException
318     {
319 
320         int version = 0;
321 
322         if ( versionStr != null )
323         {
324             versionStr = versionStr.trim();
325             Pattern pattern = Pattern.compile( "[r]([0-9]{1,3})([a-z]{0,1}).*" );
326             Matcher m = pattern.matcher( versionStr );
327             if ( m.matches() )
328             {
329                 final String group = m.group( 1 );
330                 version = Integer.parseInt( group );
331             }
332         }
333 
334         if ( version < desiredVersion )
335         {
336             throw new MojoExecutionException( "You are running an old NDK (version " + versionStr + "), please update "
337                     + "to at least r'" + desiredVersion + "' or later" );
338         }
339     }

341     public static String[] getAppAbi( File applicationMakefile )
342     {
343         Scanner scanner = null;
344         try
345         {
346             if ( applicationMakefile != null && applicationMakefile.exists() )
347             {
348                 scanner = new Scanner( applicationMakefile );
349                 while ( scanner.hasNextLine( ) )
350                 {
351                     String line = scanner.nextLine().trim();
352                     if ( line.startsWith( "APP_ABI" ) )
353                     {
354                         return line.substring( line.indexOf( ":=" ) + 2 ).trim().split( " " );
355                     }
356                 }
357             }
358         }
359         catch ( FileNotFoundException e )
360         {
361             // do nothing
362         }
363         finally
364         {
365             if ( scanner != null )
366             {
367                 scanner.close();
368             }
369         }
370         return null;
371     }

374     /** Extracts, if embedded correctly, the artifacts architecture from its classifier.  The format of the
375      * classifier, if including the architecture is &lt;architecture&gt;-&lt;classifier&gt;.  If no
376      * architecture is embedded in the classifier, 'armeabi' will be returned.
377      *
378      *
379      * @param artifact The artifact to retrieve the classifier from.
380      * @param defaultArchitecture The architecture to return if can't be resolved from the classifier
381      * @return The retrieved architecture, or <code>defaulArchitecture</code> if not resolveable
382      */
383     public static String extractArchitectureFromArtifact( Artifact artifact, final String defaultArchitecture )
384     {
385         String classifier = artifact.getClassifier();
386         if ( classifier != null )
387         {
388             //
389             // We loop backwards to catch the case where the classifier is
390             // potentially armeabi-v7a - this collides with armeabi if looping
391             // through this loop in the other direction
392             //
393 
394             for ( int i = AndroidNdk.NDK_ARCHITECTURES.length - 1; i >= 0; i-- )
395             {
396                 String ndkArchitecture = AndroidNdk.NDK_ARCHITECTURES[i];
397                 if ( classifier.startsWith( ndkArchitecture ) )
398                 {
399                     return ndkArchitecture;
400                 }
401             }
402 
403         }
404         // Default case is to return the default architecture
405         return defaultArchitecture;
406     }

408     /** Attempts to extract, from various sources, the applicable list of NDK architectures to support
409      * as part of the build.
410      * <br/>
411      * <br/>
412      * It retrieves the list from the following places:
413      * <ul>
414      *     <li>ndkArchitecture parameter</li>
415      *     <li>projects Application.mk - currently only a single architecture is supported by this method</li>
416      * </ul>
417      *
418      *
419      * @param ndkArchitectures Space separated list of architectures.  This may be from configuration or otherwise
420      * @param applicationMakefile The makefile (Application.mk) to retrieve the list from.
421      * @param basedir Directory the build is running from (to resolve files)
422      *
423      * @return List of architectures to be supported by build.
424      *
425      * @throws MojoExecutionException
426      */
427     public static String[] getNdkArchitectures( final String ndkArchitectures, final String applicationMakefile,
428                                                 final File basedir )
429         throws MojoExecutionException
430     {
431         // if there is a specified ndk architecture, return it
432         if ( ndkArchitectures != null )
433         {
434             return ndkArchitectures.split( " " );
435         }
436 
437         // if there is no application makefile specified, let's use the default one
438         String applicationMakefileToUse = applicationMakefile;
439         if ( applicationMakefileToUse == null )
440         {
441             applicationMakefileToUse = "jni/Application.mk";
442         }
443 
444         // now let's see if the application file exists
445         File appMK = new File( basedir, applicationMakefileToUse );
446         if ( appMK.exists() )
447         {
448             String[] foundNdkArchitectures = getAppAbi( appMK );
449             if ( foundNdkArchitectures != null )
450             {
451                 return foundNdkArchitectures;
452             }
453         }
454 
455         // return a default ndk architecture
456         return new String[] { "armeabi" };
457     }

459     /** Helper method for determining whether the specified architecture is a match for the
460      * artifact using its classifier.  When used for architecture matching, the classifier must be
461      * formed by &lt;architecture&gt;-&lt;classifier&gt;.
462      * If the artifact is legacy and defines no valid architecture, the artifact architecture will
463      * default to <strong>armeabi</strong>.
464      *
465      * @param ndkArchitecture Architecture to check for match

470                                                            String defaultArchitecture )
471     {
472         return "so".equals( artifact.getType() )
473                 && ndkArchitecture.equals( extractArchitectureFromArtifact( artifact, defaultArchitecture ) );
474     }
