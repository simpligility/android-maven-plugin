18   * Helper class to convert between Maven and Aether beans and provide helper methods for resolving artifacts.

20   * @author Johan Lindquist

22  public class AetherHelper

25      /**
26       * Converts the specified Maven artifact to a Aether artifact.  This method takes into consideration situations
27       * where some of the artifact properties (classifier, etc) are null.
28       *
29       * @param artifact The Maven artifact to convert
30       * @return The resulting Aether artifact
31       */
32      public static org.eclipse.aether.artifact.Artifact createAetherArtifact( Artifact artifact )
33      {
34          DefaultArtifact defaultArtifact;
35          if ( artifact.getClassifier() != null )
36          {
37              defaultArtifact = new DefaultArtifact( artifact.getGroupId(), artifact.getArtifactId(),
38                      artifact.getClassifier(), artifact.getType(), artifact.getVersion() );
39          }
40          else
41          {
42              defaultArtifact = new DefaultArtifact( artifact.getGroupId(), artifact.getArtifactId(), artifact.getType(),
43                      artifact.getVersion() );
44          }
45          defaultArtifact.setFile( artifact.getFile() );
46          return defaultArtifact;
47      }

49      public static Set<Artifact> resolveArtifacts( Set<Artifact> artifacts, RepositorySystem repositorySystem,
50                                                    RepositorySystemSession repositorySystemSession,
51                                                    List<RemoteRepository> repositories ) throws MojoExecutionException
52      {
53          try
54          {
55  
56              final Set<Artifact> resolvedArtifacts = new LinkedHashSet<Artifact>();
57  
58              for ( Artifact artifact : artifacts )
59              {
60                  final Artifact resolvedArtifact = AetherHelper
61                          .resolveArtifact( artifact, repositorySystem, repositorySystemSession, repositories );
62                  resolvedArtifacts.add( resolvedArtifact );
63              }
64              return resolvedArtifacts;
65          }
66          catch ( Exception e )
67          {
68              throw new MojoExecutionException( "Error resolving dependencies: " + e.getMessage(), e );
69          }
70      }

72      public static Artifact resolveArtifact( Artifact artifact, RepositorySystem repositorySystem,
73                                              RepositorySystemSession repositorySystemSession,
74                                              List<RemoteRepository> repositories ) throws MojoExecutionException
75      {
76          try
77          {
78              final ArtifactRequest artifactRequest = new ArtifactRequest();
79              org.eclipse.aether.artifact.Artifact aetherArtifact = AetherHelper.createAetherArtifact( artifact );
80              artifactRequest.setArtifact( aetherArtifact );
81              artifactRequest.setRepositories( repositories );
82              final ArtifactResult artifactResult = repositorySystem
83                      .resolveArtifact( repositorySystemSession, artifactRequest );
84  
85              final org.apache.maven.artifact.DefaultArtifact defaultArtifact
86                      = new org.apache.maven.artifact.DefaultArtifact( artifact.getGroupId(), artifact.getArtifactId(),
87                      artifact.getVersion(), artifact.getScope(), artifact.getType(), artifact.getClassifier(),
88                      artifact.getArtifactHandler() );
89              defaultArtifact.setFile( artifactResult.getArtifact().getFile() );
90              return defaultArtifact;
91          }
92          catch ( ArtifactResolutionException e )
93          {
94              throw new MojoExecutionException( "Error while resolving artifact: " + e.getMessage(), e );
95          }
96      }
