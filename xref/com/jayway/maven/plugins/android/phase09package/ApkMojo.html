183     /**
184      * Pattern for additional META-INF resources to be packaged into the apk.
185      * <p>
186      * The APK builder filters these resources and doesn't include them into the apk.
187      * This leads to bad behaviour of dependent libraries relying on these resources,
188      * for instance service discovery doesn't work.<br/>
189      * By specifying this pattern, the android plugin adds these resources to the final apk.
190      * </p>
191      * <p>The pattern is relative to META-INF, i.e. one must use
192      * <pre>
193      * <code>
194      * &lt;apkMetaIncludes&gt;
195      *     &lt;metaInclude>services/**&lt;/metaInclude&gt;
196      * &lt;/apkMetaIncludes&gt;
197      * </code>
198      * </pre>
199      * ... instead of
200      * <pre>
201      * <code>
202      * &lt;apkMetaIncludes&gt;
203      *     &lt;metaInclude>META-INF/services/**&lt;/metaInclude&gt;
204      * &lt;/apkMetaIncludes&gt;
205      * </code>
206      * </pre>
207      * <p>
208      * See also <a href="http://code.google.com/p/maven-android-plugin/issues/detail?id=97">Issue 97</a>
209      * </p>
210      *
211      * @parameter expression="${android.apk.metaIncludes}"
212      * @deprecated in favour of apk.metaInf
213      */

215     private String[] apkMetaIncludes;

277     /**
278      *
279      * @throws MojoExecutionException
280      * @throws MojoFailureException
281      */
282     public void execute() throws MojoExecutionException, MojoFailureException
283     {
284 
285         // Make an early exit if we're not supposed to generate the APK
286         if ( ! generateApk )
287         {
288             return;
289         }
290 
291         ConfigHandler cfh = new ConfigHandler( this, this.session, this.execution );
292 
293         cfh.parseConfiguration();
294 
295         generateIntermediateApk();
296 
297         // Compile resource exclusion patterns, if any
298         if ( excludeJarResources != null && excludeJarResources.length > 0 ) 
299         {
300           getLog().debug( "Compiling " + excludeJarResources.length + " patterns" );
301           
302           excludeJarResourcesPatterns = new Pattern[excludeJarResources.length];
303           
304           for ( int index = 0; index < excludeJarResources.length; ++index ) 
305           {
306             excludeJarResourcesPatterns[index] = Pattern.compile( excludeJarResources[index] );
307           }
308         }
309         
310         // Initialize apk build configuration
311         File outputFile = new File( project.getBuild().getDirectory(), project.getBuild().getFinalName() + "." + APK );
312         final boolean signWithDebugKeyStore = getAndroidSigner().isSignWithDebugKeyStore();
313 
314         if ( getAndroidSigner().shouldCreateBothSignedAndUnsignedApk() )
315         {
316             getLog().info( "Creating debug key signed apk file " + outputFile );
317             createApkFile( outputFile, true );
318             final File unsignedOutputFile = new File( project.getBuild().getDirectory(),
319                     project.getBuild().getFinalName() + "-unsigned." + APK );
320             getLog().info( "Creating additional unsigned apk file " + unsignedOutputFile );
321             createApkFile( unsignedOutputFile, false );
322             projectHelper.attachArtifact( project, unsignedOutputFile,
323                     classifier == null ? "unsigned" : classifier + "_unsigned" );
324         }
325         else
326         {
327             createApkFile( outputFile, signWithDebugKeyStore );
328         }
329 
330         if ( classifier == null )
331         {
332             // Set the generated .apk file as the main artifact (because the pom states <packaging>apk</packaging>)
333             project.getArtifact().setFile( outputFile );
334         }
335         else
336         {
337             // If there is a classifier specified, attach the artifact using that
338             projectHelper.attachArtifact( project, outputFile, classifier );
339         }
340     }

342     void createApkFile( File outputFile, boolean signWithDebugKeyStore ) throws MojoExecutionException
343     {
344         File dexFile = new File( project.getBuild().getDirectory(), "classes.dex" );
345         File zipArchive = new File( project.getBuild().getDirectory(), project.getBuild().getFinalName() + ".ap_" );
346         ArrayList<File> sourceFolders = new ArrayList<File>();
347         if ( sourceDirectories != null )
348         {
349             for ( File f : sourceDirectories )
350             {
351                 sourceFolders.add( f );
352             }
353         }
354         ArrayList<File> jarFiles = new ArrayList<File>();
355         ArrayList<File> nativeFolders = new ArrayList<File>();
356 
357         // Process the native libraries, looking both in the current build directory as well as
358         // at the dependencies declared in the pom.  Currently, all .so files are automatically included
359         processNativeLibraries( nativeFolders );
360         
361         doAPKWithAPKBuilder( outputFile, dexFile, zipArchive, sourceFolders, jarFiles, nativeFolders,
362                     signWithDebugKeyStore );
363 
364         if ( this.apkMetaInf != null )
365         {
366             try
367             {
368                 addMetaInf( outputFile, jarFiles );
369             }
370             catch ( IOException e )
371             {
372                 throw new MojoExecutionException( "Could not add META-INF resources.", e );
373             }
374         }
375     }

377     private void addMetaInf( File outputFile, ArrayList<File> jarFiles ) throws IOException
378     {
379         File tmp = File.createTempFile( outputFile.getName(), ".add", outputFile.getParentFile() );
380 
381         FileOutputStream fos = new FileOutputStream( tmp );
382         ZipOutputStream zos = new ZipOutputStream( fos );
383         Set<String> entries = new HashSet<String>();
384 
385         updateWithMetaInf( zos, outputFile, entries, false );
386 
387         for ( File f : jarFiles )
388         {
389             updateWithMetaInf( zos, f, entries, true );
390         }
391 
392         zos.close();
393 
394         outputFile.delete();
395 
396         if ( ! tmp.renameTo( outputFile ) )
397         {
398             throw new IOException( String.format( "Cannot rename %s to %s", tmp, outputFile.getName() ) );
399         }
400     }

402     private void updateWithMetaInf( ZipOutputStream zos, File jarFile, Set<String> entries, boolean metaInfOnly )
403             throws IOException
404     {
405         ZipFile zin = new ZipFile( jarFile );
406 
407         for ( Enumeration<? extends ZipEntry> en = zin.entries(); en.hasMoreElements(); )
408         {
409             ZipEntry ze = en.nextElement();
410 
411             if ( ze.isDirectory() )
412             {
413                 continue;
414             }
415 
416             String zn = ze.getName();
417 
418             if ( metaInfOnly )
419             {
420                 if ( ! zn.startsWith( "META-INF/" ) )
421                 {
422                     continue;
423                 }
424 
425                 if ( this.extractDuplicates && ! entries.add( zn ) )
426                 {
427                     continue;
428                 }
429 
430                 if ( ! this.apkMetaInf.isIncluded( zn ) )
431                 {
432                     continue;
433                 }
434             }
435 
436             zos.putNextEntry( new ZipEntry( zn ) );
437 
438             InputStream is = zin.getInputStream( ze );
439 
440             copyStreamWithoutClosing( is, zos );
441 
442             is.close();
443             zos.closeEntry();
444         }
445 
446         zin.close();
447     }

451     private void computeDuplicateFiles( File jar ) throws IOException
452     {
453         ZipFile file = new ZipFile( jar );
454         Enumeration<? extends ZipEntry> list = file.entries();
455         while ( list.hasMoreElements() )
456         {
457             ZipEntry ze = list.nextElement();
458             if ( ! ( ze.getName().contains( "META-INF/" ) || ze.isDirectory() ) )
459             { // Exclude META-INF and Directories
460                 List<File> l = jars.get( ze.getName() );
461                 if ( l == null )
462                 {
463                     l = new ArrayList<File>();
464                     jars.put( ze.getName(), l );
465                 }
466                 l.add( jar );
467             }
468         }
469     }

471     /**
472      * Creates the APK file using the internal APKBuilder.
473      *
474      * @param outputFile            the output file
475      * @param dexFile               the dex file
476      * @param zipArchive            the classes folder
477      * @param sourceFolders         the resources
478      * @param jarFiles              the embedded java files
479      * @param nativeFolders         the native folders
480      * @param signWithDebugKeyStore enables the signature of the APK using the debug key
481      * @throws MojoExecutionException if the APK cannot be created.
482      */
483     private void doAPKWithAPKBuilder( File outputFile, File dexFile, File zipArchive, ArrayList<File> sourceFolders,
484                                       ArrayList<File> jarFiles, ArrayList<File> nativeFolders,
485                                       boolean signWithDebugKeyStore ) throws MojoExecutionException
486     {
487         getLog().debug( "Building APK with internal APKBuilder" );
488         sourceFolders.add( new File( project.getBuild().getOutputDirectory() ) );
489 
490         for ( Artifact artifact : getRelevantCompileArtifacts() )
491         {
492             if ( extractDuplicates )
493             {
494                 try
495                 {
496                     computeDuplicateFiles( artifact.getFile() );
497                 }
498                 catch ( Exception e )
499                 {
500                     getLog().warn( "Cannot compute duplicates files from " + artifact.getFile().getAbsolutePath(), e );
501                 }
502             }
503             jarFiles.add( artifact.getFile() );
504         }
505 
506         // Check duplicates.
507         if ( extractDuplicates )
508         {
509             List<String> duplicates = new ArrayList<String>();
510             List<File> jarToModify = new ArrayList<File>();
511             for ( String s : jars.keySet() )
512             {
513                 List<File> l = jars.get( s );
514                 if ( l.size() > 1 )
515                 {
516                     getLog().warn( "Duplicate file " + s + " : " + l );
517                     duplicates.add( s );
518                     for ( int i = 1; i < l.size(); i++ )
519                     {
520                         if ( ! jarToModify.contains( l.get( i ) ) )
521                         {
522                             jarToModify.add( l.get( i ) );
523                         }
524                     }
525                 }
526             }
527 
528             // Rebuild jars.
529             for ( File file : jarToModify )
530             {
531                 File newJar;
532                 newJar = removeDuplicatesFromJar( file, duplicates );
533                 int index = jarFiles.indexOf( file );
534                 if ( newJar != null )
535                 {
536                     jarFiles.set( index, newJar );
537                 }
538 
539             }
540         }
541         
542         String debugKeyStore;
543         ApkBuilder apkBuilder; 
544         try 
545         {
546             debugKeyStore = ApkBuilder.getDebugKeystore();
547             apkBuilder = 
548                     new ApkBuilder( outputFile, zipArchive, dexFile, 
549                             ( signWithDebugKeyStore ) ? debugKeyStore : null, null );
550             if ( apkDebug )
551             {
552                 apkBuilder.setDebugMode( apkDebug );
553             }
554 
555             for ( File sourceFolder : sourceFolders )
556             {
557                 apkBuilder.addSourceFolder( sourceFolder );
558             }
559      
560             for ( File jarFile : jarFiles )
561             {
562                 boolean excluded = false;
563               
564                 if ( excludeJarResourcesPatterns != null )
565                 {
566                     final String name = jarFile.getName();
567                     getLog().debug( "Checking " + name + " against patterns" );
568                     for ( Pattern pattern : excludeJarResourcesPatterns )
569                     {
570                         final Matcher matcher = pattern.matcher( name );
571                         if ( matcher.matches() ) 
572                         {
573                             getLog().debug( "Jar " + name + " excluded by pattern " + pattern );
574                             excluded = true;
575                             break;
576                         } 
577                         else 
578                         {
579                             getLog().debug( "Jar " + name + " not excluded by pattern " + pattern );
580                         }
581                     }
582                 }
583     
584                 if ( excluded )
585                 {
586                     continue;
587                 }
588                 
589                 if ( jarFile.isDirectory() )
590                 {
591                     String[] filenames = jarFile.list( new FilenameFilter()
592                     {
593                         public boolean accept( File dir, String name )
594                         {
595                             return PATTERN_JAR_EXT.matcher( name ).matches();
596                         }
597                     } );
598     
599                     for ( String filename : filenames )
600                     {
601                         apkBuilder.addResourcesFromJar( new File( jarFile, filename ) );
602                     }
603                 }
604                 else
605                 {
606                     apkBuilder.addResourcesFromJar( jarFile );
607                 }
608             }
609 
610             for ( File nativeFolder : nativeFolders )
611             {
612                 apkBuilder.addNativeLibraries( nativeFolder );
613             }
614             apkBuilder.sealApk();
615         } 
616         catch ( ApkCreationException e )
617         {
618             throw new MojoExecutionException( e.getMessage() );
619         } 
620         catch ( DuplicateFileException e )
621         {
622             throw new MojoExecutionException( e.getMessage() );
623         } 
624         catch ( SealedApkException e )
625         {
626             throw new MojoExecutionException( e.getMessage() );
627         }
628     }

630     private File removeDuplicatesFromJar( File in, List<String> duplicates )
631     {
632         String target = project.getBuild().getOutputDirectory();
633         File tmp = new File( target, "unpacked-embedded-jars" );
634         tmp.mkdirs();
635         File out = new File( tmp, in.getName() );
636 
637         if ( out.exists() )
638         {
639             return out;
640         }
641         else
642         {
643             try
644             {
645                 out.createNewFile();
646             }
647             catch ( IOException e )
648             {
649                 e.printStackTrace();
650             }
651         }
652 
653         // Create a new Jar file
654         FileOutputStream fos = null;
655         ZipOutputStream jos = null;
656         try
657         {
658             fos = new FileOutputStream( out );
659             jos = new ZipOutputStream( fos );
660         }
661         catch ( FileNotFoundException e1 )
662         {
663             getLog().error( "Cannot remove duplicates : the output file " + out.getAbsolutePath() + " does not found" );
664             return null;
665         }
666 
667         ZipFile inZip = null;
668         try
669         {
670             inZip = new ZipFile( in );
671             Enumeration<? extends ZipEntry> entries = inZip.entries();
672             while ( entries.hasMoreElements() )
673             {
674                 ZipEntry entry = entries.nextElement();
675                 // If the entry is not a duplicate, copy.
676                 if ( ! duplicates.contains( entry.getName() ) )
677                 {
678                     // copy the entry header to jos
679                     jos.putNextEntry( entry );
680                     InputStream currIn = inZip.getInputStream( entry );
681                     copyStreamWithoutClosing( currIn, jos );
682                     currIn.close();
683                     jos.closeEntry();
684                 }
685             }
686         }
687         catch ( IOException e )
688         {
689             getLog().error( "Cannot removing duplicates : " + e.getMessage() );
690             return null;
691         }
692 
693         try
694         {
695             if ( inZip != null )
696             {
697                 inZip.close();
698             }
699             jos.close();
700             fos.close();
701             jos = null;
702             fos = null;
703         }
704         catch ( IOException e )
705         {
706             // ignore it.
707         }
708         getLog().info( in.getName() + " rewritten without duplicates : " + out.getAbsolutePath() );
709         return out;
710     }

712     /**
713      * Copies an input stream into an output stream but does not close the streams.
714      *
715      * @param in  the input stream
716      * @param out the output stream
717      * @throws IOException if the stream cannot be copied
718      */
719     private static void copyStreamWithoutClosing( InputStream in, OutputStream out ) throws IOException
720     {
721         final int bufferSize = 4096;
722         byte[] b = new byte[ bufferSize ];
723         int n;
724         while ( ( n = in.read( b ) ) != - 1 )
725         {

746     private void processNativeLibraries( final List<File> natives, String ndkArchitecture )
747             throws MojoExecutionException
748     {
749         // Examine the native libraries directory for content. This will only be true if:
750         // a) the directory exists
751         // b) it contains at least 1 file
752         final boolean hasValidNativeLibrariesDirectory = hasValidNativeLibrariesDirectory();
753         final boolean hasValidBuildNativeLibrariesDirectory = hasValidBuildNativeLibrariesDirectory();
754         final Set<Artifact> artifacts = getNativeLibraryArtifacts();
755 
756         if ( artifacts.isEmpty() && hasValidNativeLibrariesDirectory && ! hasValidBuildNativeLibrariesDirectory )
757         {
758 
759             getLog().debug(
760                     "No native library dependencies detected, will point directly to " + nativeLibrariesDirectory );
761 
762             // Point directly to the directory in this case - no need to copy files around
763             addNativeDirectory( natives, nativeLibrariesDirectory );
764 
765             // FIXME: This would pollute a libs folder which is under source control
766             // FIXME: Would be better to not support this case?
767             optionallyCopyGdbServer( nativeLibrariesDirectory, ndkArchitecture );
768 
769         }
770         else
771         {
772             if ( ! artifacts.isEmpty() || hasValidNativeLibrariesDirectory )
773             {
774                 // In this case, we may have both .so files in it's normal location
775                 // as well as .so dependencies
776 
777                 final File destinationDirectory = makeNativeLibrariesOutputDirectory();
778 
779                 // Point directly to the directory
780                 addNativeDirectory( natives, destinationDirectory );
781 
782                 // If we have a valid native libs, copy those files - these already come in the structure required
783                 if ( hasValidNativeLibrariesDirectory )
784                 {
785                     copyLocalNativeLibraries( nativeLibrariesDirectory, destinationDirectory );
786                 }
787 
788                 if ( ! artifacts.isEmpty() )
789                 {
790                     for ( Artifact resolvedArtifact : artifacts )
791                     {
792                         if ( NativeHelper.artifactHasHardwareArchitecture( resolvedArtifact,
793                                 ndkArchitecture, nativeLibrariesDependenciesHardwareArchitectureDefault ) )
794                         {
795                             copyNativeLibraryArtifactFileToDirectory( resolvedArtifact, destinationDirectory,
796                                     ndkArchitecture );
797                         }
798                         else if ( APKLIB.equals( resolvedArtifact.getType() )
799                                 || AAR.equals( resolvedArtifact.getType() ) )
800                         {
801                                 addNativeDirectory( natives, new File( getLibraryUnpackDirectory( resolvedArtifact )
802                                                                            + "/libs" ) );
803                         }
804                     }
805                 }
806 
807                 // Finally, think about copying the gdbserver binary into the APK output as well
808                 optionallyCopyGdbServer( destinationDirectory, ndkArchitecture );
809             }
810         }
811     }

927     private void copyLocalNativeLibraries( final File localNativeLibrariesDirectory, final File destinationDirectory )
928             throws MojoExecutionException
929     {

935

937             IOFileFilter orFilter = FileFilterUtils.or( libSuffixFilter, gdbserverNameFilter );

953     /**
954      * Generates an intermediate apk file (actually .ap_) containing the resources and assets.
955      *
956      * @throws MojoExecutionException
957      */
958     private void generateIntermediateApk() throws MojoExecutionException
959     {
960         CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
961         executor.setLogger( this.getLog() );
962         File[] overlayDirectories = getResourceOverlayDirectories();
963 
964         File androidJar = getAndroidSdk().getAndroidJar();
965         File outputFile = new File( project.getBuild().getDirectory(), project.getBuild().getFinalName() + ".ap_" );
966 
967         List<String> commands = new ArrayList<String>();
968         commands.add( "package" );
969         commands.add( "-f" );
970         commands.add( "-M" );
971         commands.add( androidManifestFile.getAbsolutePath() );
972         for ( File resOverlayDir : overlayDirectories )
973         {
974             if ( resOverlayDir != null && resOverlayDir.exists() )
975             {
976                 commands.add( "-S" );
977                 commands.add( resOverlayDir.getAbsolutePath() );
978             }
979         }
980         if ( resourceDirectory.exists() )
981         {
982             commands.add( "-S" );
983             commands.add( resourceDirectory.getAbsolutePath() );
984         }
985         for ( Artifact artifact : getAllRelevantDependencyArtifacts() )
986         {
987             if ( artifact.getType().equals( APKLIB ) || artifact.getType().equals( AAR ) )
988             {
989                 final String apkLibResDir = getLibraryUnpackDirectory( artifact ) + "/res";
990                 if ( new File( apkLibResDir ).exists() )
991                 {
992                     commands.add( "-S" );
993                     commands.add( apkLibResDir );
994                 }
995             }
996         }
997         commands.add( "--auto-add-overlay" );
998 
999         // NB aapt only accepts a single assets parameter - combinedAssets is a merge of all assets
1000         if ( combinedAssets.exists() )
1001         {
1002             getLog().debug( "Adding assets folder : " + combinedAssets );
1003             commands.add( "-A" );
1004             commands.add( combinedAssets.getAbsolutePath() );
1005         }
1006 
1007         if ( StringUtils.isNotBlank( renameManifestPackage ) )
1008         {
1009             commands.add( "--rename-manifest-package" );
1010             commands.add( renameManifestPackage );
1011         }
1012 
1013         if ( StringUtils.isNotBlank( renameInstrumentationTargetPackage ) )
1014         {
1015             commands.add( "--rename-instrumentation-target-package" );
1016             commands.add( renameInstrumentationTargetPackage );
1017         }
1018 
1019         commands.add( "-I" );
1020         commands.add( androidJar.getAbsolutePath() );
1021         commands.add( "-F" );
1022         commands.add( outputFile.getAbsolutePath() );
1023         if ( StringUtils.isNotBlank( configurations ) )
1024         {
1025             commands.add( "-c" );
1026             commands.add( configurations );
1027         }
1028 
1029         for ( String aaptExtraArg : aaptExtraArgs )
1030         {
1031             commands.add( aaptExtraArg );
1032         }
1033 
1034         if ( !release )
1035         {
1036             getLog().info( "Enabling debug build for apk." );
1037             commands.add( "--debug-mode" );
1038         }
1039         else 
1040         {
1041             getLog().info( "Enabling release build for apk." );
1042         }
1043 
1044         getLog().info( getAndroidSdk().getAaptPath() + " " + commands.toString() );
1045         try
1046         {
1047             executor.executeCommand( getAndroidSdk().getAaptPath(), commands, project.getBasedir(), false );
1048         }
1049         catch ( ExecutionException e )
1050         {
1051             throw new MojoExecutionException( "", e );
1052         }
1053     }
