334     protected void instrument() throws MojoExecutionException, MojoFailureException
335     {
336         parseConfiguration();
337 
338         if ( parsedInstrumentationPackage == null )
339         {
340             parsedInstrumentationPackage = extractPackageNameFromAndroidManifest( androidManifestFile );
341         }
342 
343         if ( parsedInstrumentationRunner == null )
344         {
345             parsedInstrumentationRunner = extractInstrumentationRunnerFromAndroidManifest( androidManifestFile );
346         }
347 
348         // only run Tests in specific package
349         packagesList = buildCommaSeparatedString( parsedPackages );
350         packagesExists = StringUtils.isNotBlank( packagesList );
351 
352         if ( parsedClasses != null )
353         {
354             classesExists = parsedClasses.size() > 0;
355         }
356         else
357         {
358             classesExists = false;
359         }
360 
361         if ( classesExists && packagesExists )
362         {
363             // if both packages and classes are specified --> ERROR
364             throw new  MojoFailureException( "packages and classes are mutually exclusive. They cannot be specified at"
365                     + " the same time. Please specify either packages or classes. For details, see "
366                     + "http://developer.android.com/guide/developing/testing/testing_otheride.html" );
367         }

369         DeviceCallback instrumentationTestExecutor = new DeviceCallback()
370         {
371             public void doWithDevice( final IDevice device ) throws MojoExecutionException, MojoFailureException
372             {
373                 String deviceLogLinePrefix = DeviceHelper.getDeviceLogLinePrefix( device );
374 
375                 RemoteAndroidTestRunner remoteAndroidTestRunner = new RemoteAndroidTestRunner(
376                         parsedInstrumentationPackage, parsedInstrumentationRunner, device );
377 
378                 if ( packagesExists )
379                 {
380                     for ( String str : packagesList.split( "," ) )
381                     {
382                         remoteAndroidTestRunner.setTestPackageName( str );
383                         getLog().info( deviceLogLinePrefix + "Running tests for specified test package: " + str );
384                     }
385                 }
386 
387                 if ( classesExists )
388                 {
389                     remoteAndroidTestRunner
390                             .setClassNames( parsedClasses.toArray( new String[ parsedClasses.size() ] ) );
391                     getLog().info( deviceLogLinePrefix + "Running tests for specified test classes/methods: " 
392                             + parsedClasses );
393                 }
394 
395                 if ( parsedAnnotations != null )
396                 {
397                     for ( String annotation : parsedAnnotations )
398                     {
399                         remoteAndroidTestRunner.addInstrumentationArg( "annotation", annotation );
400                     }
401                 }
402 
403                 if ( parsedExcludeAnnotations != null )
404                 {
405                     for ( String annotation : parsedExcludeAnnotations )
406                     {
407                         remoteAndroidTestRunner.addInstrumentationArg( "notAnnotation", annotation );
408                     }
409 
410                 }
411 
412                 remoteAndroidTestRunner.setDebug( parsedDebug );
413                 remoteAndroidTestRunner.setCoverage( parsedCoverage );
414                 if ( ! "".equals( parsedCoverageFile ) )
415                 {
416                     remoteAndroidTestRunner.addInstrumentationArg( "coverageFile", parsedCoverageFile );
417                 }
418                 remoteAndroidTestRunner.setLogOnly( parsedLogOnly );
419 
420                 if ( StringUtils.isNotBlank( parsedTestSize ) )
421                 {
422                     IRemoteAndroidTestRunner.TestSize validSize = IRemoteAndroidTestRunner.TestSize
423                             .getTestSize( parsedTestSize );
424                     remoteAndroidTestRunner.setTestSize( validSize );
425                 }
426 
427                 addAllInstrumentationArgs( remoteAndroidTestRunner, parsedInstrumentationArgs );
428 
429                 getLog().info( deviceLogLinePrefix +  "Running instrumentation tests in " 
430                         + parsedInstrumentationPackage );
431                 try
432                 {
433                     AndroidTestRunListener testRunListener = new AndroidTestRunListener( project, device );
434                     remoteAndroidTestRunner.run( testRunListener );
435                     if ( testRunListener.hasFailuresOrErrors() )
436                     {
437                         throw new MojoFailureException( deviceLogLinePrefix +  "Tests failed on device." );
438                     }
439                     if ( testRunListener.testRunFailed() )
440                     {
441                         throw new MojoFailureException( deviceLogLinePrefix + "Test run failed to complete: " 
442                                 + testRunListener.getTestRunFailureCause() );
443                     }
444                     if ( testRunListener.threwException() )
445                     {
446                         throw new MojoFailureException( deviceLogLinePrefix +  testRunListener.getExceptionMessages() );
447                     }
448                 }
449                 catch ( TimeoutException e )
450                 {
451                     throw new MojoExecutionException( deviceLogLinePrefix + "timeout", e );
452                 }
453                 catch ( AdbCommandRejectedException e )
454                 {
455                     throw new MojoExecutionException( deviceLogLinePrefix + "adb command rejected", e );
456                 }
457                 catch ( ShellCommandUnresponsiveException e )
458                 {
459                     throw new MojoExecutionException( deviceLogLinePrefix + "shell command " + "unresponsive", e );
460                 }
461                 catch ( IOException e )
462                 {
463                     throw new MojoExecutionException( deviceLogLinePrefix + "IO problem", e );
464                 }
465             }
466         };
467 
468         instrumentationTestExecutor = new ScreenshotServiceWrapper( instrumentationTestExecutor, project, getLog() );
469 
470         doWithDevices( instrumentationTestExecutor );
471     }

473     private void addAllInstrumentationArgs(

483     private void parseConfiguration()
484     {
485         // we got config in pom ... lets use it,
486         if ( test != null )
487         {
488             if ( StringUtils.isNotEmpty( test.getSkip() ) )
489             {
490                 parsedSkip = test.getSkip();
491             }
492             else
493             {
494                 parsedSkip = testSkip;
495             }
496             if ( StringUtils.isNotEmpty( test.getInstrumentationPackage() ) )
497             {
498                 parsedInstrumentationPackage = test.getInstrumentationPackage();
499             }
500             else
501             {
502                 parsedInstrumentationPackage = testInstrumentationPackage;
503             }
504             if ( StringUtils.isNotEmpty( test.getInstrumentationRunner() ) )
505             {
506                 parsedInstrumentationRunner = test.getInstrumentationRunner();
507             }
508             else
509             {
510                 parsedInstrumentationRunner = testInstrumentationRunner;
511             }
512             if ( test.getClasses() != null && ! test.getClasses().isEmpty() )
513             {
514                 parsedClasses = test.getClasses();
515             }
516             else
517             {
518                 parsedClasses = testClasses;
519             }
520             if ( test.getAnnotations() != null && ! test.getAnnotations().isEmpty() )
521             {
522                 parsedAnnotations = test.getAnnotations();
523             }
524             else
525             {
526                 parsedAnnotations =  testAnnotations;
527             }
528             if ( test.getExcludeAnnotations() != null && ! test.getExcludeAnnotations().isEmpty() )
529             {
530                 parsedExcludeAnnotations = test.getExcludeAnnotations();
531             }
532             else
533             {
534                 parsedExcludeAnnotations = testExcludeAnnotations;
535             }
536             if ( test.getPackages() != null && ! test.getPackages().isEmpty() )
537             {
538                 parsedPackages = test.getPackages();
539             }
540             else
541             {
542                 parsedPackages = testPackages;
543             }
544             if ( StringUtils.isNotEmpty( test.getTestSize() ) )
545             {
546                 parsedTestSize = test.getTestSize();
547             }
548             else
549             {
550                 parsedTestSize = testTestSize;
551             }
552             if ( test.isCoverage() != null )
553             {
554                 parsedCoverage = test.isCoverage();
555             }
556             else
557             {
558                 parsedCoverage = testCoverage;
559             }
560             if ( test.getCoverageFile() != null )
561             {
562                 parsedCoverageFile = test.getCoverageFile();
563             }
564             else
565             {
566                 parsedCoverageFile = "";
567             }
568             if ( test.isDebug() != null )
569             {
570                 parsedDebug = test.isDebug();
571             }
572             else
573             {
574                 parsedDebug = testDebug;
575             }
576             if ( test.isLogOnly() != null )
577             {
578                 parsedLogOnly = test.isLogOnly();
579             }
580             else
581             {
582                 parsedLogOnly = testLogOnly;
583             }
584             if ( test.isCreateReport() != null )
585             {
586                 parsedCreateReport = test.isCreateReport();
587             }
588             else
589             {
590                 parsedCreateReport = testCreateReport;
591             }
592 
593             parsedInstrumentationArgs = InstrumentationArgumentParser.parse( test.getInstrumentationArgs() );
594         }
595         // no pom, we take properties
596         else
597         {
598             parsedSkip = testSkip;
599             parsedInstrumentationPackage = testInstrumentationPackage;
600             parsedInstrumentationRunner = testInstrumentationRunner;
601             parsedClasses = testClasses;
602             parsedAnnotations = testAnnotations;
603             parsedExcludeAnnotations = testExcludeAnnotations;
604             parsedPackages = testPackages;
605             parsedTestSize = testTestSize;
606             parsedCoverage = testCoverage;
607             parsedCoverageFile = testCoverageFile;
608             parsedDebug = testDebug;
609             parsedLogOnly = testLogOnly;
610             parsedCreateReport = testCreateReport;
611             parsedInstrumentationArgs = InstrumentationArgumentParser.parse( testInstrumentationArgs );
612         }
613     }

615     /**
616      * Whether or not to execute integration test related goals. Reads from configuration parameter
617      * <code>enableIntegrationTest</code>, but can be overridden with <code>-Dmaven.test.skip</code>.
618      *
619      * @return <code>true</code> if integration test goals should be executed, <code>false</code> otherwise.
620      */
621     protected boolean isEnableIntegrationTest() throws MojoFailureException, MojoExecutionException
622     {
623         parseConfiguration();
624         if ( mavenTestSkip )
625         {
626             getLog().info( "maven.test.skip set - skipping tests" );
627             return false;
628         }
629 
630         if ( mavenSkipTests )
631         {
632             getLog().info( "maven.skip.tests set - skipping tests" );
633             return false;
634         }
635 
636         if ( "true".equalsIgnoreCase( parsedSkip ) )
637         {
638             getLog().info( "android.test.skip set - skipping tests" );
639             return false;
640         }
641 
642         if ( "false".equalsIgnoreCase( parsedSkip ) )
643         {
644             return true;
645         }
646 
647         if ( parsedSkip == null || "auto".equalsIgnoreCase( parsedSkip ) )
648         {
649             if ( extractInstrumentationRunnerFromAndroidManifest( androidManifestFile ) == null )
650             {
651                 getLog().info( "No InstrumentationRunner found - skipping tests" );
652                 return false;
653             }
654             return AndroidTestFinder.containsAndroidTests( new File( project.getBuild().getOutputDirectory() ) );
655         }
656 
657         throw new MojoFailureException( "android.test.skip must be configured as 'true', 'false' or 'auto'." );
658 
659     }

661     /**
662      * Helper method to build a comma separated string from a list.
663      * Blank strings are filtered out
664      *
665      * @param lines A list of strings
666      * @return Comma separated String from given list
667      */
668     protected static String buildCommaSeparatedString( List<String> lines )
669     {
670         if ( lines == null || lines.size() == 0 )
671         {
672             return null;
673         }
674 
675         List<String> strings = new ArrayList<String>( lines.size() );
676         for ( String str : lines )
677         { // filter out blank strings
678             if ( StringUtils.isNotBlank( str ) )
679             {
680                 strings.add( StringUtils.trimToEmpty( str ) );
681             }
682         }
683 
684         return StringUtils.join( strings, "," );
685     }

687     /**
688      * AndroidTestRunListener produces a nice output for the log for the test
689      * run as well as an xml file compatible with the junit xml report file
690      * format understood by many tools.
691      * <p/>
692      * It will do so for each device/emulator the tests run on.
693      */
694     private class AndroidTestRunListener implements ITestRunListener
695     {
696         /**
697          * the indent used in the log to group items that belong together visually *
698          */
699         private static final String INDENT = "  ";
700 
701         /**
702          * Junit report schema documentation is sparse. Here are some hints

1001         /**
1002          * Parse a trace string for the message in it. Assumes that the message is located after ":" and before
1003          * "\r\n".
1004          *
1005          * @param trace
1006          * @return message or empty string
1007          */
1008         private String parseForMessage( String trace )
1009         {
1010             if ( StringUtils.isNotBlank( trace ) )
1011             {
1012                 String newline = "\r\n";
1013                 // if there is message like
1014                 // junit.junit.framework.AssertionFailedError ... there is no message
1015                 int messageEnd = trace.indexOf( newline );
1016                 boolean hasMessage = ! trace.startsWith( "junit." ) && messageEnd > 0;
1017                 if ( hasMessage )
1018                 {
1019                     int messageStart = trace.indexOf( ":" ) + 2;
1020                     if ( messageStart > messageEnd )
1021                     {
1022                         messageEnd = trace.indexOf( newline + "at" );
1023                         // match start of stack trace "\r\nat org.junit....."
1024                         if ( messageStart > messageEnd )
1025                         {
1026                             //':' wasn't found in message but in stack trace
1027                             messageStart = 0;
1028                         }
1029                     }
1030                     return trace.substring( messageStart, messageEnd );
1031                 }
1032                 else
1033                 {
1034                     return StringUtils.EMPTY;
1035                 }
1036             }
1037             else
1038             {
1039                 return StringUtils.EMPTY;
1040             }
1041         }

1043         /**
1044          * Parse a trace string for the exception class. Assumes that it is the start of the trace and ends at the first
1045          * ":".
1046          *
1047          * @param trace
1048          * @return Exception class as string or empty string
1049          */
1050         private String parseForException( String trace )
1051         {
1052             if ( StringUtils.isNotBlank( trace ) )
1053             {
1054                 return trace.substring( 0, trace.indexOf( ":" ) );
1055             }
1056             else
1057             {
1058                 return StringUtils.EMPTY;
1059             }
1060         }

1063          * Write the junit report xml file.
1064          */
1065         private void writeJunitReportToFile()
1066         {
1067             TransformerFactory xfactory = TransformerFactory.newInstance();
1068             Transformer xformer = null;
1069             try
1070             {
1071                 xformer = xfactory.newTransformer();
1072             }
1073             catch ( TransformerConfigurationException e )
1074             {
1075                 e.printStackTrace();
1076             }
1077             Source source = new DOMSource( junitReport );
1078 
1079             FileWriter writer = null;
1080             try
1081             {
1082                 String directory = new StringBuilder().append( project.getBuild().getDirectory() )
1083                         .append( "/surefire-reports" ).toString();
1084 
1085                 FileUtils.forceMkdir( new File( directory ) );
1086 
1087                 String fileName = new StringBuilder().append( directory ).append( "/TEST-" )
1088                         .append( DeviceHelper.getDescriptiveName( device ) ).append( ".xml" ).toString();
1089                 File reportFile = new File( fileName );
1090                 writer = new FileWriter( reportFile );
1091                 Result result = new StreamResult( writer );
1092 
1093                 xformer.transform( source, result );
1094                 getLog().info( deviceLogLinePrefix + "Report file written to " + reportFile.getAbsolutePath() );
1095             }
1096             catch ( IOException e )
1097             {
1098                 threwException = true;
1099                 exceptionMessages.append( "Failed to write test report file" );
1100                 exceptionMessages.append( e.getMessage() );
1101             }
1102             catch ( TransformerException e )
1103             {
1104                 threwException = true;
1105                 exceptionMessages.append( "Failed to transform document to write to test report file" );
1106                 exceptionMessages.append( e.getMessage() );
1107             }
1108             finally
1109             {
1110                 IOUtils.closeQuietly( writer );
1111             }
1112         }
