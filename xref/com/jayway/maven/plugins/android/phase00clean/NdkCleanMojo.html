1   package com.jayway.maven.plugins.android.phase00clean;
2   
3   import org.apache.maven.plugin.AbstractMojo;
4   import org.apache.maven.plugin.MojoExecutionException;
5   import org.apache.maven.plugin.MojoFailureException;
6   import org.apache.maven.plugins.annotations.LifecyclePhase;
7   import org.apache.maven.plugins.annotations.Mojo;
8   import org.apache.maven.plugins.annotations.Parameter;
9   import org.codehaus.plexus.util.FileUtils;
10  
11  import java.io.File;
12  import java.io.IOException;
13  
14  /**
15   * @author Johan Lindquist <johanlindquist@gmail.com>
16   */
17  @Mojo(
18          name = "clean",
19          defaultPhase = LifecyclePhase.CLEAN,
20          requiresOnline = true // TODO Why?
21  )
22  public class NdkCleanMojo extends AbstractMojo
23  {
24  
25      @Parameter( property = "android.nativeBuildLibsOutputDirectory", defaultValue = "${project.basedir}/libs" )
26      File ndkBuildLibsOutputDirectory;
27  
28      @Parameter( property = "android.nativeBuildObjOutputDirectory", defaultValue = "${project.basedir}/obj" )
29      File ndkBuildObjOutputDirectory;
30  
31      /**
32       * Forces the clean process to be skipped.
33       */
34      @Parameter( property = "android.nativeBuildSkipClean", defaultValue = "false" )
35      boolean skipClean = false;
36  
37      /**
38       * Specifies whether the deletion of the libs/ folder structure should be skipped.  This is by default set to
39       * skip (true) to avoid unwanted deletions of libraries already present in this structure.
40       */
41      @Parameter( property = "android.nativeBuildSkipCleanLibsOutputDirectory", defaultValue = "true" )
42      boolean skipBuildLibsOutputDirectory = true;
43  
44      /**
45       * Specifies whether the obj/ build folder structure should be deleted.
46       */
47      @Parameter( property = "android.nativeBuildSkipCleanLibsOutputDirectory", defaultValue = "false" )
48      boolean skipBuildObjsOutputDirectory = false;
49  
50      @Override
51      public void execute() throws MojoExecutionException, MojoFailureException
52      {
53          if ( ndkBuildLibsOutputDirectory.exists() )
54          {
55              if ( ! skipBuildLibsOutputDirectory )
56              {
57                  getLog().debug( "Cleaning out native library code directory : " + ndkBuildLibsOutputDirectory
58                          .getAbsolutePath() );
59                  try
60                  {
61                      FileUtils.deleteDirectory( ndkBuildLibsOutputDirectory );
62                  }
63                  catch ( IOException e )
64                  {
65                      getLog().error( "Error deleting directory: " + e.getMessage(), e );
66                  }
67              }
68          }
69  
70          if ( ndkBuildObjOutputDirectory.exists() )
71          {
72              if ( ! skipBuildObjsOutputDirectory )
73              {
74                  getLog().debug(
75                          "Cleaning out native object code directory: " + ndkBuildObjOutputDirectory.getAbsolutePath() );
76                  try
77                  {
78                      FileUtils.deleteDirectory( ndkBuildObjOutputDirectory );
79                  }
80                  catch ( IOException e )
81                  {
82                      getLog().error( "Error deleting directory: " + e.getMessage(), e );
83                  }
84              }
85          }
86  
87      }
88  
89  }
