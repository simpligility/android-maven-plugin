164     /**
165      * Start the Android Emulator with the specified options.
166      *
167      * @throws org.apache.maven.plugin.MojoExecutionException
168      *
169      * @see #emulatorAvd
170      * @see #emulatorWait
171      * @see #emulatorOptions
172      */
173     protected void startAndroidEmulator() throws MojoExecutionException
174     {
175         parseParameters();
176 
177         CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
178         executor.setLogger( this.getLog() );
179 
180         try
181         {
182             String filename;
183             if ( isWindows() )
184             {
185                 filename = writeEmulatorStartScriptWindows();
186             }
187             else
188             {
189                 filename = writeEmulatorStartScriptUnix();
190             }
191 
192             final AndroidDebugBridge androidDebugBridge = initAndroidDebugBridge();
193             if ( androidDebugBridge.isConnected() )
194             {
195                 waitForInitialDeviceList( androidDebugBridge );
196                 List<IDevice> devices = Arrays.asList( androidDebugBridge.getDevices() );
197                 int numberOfDevices = devices.size();
198                 getLog().info( "Found " + numberOfDevices + " devices connected with the Android Debug Bridge" );
199 
200                 IDevice existingEmulator = findExistingEmulator( devices );
201                 if ( existingEmulator == null )
202                 {
203                     getLog().info( START_EMULATOR_MSG + filename );
204                     executor.executeCommand( filename, null );
205 
206                     getLog().info( START_EMULATOR_WAIT_MSG + parsedWait );
207                     // wait for the emulator to start up
208                     boolean connected = waitUntilDeviceIsConnectedOrTimeout( androidDebugBridge );
209                     if ( connected )
210                     {
211                         getLog().info( "Emulator is up and running." );
212                         unlockEmulator( androidDebugBridge );
213                     }
214                     else
215                     {
216                         throw new MojoExecutionException( "Timeout while waiting for emulator to startup." );
217                     }
218 
219                 }
220                 else
221                 {
222                     getLog().info( String.format(
223                             "Emulator already running [Serial No: '%s', AVD Name '%s']. " + "Skipping start and wait.",
224                             existingEmulator.getSerialNumber(), existingEmulator.getAvdName() ) );
225                 }
226             }
227         }
228         catch ( Exception e )
229         {
230             throw new MojoExecutionException( "", e );
231         }
232     }

234     /**
235      * Unlocks the emulator.
236      * @param androidDebugBridge
237      */
238     void unlockEmulator( AndroidDebugBridge androidDebugBridge )
239     {
240         if ( emulatorUnlock )
241         {
242             IDevice myEmulator = findExistingEmulator( Arrays.asList( androidDebugBridge.getDevices() ) );
243             int devicePort = extractPortFromDevice( myEmulator );
244             if ( devicePort == -1 )
245             {
246                 getLog().info( "Unable to retrieve port to unlock emulator "
247                         + DeviceHelper.getDescriptiveName( myEmulator ) );
248             }
249             else
250             {
251                 getLog().info( "Unlocking emulator "
252                         + DeviceHelper.getDescriptiveName( myEmulator ) );
253 
254                 sendEmulatorCommand( devicePort,
255                         "event send EV_KEY:KEY_SOFT1:1" );
256                 sendEmulatorCommand( devicePort,
257                         "event send EV_KEY:KEY_SOFT1:0" );
258                 sendEmulatorCommand( devicePort,
259                         "event send EV_KEY:KEY_SOFT1:1" );
260                 sendEmulatorCommand( devicePort,
261                         "event send EV_KEY:KEY_SOFT1:0" );
262             }
263         }

266     boolean waitUntilDeviceIsConnectedOrTimeout( AndroidDebugBridge androidDebugBridge )
267             throws MojoExecutionException
268     {
269         long timeout = System.currentTimeMillis() + new Long( parsedWait );
270         while ( System.currentTimeMillis() < timeout )
271         {
272             IDevice myEmulator = findExistingEmulator( Arrays.asList( androidDebugBridge.getDevices() ) );
273             if ( ( myEmulator != null ) && ( myEmulator.isOnline() ) )
274             {
275                 return true;
276             }
277 
278             try
279             {
280                 Thread.sleep( MILLIS_TO_SLEEP_BETWEEN_DEVICE_ONLINE_CHECKS );
281             }
282             catch ( InterruptedException e )
283             {
284                 throw new MojoExecutionException( "Interrupted waiting for device to become ready" );
285             }
286         }
287         return false;
288     }

290     private IDevice findExistingEmulator( List<IDevice> devices )
291     {
292         IDevice existingEmulator = null;
293 
294         for ( IDevice device : devices )
295         {
296             if ( device.isEmulator() )
297             {
298                 if ( isExistingEmulator( device ) )
299                 {
300                     existingEmulator = device;
301                     break;
302                 }
303             }
304         }
305         return existingEmulator;
306     }

322      * Writes the script to start the emulator in the background for windows based environments.
323      *
324      * @return absolute path name of start script
325      * @throws IOException
326      * @throws MojoExecutionException
327      */
328     private String writeEmulatorStartScriptWindows() throws MojoExecutionException
329     {
330 
331         String filename = SCRIPT_FOLDER + "\\android-maven-plugin-emulator-start.vbs";
332 
333         File file = new File( filename );
334         PrintWriter writer = null;
335         try
336         {
337             writer = new PrintWriter( new FileWriter( file ) );
338 
339 
340             // command needs to be assembled before unique window title since it parses settings and sets up parsedAvd
341             // and others.
342             String command = assembleStartCommandLine();
343             String uniqueWindowTitle = "AndroidMavenPlugin-AVD" + parsedAvd;
344             writer.println( "Dim oShell" );
345             writer.println( "Set oShell = WScript.CreateObject(\"WScript.shell\")" );
346             String cmdPath = System.getenv( "COMSPEC" );
347             if ( cmdPath == null )
348             {
349                 cmdPath = "cmd.exe";
350             }
351             String cmd = cmdPath + " /X /C START /SEPARATE \"\"" + uniqueWindowTitle + "\"\"  " + command.trim();
352             writer.println( "oShell.run \"" + cmd + "\"" );
353         }
354         catch ( IOException e )
355         {
356             getLog().error( "Failure writing file " + filename );
357         }
358         finally
359         {
360             if ( writer != null )
361             {
362                 writer.flush();
363                 writer.close();
364             }
365         }
366         file.setExecutable( true );
367         return filename;
368     }

386         }
387         if ( !sh.exists() )

399             writer.print( " 1>/dev/null 2>&1 &" ); // redirect outputs and run as background task
400         }
401         catch ( IOException e )

480     /**
481      * This method contains the code required to stop an emulator
482      *
483      * @param device The device to stop
484      */
485     private void stopEmulator( IDevice device )
486     {
487         int devicePort = extractPortFromDevice( device );
488         if ( devicePort == -1 )
489         {
490             getLog().info( "Unable to retrieve port to stop emulator " + DeviceHelper.getDescriptiveName( device ) );
491         }
492         else
493         {
494             getLog().info( "Stopping emulator " + DeviceHelper.getDescriptiveName( device ) );
495 
496             sendEmulatorCommand( devicePort, "avd stop" );
497             boolean killed = sendEmulatorCommand( devicePort, "kill" );
498             if ( !killed )
499             {
500                 getLog().info( "Emulator failed to stop " + DeviceHelper.getDescriptiveName( device ) );
501             }
502             else
503             {
504                 getLog().info( "Emulator stopped successfully " + DeviceHelper.getDescriptiveName( device ) );
505             }
506         }
507     }

509     /**
510      * This method extracts a port number from the serial number of a device.
511      * It assumes that the device name is of format [xxxx-nnnn] where nnnn is the
512      * port number.
513      *
514      * @param device The device to extract the port number from.
515      * @return Returns the port number of the device
516      */
517     private int extractPortFromDevice( IDevice device )
518     {
519         String portStr = StringUtils.substringAfterLast( device.getSerialNumber(), "-" );
520         if ( StringUtils.isNotBlank( portStr ) && StringUtils.isNumeric( portStr ) )
521         {
522             return Integer.parseInt( portStr );
523         }
524 
525         //If the port is not available then return -1
526         return -1;
527     }

529     /**
530      * Sends a user command to the running emulator via its telnet interface.
531      *
532      * @param port    The emulator's telnet port.
533      * @param command The command to execute on the emulator's telnet interface.
534      * @return Whether sending the command succeeded.
535      */
536     private boolean sendEmulatorCommand(
537             //final Launcher launcher,
538             //final PrintStream logger,
539             final int port, final String command )
540     {
541         Callable<Boolean> task = new Callable<Boolean>()
542         {
543             public Boolean call() throws IOException
544             {
545                 Socket socket = null;
546                 BufferedReader in = null;
547                 PrintWriter out = null;
548                 try
549                 {
550                     socket = new Socket( "127.0.0.1", port );
551                     out = new PrintWriter( socket.getOutputStream(), true );
552                     in = new BufferedReader( new InputStreamReader( socket.getInputStream() ) );
553                     if ( in.readLine() == null )
554                     {
555                         return false;
556                     }
557 
558                     out.write( command );
559                     out.write( "\r\n" );
560                 }
561                 finally
562                 {
563                     try
564                     {
565                         out.close();
566                         in.close();
567                         socket.close();
568                     }
569                     catch ( Exception e )
570                     {
571                         // Do nothing
572                     }
573                 }
574 
575                 return true;
576             }
577 
578             private static final long serialVersionUID = 1L;
579         };
580 
581         boolean result = false;
582         try
583         {
584             ExecutorService executor = Executors.newSingleThreadExecutor();
585             Future<Boolean> future = executor.submit( task );
586             result = future.get();
587         }
588         catch ( Exception e )
589         {
590             getLog().error( String.format( "Failed to execute emulator command '%s': %s", command, e ) );
591         }
592 
593         return result;
594     }

597      * Assemble the command line for starting the emulator based on the parameters supplied in the pom file and on the
598      * command line. It should not be that painful to do work with command line and pom supplied values but evidently

605     private String assembleStartCommandLine() throws MojoExecutionException
606     {
607         String emulatorPath = new File ( getAndroidSdk().getToolsPath(), parsedExecutable ).getAbsolutePath();
608         StringBuilder startCommandline = new StringBuilder( "\"\"" ).append( emulatorPath ).append( "\"\"" )
609                 .append( " -avd " ).append( parsedAvd ).append( " " );
610         if ( !StringUtils.isEmpty( parsedOptions ) )
611         {
612             startCommandline.append( parsedOptions );
613         }
614         getLog().info( "Android emulator command: " + startCommandline );
615         return startCommandline.toString();

618     private void parseParameters()
619     {
620         // <emulator> exist in pom file
621         if ( emulator != null )
622         {
623             // <emulator><avd> exists in pom file
624             if ( emulator.getAvd() != null )
625             {
626                 parsedAvd = emulator.getAvd();
627             }
628             else
629             {
630                 parsedAvd = determineAvd();
631             }
632             // <emulator><options> exists in pom file
633             if ( emulator.getOptions() != null )
634             {
635                 parsedOptions = emulator.getOptions();
636             }
637             else
638             {
639                 parsedOptions = determineOptions();
640             }
641             // <emulator><wait> exists in pom file
642             if ( emulator.getWait() != null )
643             {
644                 parsedWait = emulator.getWait();
645             }
646             else
647             {
648                 parsedWait = determineWait();
649             }
650             // <emulator><emulatorExecutable> exists in pom file
651             if ( emulator.getExecutable() != null )
652             {
653                 parsedExecutable = emulator.getExecutable();
654             }
655             else
656             {
657                 parsedExecutable = determineExecutable();
658             }
659         }
660         // commandline options
661         else
662         {
663             parsedAvd = determineAvd();
664             parsedOptions = determineOptions();
665             parsedWait = determineWait();
666             parsedExecutable = determineExecutable();
667         }
668     }
