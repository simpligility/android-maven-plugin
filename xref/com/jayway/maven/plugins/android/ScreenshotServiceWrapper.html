20  /**
21   * ScreenshotServiceWrapper wraps the feature to capture a screenshot during an instrumentation test run.
22   */
23  public class ScreenshotServiceWrapper implements DeviceCallback
24  {
25  
26      private final DeviceCallback delegate;
27      private final Log log;
28      private final File screenshotParentDir;
29      private static final int MAX_BOUNDS = 320;
30  
31      public ScreenshotServiceWrapper( DeviceCallback delegate, MavenProject project, Log log )
32      {
33          this.delegate = delegate;
34          this.log = log;
35          screenshotParentDir = new File( project.getBuild().getDirectory(), "screenshots" );
36          create( screenshotParentDir );
37      }
38  
39  
40      @Override
41      public void doWithDevice( final IDevice device ) throws MojoExecutionException, MojoFailureException
42      {
43          String deviceName = getDescriptiveName( device );
44  
45          File deviceGifFile = new File( screenshotParentDir, deviceName + ".gif" );
46          File deviceScreenshotDir = new File( screenshotParentDir, deviceName );
47          create( deviceScreenshotDir );
48  
49  
50          OnDemandScreenshotService screenshotService = new OnDemandScreenshotService( device,
51                  new ImageSaver( deviceScreenshotDir ),
52                  new ImageScaler( new AnimatedGifCreator( deviceGifFile ), square( MAX_BOUNDS ) ) );
53  
54          screenshotService.start();
55  
56          delegate.doWithDevice( device );
57  
58          screenshotService.finish();
59      }
60  
61      private void create( File dir )
62      {
63          try
64          {
65              forceMkdir( dir );
66          }
67          catch ( IOException e )
68          {
69              log.warn( "Unable to create screenshot directory: " + screenshotParentDir.getAbsolutePath(), e );
70          }
71      }
