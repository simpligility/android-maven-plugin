22  @RunWith( MavenJUnitTestRunner.class )
23  @MavenVersions( {"3.0.5", "3.2.5" } )
24  public class LintMojoIntegrationTest
25  {
26      @Rule
27      public final TestResources resources = new TestResources();
28  
29      public final MavenRuntime verifier;
30  
31      public LintMojoIntegrationTest(MavenRuntime.MavenRuntimeBuilder builder)
32      throws Exception
33      {
34          this.verifier = builder.withCliOptions( "-X" ).build();
35      }
36  
37      /**
38       * Test project with skip lint flag to true
39       * 
40       * @throws Exception
41       */
42      @Test
43      public void testSkipConfig() throws Exception
44      {
45          File basedir = resources.getBasedir( "lint-config-project0" );
46          MavenExecutionResult result = verifier
47                  .forProject(basedir)
48                  .execute( PluginInfo.getQualifiedGoal( "lint" ) );
49  
50          result.assertErrorFreeLog();
51  
52          result.assertLogText( "Skipping lint analysis" );
53      }
54  
55      /**
56       * Tests execution of Lint
57       * 
58       * @throws Exception
59       */
60      @Test
61      public void testDefaultUnskippedLintConfig() throws Exception
62      {
63          File basedir = resources.getBasedir( "lint-config-project1" );
64          MavenExecutionResult result = verifier
65                  .forProject(basedir)
66                  .execute( PluginInfo.getQualifiedGoal( "lint" ) );
67  
68          result.assertErrorFreeLog();
69  
70          result.assertLogText( "Lint analysis completed successfully" );
71      }
