202     public void testAllLintCommandParametersWithDefaultUnskippedConfig() throws Exception
203     {
204         LintMojo mojo = createMojo( "lint-config-project1" );
205 
206         MavenProject project = EasyMock.createNiceMock( MavenProject.class );
207         Whitebox.setInternalState( mojo, "project", project );
208         File projectBaseDir = new File( getBasedir() );
209         Build projectBuild = new Build();
210         projectBuild.setDirectory( "target/" );
211         projectBuild.setSourceDirectory( "src/" );
212         projectBuild.setOutputDirectory( "classes/" );
213         EasyMock.expect( project.getBasedir() ).andReturn( projectBaseDir ).anyTimes();
214         EasyMock.expect( project.getBuild() ).andReturn( projectBuild ).anyTimes();
215         final CommandExecutor mockExecutor = PowerMock.createMock( CommandExecutor.class );
216         PowerMock.replace( CommandExecutor.Factory.class.getDeclaredMethod( "createDefaultCommmandExecutor" ) ).with(
217                 new InvocationHandler()
218                 {
219 
220                     @Override
221                     public Object invoke( Object proxy, Method method, Object[] args ) throws Throwable
222                     {
223                         return mockExecutor;
224                     }
225                 } );
226 
227         PowerMock.expectNew( ConfigHandler.class, mojo, this.session, this.execution )
228             .andReturn( EasyMock.createNiceMock( ConfigHandler.class ) );
229         Capture< List< String > > capturedArgument = new Capture< List< String > >();
230 
231         mockExecutor.setLogger( EasyMock.anyObject( Log.class ) );
232         mockExecutor.executeCommand( EasyMock.anyObject( String.class ), EasyMock.capture( capturedArgument ),
233                 EasyMock.eq( false ) );
234         PowerMock.replay( project );
235         PowerMock.replay( mockExecutor );
236 
237         mojo.execute();
238 
239         PowerMock.verify( mockExecutor );
240         List< String > parameters = capturedArgument.getValue();
241         List< String > parametersExpected = new ArrayList< String >();
242         parametersExpected.add( "--showall" );
243         parametersExpected.add( "--xml" );
244         parametersExpected.add( projectBaseDir.getAbsolutePath()
245                 + FilenameUtils.separatorsToSystem( "/target/lint-results/lint-results.xml" ) );
246         parametersExpected.add( "--sources" );
247         parametersExpected.add( projectBaseDir.getAbsolutePath() + File.separator + "src" );
248         parametersExpected.add( projectBaseDir.getAbsolutePath() );
249         parametersExpected.add( "--exitcode" );
250         assertEquals( parametersExpected, parameters );
251     }
