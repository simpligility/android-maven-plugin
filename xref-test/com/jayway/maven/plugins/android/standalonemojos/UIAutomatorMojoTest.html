56      /**
57       * Tests all options, checks if their default values are correct.
58       * 
59       * @throws Exception
60       */
61      public void testDefaultUnskippedUIAutomatorConfig() throws Exception
62      {
63          // given
64          UIAutomatorMojo mojo = createMojo( "ui-automator-config-project1" );
65  
66          MavenProject project = EasyMock.createNiceMock( MavenProject.class );
67          Whitebox.setInternalState( mojo, "project", project );
68          File projectBaseDir = new File( getBasedir() );
69          Build projectBuild = new Build();
70          String buildName = "ui-automator-config-project1-15.4.3.1011";
71          projectBuild.setFinalName( buildName );
72          projectBuild.setDirectory( "target/" );
73          projectBuild.setSourceDirectory( "src/" );
74          projectBuild.setOutputDirectory( "classes/" );
75          EasyMock.expect( project.getBasedir() ).andReturn( projectBaseDir ).anyTimes();
76          EasyMock.expect( project.getBuild() ).andReturn( projectBuild ).anyTimes();
77          PowerMock.replay( project );
78  
79          // when
80          final ConfigHandler cfh = new ConfigHandler( mojo, this.session, this.execution );
81          cfh.parseConfiguration();
82  
83          // then
84          Boolean automatorSkip = Whitebox.getInternalState( mojo, "parsedSkip" );
85          Boolean automatorDebug = Whitebox.getInternalState( mojo, "parsedDebug" );
86          String automatorJarFile = Whitebox.getInternalState( mojo, "parsedJarFile" );
87          String[] automatorTestClassOrMethods = Whitebox.getInternalState( mojo, "parsedTestClassOrMethods" );
88          Boolean automatorTakeScreenshotOnFailure = Whitebox.getInternalState( mojo, "parsedTakeScreenshotOnFailure" );
89          String automatorScreenshotsPathOnDevice = Whitebox.getInternalState( mojo, "parsedScreenshotsPathOnDevice" );
90  		Boolean automatorCreateReport = Whitebox.getInternalState( mojo, "parsedCreateReport" );
91  		String automatorReportSuffix = Whitebox.getInternalState( mojo, "parsedReportSuffix" );
92  		String automatorPropertiesKeyPrefix = Whitebox.getInternalState( mojo, "parsedPropertiesKeyPrefix" );
93  
94          assertFalse( "UIAutomator skip parameter should be false", automatorSkip );
95          assertFalse( "UIAutomator debug parameter should be false", automatorDebug );
96          String expectedJarFile = buildName + ".jar";
97          assertNotNull( "UIAutomator jarFile parameter should be not null", automatorJarFile );
98          assertEquals( "UIAutomator jarFile parameter should be equal to artifact name", expectedJarFile,
99                  automatorJarFile );
100         assertNull( "UIAutomator testClassOrMethods parameter should be null", automatorTestClassOrMethods );
101         assertFalse( "UIAutomator takeScreenshotOnFailure parameter should be equal false",
102                 automatorTakeScreenshotOnFailure );
103         assertEquals( "UIAutomator screenshotsPath on device be equal /sdcard/uiautomator-screenshots/",
104                 "/sdcard/uiautomator-screenshots/", automatorScreenshotsPathOnDevice );
105 		assertFalse( "UIAutomator createReport parameter should be false", automatorCreateReport );
106 		assertNull( "UIAutomator reportSuffix parameter should be null", automatorReportSuffix );
107 		assertNull( "UIAutomator propertiesKeyPrefix parameter should be null", automatorPropertiesKeyPrefix );
108     }

110     /**
111      * Tests all options, checks if they are parsed correctly.
112      * 
113      * @throws Exception
114      */
115     public void testCustomUIAutomatorConfig() throws Exception
116     {
117         // given
118         UIAutomatorMojo mojo = createMojo( "ui-automator-config-project2" );
119         MavenProject project = EasyMock.createNiceMock( MavenProject.class );
120         Whitebox.setInternalState( mojo, "project", project );
121         File projectBaseDir = new File( getBasedir() );
122         Build projectBuild = new Build();
123         String buildName = "ui-automator-config-project1-15.4.3.1011";
124         projectBuild.setFinalName( buildName );
125         projectBuild.setDirectory( "target/" );
126         projectBuild.setSourceDirectory( "src/" );
127         projectBuild.setOutputDirectory( "classes/" );
128         EasyMock.expect( project.getBasedir() ).andReturn( projectBaseDir ).anyTimes();
129         EasyMock.expect( project.getBuild() ).andReturn( projectBuild ).anyTimes();
130 
131         PowerMock.replay( project );
132 
133         // when
134         final ConfigHandler cfh = new ConfigHandler( mojo, this.session, this.execution );
135         cfh.parseConfiguration();
136 
137         // then
138         Boolean automatorSkip = Whitebox.getInternalState( mojo, "parsedSkip" );
139         Boolean automatorDebug = Whitebox.getInternalState( mojo, "parsedDebug" );
140         String automatorJarFile = Whitebox.getInternalState( mojo, "parsedJarFile" );
141         String[] automatorTestClassOrMethods = Whitebox.getInternalState( mojo, "parsedTestClassOrMethods" );
142         Boolean automatorTakeScreenshotOnFailure = Whitebox.getInternalState( mojo, "parsedTakeScreenshotOnFailure" );
143         String automatorScreenshotsPathOnDevice = Whitebox.getInternalState( mojo, "parsedScreenshotsPathOnDevice" );
144 		Boolean automatorCreateReport = Whitebox.getInternalState( mojo, "parsedCreateReport" );
145 		String automatorReportSuffix = Whitebox.getInternalState( mojo, "parsedReportSuffix" );
146 		String automatorPropertiesKeyPrefix = Whitebox.getInternalState( mojo, "parsedPropertiesKeyPrefix" );
147 
148         assertFalse( "UIAutomator skip parameter should be false", automatorSkip );
149         assertFalse( "UIAutomator debug parameter should be false", automatorDebug );
150         assertNotNull( "UIAutomator jarFile parameter should be not null", automatorJarFile );
151         String expectedJarFile = buildName + ".jar";
152         assertEquals( "UIAutomator jarFile parameter should be equal to artifact name", expectedJarFile,
153                 automatorJarFile );
154 
155         assertNotNull( "UIAutomator jarFile parameter should be not null", automatorTestClassOrMethods );
156         String[] expectedTestClassOrMethods = new String[]
157         { "a", "b#c" };
158         assertTrue( "UIAutomator testClassOrMethods parameter should be equal [a,b#c]",
159                 Arrays.equals( expectedTestClassOrMethods, automatorTestClassOrMethods ) );
160         assertTrue( "UIAutomator takeScreenshotOnFailure parameter should be equal true",
161                 automatorTakeScreenshotOnFailure );
162         assertEquals( "UIAutomator screenshotsPath on device be equal /mnt/sdcard/screenshots/",
163                 "/mnt/sdcard/screenshots/", automatorScreenshotsPathOnDevice );
164 		assertTrue( "UIAutomator createReport parameter should be true", automatorCreateReport );
165 		String expectedReportSuffix = "-mySpecialReport";
166 		assertEquals( "UIAutomator reportSuffix parameter should be equal to "+expectedReportSuffix,
167 				expectedReportSuffix, automatorReportSuffix );
168 		String expectedPropertiesKeyPrefix = "UIA";
169 		assertEquals( "UIAutomator propertiesKeyPrefix should be equal to " + expectedPropertiesKeyPrefix,
170 				expectedPropertiesKeyPrefix, automatorPropertiesKeyPrefix);
171     }
